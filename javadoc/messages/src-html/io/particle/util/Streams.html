001package io.particle.util;
002
003import java.io.InputStream;
004import java.io.OutputStream;
005import java.io.IOException;
006
007public class Streams {
008    public static int readUint8(InputStream in) {
009        int b;
010        try {
011            b = in.read();
012        } catch (IOException e) {
013            throw new RuntimeException("Failed to read from stream", e);
014        }
015        if (b < 0) {
016            throw new RuntimeException("Unexpected end of stream");
017        }
018        return b;
019    }
020
021    public static void writeUint8(OutputStream out, int val) {
022        try {
023            out.write(val);
024        } catch (IOException e) {
025            throw new RuntimeException("Failed to write to stream", e);
026        }
027    }
028
029    public static int readUint16Be(InputStream in) {
030        byte[] b = read(in, 2);
031        return ((int)(b[0] & 0xff) << 8) |
032                (int)(b[1] & 0xff);
033    }
034
035    public static void writeUint16Be(OutputStream out, int val) {
036        byte[] b = new byte[2];
037        b[0] = (byte)((val >>> 8) & 0xff);
038        b[1] = (byte)(val & 0xff);
039        write(out, b);
040    }
041
042    public static void writeUint32Be(OutputStream out, long val) {
043        byte[] b = new byte[4];
044        b[0] = (byte)((val >>> 24) & 0xff);
045        b[1] = (byte)((val >>> 16) & 0xff);
046        b[2] = (byte)((val >>> 8) & 0xff);
047        b[3] = (byte)(val & 0xff);
048        write(out, b);
049    }
050
051    public static byte[] read(InputStream in, int len) {
052        byte[] b = new byte[len];
053        int offs = 0;
054        while (offs < len) {
055            int r;
056            try {
057                r = in.read(b, offs, len - offs);
058            } catch (IOException e) {
059                throw new RuntimeException("Failed to read from stream", e);
060            }
061            if (r < 0) {
062                throw new RuntimeException("Unexpected end of stream");
063            }
064            offs += r;
065        }
066        return b;
067    }
068
069    public static void write(OutputStream out, byte[] data) {
070        try {
071            out.write(data);
072        } catch (IOException e) {
073            throw new RuntimeException("Failed to write to stream", e);
074        }
075    }
076}
