024/**
025 * An implementation of the Device OS control request protocol for BLE.
026 */
027public class BleRequestChannel {
028    /**
029     * The default maximum number of requests that can be sent to the device concurrently.
030     */
031    public static final int DEFAULT_MAX_CONCURRENT_REQUESTS = 1;
032
033    private static final String HASH_NAME = "SHA-256";
034    private static final String MAC_NAME = "HmacSHA256";
035    private static final byte[] CLIENT_ID = "client".getBytes();
036    private static final byte[] SERVER_ID = "server".getBytes();
037    private static final int NONCE_SIZE = 12;
038    private static final int TAG_SIZE = 8;
039
040    private static final int REQUEST_PACKET_OVERHEAD = TAG_SIZE + 8;
041    private static final int RESPONSE_PACKET_OVERHEAD = TAG_SIZE + 8;
042    private static final int HANDSHAKE_PACKET_OVERHEAD = 2;
043    private static final int MAX_PAYLOAD_SIZE = 65535;
044    private static final int MAX_REQUEST_ID = 65535;
045
046    /**
047     * The channel state.
048     */
049    public enum State {
050        /**
051         * The channel is ready to be opened.
052         */
053        NEW,
054        /**
055         * The channel is being opened.
056         */
057        OPENING,
058        /**
059         * The channel is open.
060         */
061        OPEN,
062        /**
063         * The channel is closed.
064         */
065        CLOSED
066    }
067
068    private static class Handshake {
069        enum State {
070            ROUND_1,
071            ROUND_2,
072            CONFIRM
073        }
074
075        EcJpake jpake;
076        MessageDigest cliHash;
077        MessageDigest servHash;
078        State state;
079        byte[] secret;
080    }
081
082    private static class Request {
083        int type;
084        byte[] data;
085    }
086
087    private BleRequestChannelCallback callback;
088    private SecureRandom rand;
089    private byte[] preSecret;
090    private int maxConcurReqs;
091
092    private HashSet<Integer> sentReqs;
093    private LinkedHashMap<Integer, Request> queuedReqs;
094    private CCMBlockCipher cipher;
095    private KeyParameter cipherKey;
096    private Handshake handshake;
097    private ByteBuffer buf;
098    private State state;
099    private byte[] cliNonce;
100    private byte[] servNonce;
101    private int lastCliCtr;
102    private int lastServCtr;
103    private int nextReqId;
104    private boolean reading;
105    private boolean sending;
106    private boolean closing;

118    /**
119     * Construct a channel.
120     *
121     * @param secret The pre-shared secret.
122     * @param callback The channel callbacks.
123     * @param maxConcurrentRequests The maximum number of requests that can be sent to the device
124     *        concurrently.
125     */
126    public BleRequestChannel(byte[] secret, BleRequestChannelCallback callback, int maxConcurrentRequests) {
127        this(secret, callback, maxConcurrentRequests, new SecureRandom());
128    }
129
130    /**
131     * Construct a channel.
132     *
133     * @param secret The pre-shared secret.
134     * @param callback The channel callbacks.
135     * @param maxConcurrentRequests The maximum number of requests that can be sent to the device
136     *        concurrently.
137     * @param random The secure random number generator to use.
138     */
139    public BleRequestChannel(byte[] secret, BleRequestChannelCallback callback, int maxConcurrentRequests,
140            SecureRandom random) {
141        if (secret == null || secret.length == 0) {
142            throw new IllegalArgumentException("Secret cannot be empty");
143        }
144        this.preSecret = secret;
145        if (callback == null) {
146            throw new IllegalArgumentException("Callback instance cannot be null");
147        }
148        this.callback = callback;
149        if (maxConcurrentRequests <= 0) {
150            throw new IllegalArgumentException("Invalid number of requests");
151        }
152        this.maxConcurReqs = maxConcurrentRequests;
153        if (random == null) {
154            throw new IllegalArgumentException("Random number generator instance cannot be null");
155        }
156        this.rand = random;
157        this.sentReqs = new HashSet<>();
158        this.queuedReqs = new LinkedHashMap<>();
159        this.buf = ByteBuffer.allocate(0);
160        this.buf.order(ByteOrder.LITTLE_ENDIAN);
161        this.lastCliCtr = 0;
162        this.lastServCtr = 0;
163        this.nextReqId = (int)((random.nextInt() & 0xffffffffl) % (MAX_REQUEST_ID + 1));
164        this.reading = false;
165        this.sending = false;
166        this.closing = false;
167        this.state = State.NEW;
168    }
169
170    /**
171     * Open the channel.
172     *
173     * The {@link BleRequestChannelCallback#onChannelOpen} callback will be invoked when the channel
174     * has been opened.
175     *
176     * It is the responsibility of the calling code to track the timeout of the operation.
177     */
178    public void open() {
179        if (this.state != State.NEW) {
180            throw new IllegalStateException("Invalid channel state");
181        }
182        try {
183            this.handshake = new Handshake();
184            this.handshake.state = Handshake.State.ROUND_1;
185            this.handshake.jpake = new EcJpake(EcJpake.Role.CLIENT, this.preSecret, this.rand);
186            this.preSecret = null;
187            try {
188                this.handshake.cliHash = MessageDigest.getInstance(HASH_NAME);
189                this.handshake.servHash = MessageDigest.getInstance(HASH_NAME);
190            } catch (NoSuchAlgorithmException e) {
191                throw new UnsupportedOperationException("Unsupported hash algorithm", e);
192            }
193            byte[] cliRound1 = this.handshake.jpake.getRound1();
194            this.handshake.cliHash.update(cliRound1);
195            this.handshake.servHash.update(cliRound1);
196            this.state = State.OPENING;
197            this.writeHandshake(cliRound1);
198        } catch (Exception e) {
199            this.closeWithError(new RequestError("Channel error", e));
200            throw e;
201        }
202    }
203
204    /**
205     * Close the channel.
206     *
207     * All pending requests will be cancelled with an error.
208     *
209     * A closed channel cannot be reopened.
210     */
211    public void close() {
212        if (this.state == State.CLOSED) {
213            return;
214        }
215        this.closeWithError(new RequestError("Channel closed"));
216    }

218    /**
219     * Read data received from the device.
220     *
221     * @param data The received data.
222     */
223    public void read(byte[] data) {
224        if (this.state != State.OPEN && this.state != State.OPENING) {
225            throw new IllegalStateException("Invalid channel state");
226        }
227        try {
228            this.appendToInputBuf(data);
229            if (this.reading) {
230                return;
231            }
232            this.reading = true;
233            for (;;) {
234                if (this.buf.remaining() < 2) {
235                    break;
236                }
237                this.buf.mark();
238                int payloadLen = this.buf.getShort() & 0xffff;
239                this.buf.reset();
240                boolean isOpen = (this.state == State.OPEN);
241                int packetLen = payloadLen + (isOpen ? RESPONSE_PACKET_OVERHEAD : HANDSHAKE_PACKET_OVERHEAD);
242                if (this.buf.remaining() < packetLen) {
243                    break;
244                }
245                ByteBuffer packet = this.buf.slice();
246                packet.limit(packetLen);
247                if (isOpen) {
248                    this.readResponse(packet);
249                } else {
250                    this.readHandshake(packet);
251                }
252                this.buf.position(this.buf.position() + packetLen);
253                this.buf.compact();
254                this.buf.limit(this.buf.position());
255                this.buf.rewind();
256            }
257            this.reading = false;
258        } catch (Exception e) {
259            this.closeWithError(new RequestError("Channel error", e));
260            throw e;
261        }
262    }

265     * Send a request to the device.

270    public int sendRequest(int type) {
271        return this.sendRequest(type, null);
272    }

274    /**
275     * Send a request to the device.
276     *
277     * It is the responsibility of the calling code to track the timeout of the operation.
278     * If necessary, the request can be cancelled at any time via {@link #cancelRequest}.
279     *
280     * @param type The request type.
281     * @param type The payload data.
282     * @return The request ID.
283     */
284    public int sendRequest(int type, byte[] data) {
285        if (data != null && data.length > MAX_PAYLOAD_SIZE) {
286            throw new IllegalArgumentException("Payload data is too long");
287        }
288        if (this.state != State.OPEN && this.state != State.OPENING) {
289            throw new IllegalStateException("Invalid channel state");
290        }
291        try {
292            int reqId = this.nextReqId;
293            if (++this.nextReqId > MAX_REQUEST_ID) {
294                this.nextReqId = 0;
295            }
296            Request req = new Request();
297            req.type = type;
298            req.data = (data != null) ? data : new byte[0];
299            this.queuedReqs.put(reqId, req);
300            this.sendNextRequest();
301            return reqId;
302        } catch (Exception e) {
303            this.closeWithError(new RequestError("Channel error", e));
304            throw e;
305        }
306    }
307
308    /**
309     * Cancel a request.
310     *
311     * @param id The request ID.
312     * @return `true` if the request was cancelled, otherwise `false`.
313     */
314    public boolean cancelRequest(int id) {
315        if (this.state != State.OPEN && this.state != State.OPENING) {
316            return false;
317        }
318        if (this.sentReqs.remove(id) || this.queuedReqs.remove(id) != null) {
319            this.sendNextRequest();
320            return true;
321        }
322        return false;
323    }

325    /**
326     * Get the channel state.
327     *
328     * @return The channel state.
329     */
330    public State state() {
331        return this.state;
332    }

334    private void sendNextRequest() {
335        if (this.state != State.OPEN || this.sending) {
336            return;
337        }
338        this.sending = true;
339        while (!this.queuedReqs.isEmpty() && this.sentReqs.size() < this.maxConcurReqs) {
340            Iterator<Map.Entry<Integer, Request>> it = this.queuedReqs.entrySet().iterator();
341            Map.Entry<Integer, Request> entry = it.next();
342            int reqId = entry.getKey();
343            Request req = entry.getValue();
344            it.remove();
345            this.sentReqs.add(reqId);
346            this.writeRequest(reqId, req.type, req.data);
347        }
348        this.sending = false;
349    }
350
351    private void readResponse(ByteBuffer packet) {
352        byte[] aad = new byte[2];
353        packet.get(aad);
354        byte[] data = new byte[packet.remaining()];
355        packet.get(data);
356        byte[] nonce = genNonce(this.servNonce, ++this.lastServCtr, true /* isResp */);
357        data = this.runCipher(false /* encrypt */, data, nonce, aad);
358        ByteBuffer b = ByteBuffer.wrap(data);
359        b.order(ByteOrder.LITTLE_ENDIAN);
360        int reqId = b.getShort() & 0xffff;
361        if (this.sentReqs.contains(reqId)) {
362            int result = b.getInt();
363            data = new byte[b.remaining()];
364            b.get(data);
365            this.sentReqs.remove(reqId);
366            this.callback.onRequestResponse(reqId, result, data);
367            this.sendNextRequest();
368        }
369    }
370
371    private void writeRequest(int id, int type, byte[] data) {
372        ByteBuffer b = ByteBuffer.allocate(data.length + REQUEST_PACKET_OVERHEAD);
373        b.order(ByteOrder.LITTLE_ENDIAN);
374        b.putShort((short)(data.length & 0xffff));
375        b.mark();
376        b.putShort((short)(id & 0xffff));
377        b.putShort((short)(type & 0xffff));
378        b.putShort((short)0); // Reserved
379        b.put(data);
380        data = b.array();
381        byte[] plain = Arrays.copyOfRange(data, 2, data.length - TAG_SIZE);
382        byte[] aad = Arrays.copyOfRange(data, 0, 2);
383        byte[] nonce = genNonce(this.cliNonce, ++this.lastCliCtr, false /* isResp */);
384        data = this.runCipher(true /* encrypt */, plain, nonce, aad);
385        b.reset();
386        b.put(data);
387        this.callback.onChannelWrite(b.array());
388    }

390    private void readHandshake(ByteBuffer packet) {
391        switch (this.handshake.state) {
392        case ROUND_1: {
393            packet.getShort(); // Skip header
394            byte[] servRound1 = new byte[packet.remaining()];
395            packet.get(servRound1);
396            this.handshake.jpake.readRound1(servRound1);
397            this.handshake.cliHash.update(servRound1);
398            this.handshake.servHash.update(servRound1);
399            this.handshake.state = Handshake.State.ROUND_2;
400            break;
401        }
402        case ROUND_2: {
403            packet.getShort();
404            byte[] servRound2 = new byte[packet.remaining()];
405            packet.get(servRound2);
406            this.handshake.jpake.readRound2(servRound2);
407            this.handshake.cliHash.update(servRound2);
408            this.handshake.servHash.update(servRound2);
409            byte[] cliRound2 = this.handshake.jpake.getRound2();
410            this.writeHandshake(cliRound2);
411            this.handshake.cliHash.update(cliRound2);
412            this.handshake.servHash.update(cliRound2);
413            this.handshake.secret = this.handshake.jpake.deriveSecret();
414            byte[] cliConfirm = genConfirm(this.handshake.secret, CLIENT_ID, SERVER_ID, this.handshake.cliHash.digest());
415            this.writeHandshake(cliConfirm);
416            this.handshake.servHash.update(cliConfirm);
417            this.handshake.state = Handshake.State.CONFIRM;
418            break;
419        }
420        case CONFIRM: {
421            packet.getShort();
422            byte[] servConfirm = new byte[packet.remaining()];
423            packet.get(servConfirm);
424            byte[] expectedConfirm = genConfirm(this.handshake.secret, SERVER_ID, CLIENT_ID, this.handshake.servHash.digest());
425            if (!Arrays.equals(servConfirm, expectedConfirm)) {
426                throw new RequestChannelError("Key confirmation failed");
427            }
428            this.cipher = new CCMBlockCipher(new AESEngine());
429            this.cipherKey = new KeyParameter(Arrays.copyOfRange(this.handshake.secret, 0, 16));
430            this.cliNonce = Arrays.copyOfRange(this.handshake.secret, 16, 24);
431            this.servNonce = Arrays.copyOfRange(this.handshake.secret, 24, 32);
432            this.handshake = null;
433            this.state = State.OPEN;
434            this.callback.onChannelOpen();
435            this.sendNextRequest();
436            break;
437        }
438        default:
439            break;
440        }
441    }

443    private void writeHandshake(byte[] data) {
444        if (data.length > MAX_PAYLOAD_SIZE) {
445            throw new RequestChannelError("Handshake packet is too long"); // Internal error
446        }
447        ByteBuffer b = ByteBuffer.allocate(data.length + HANDSHAKE_PACKET_OVERHEAD);
448        b.order(ByteOrder.LITTLE_ENDIAN);
449        b.putShort((short)data.length);
450        b.put(data);
451        this.callback.onChannelWrite(b.array());
452    }
453
454    private byte[] runCipher(boolean encrypt, byte[] in, byte[] nonce, byte[] aad) {
455        AEADParameters param = new AEADParameters(this.cipherKey, TAG_SIZE * 8, nonce, aad);
456        this.cipher.init(encrypt, param);
457        byte[] out;
458        try {
459            out = this.cipher.processPacket(in, 0, in.length);
460        } catch (InvalidCipherTextException e) {
461            throw new RequestChannelError("Decryption error", e);
462        }
463        return out;
464    }

466    private void appendToInputBuf(byte[] data) {
467        if (this.buf.capacity() - this.buf.limit() < data.length) {
468            int minCapacity = Math.max(this.buf.remaining() + data.length, 128);
469            int newCapacity = Math.max(this.buf.capacity() * 3 / 2, minCapacity);
470            ByteBuffer newBuf = ByteBuffer.allocate(newCapacity);
471            newBuf.order(this.buf.order());
472            newBuf.put(this.buf);
473            newBuf.limit(newBuf.position());
474            newBuf.rewind();
475            this.buf = newBuf;
476        }
477        this.buf.mark();
478        this.buf.position(this.buf.limit());
479        this.buf.limit(this.buf.limit() + data.length);
480        this.buf.put(data);
481        this.buf.reset();
482    }

484    private void closeWithError(RequestError err) {
485        if (this.state == State.CLOSED || this.closing) {
486            return;
487        }
488        this.closing = true;
489        for (int reqId: this.sentReqs) {
490            this.callback.onRequestError(reqId, err);
491        }
492        this.sentReqs.clear();
493        for (int reqId: this.queuedReqs.keySet()) {
494            this.callback.onRequestError(reqId, err);
495        }
496        this.queuedReqs.clear();
497        this.closing = false;
498        this.state = State.CLOSED;
499    }

501    private static byte[] genNonce(byte[] fixed, int ctr, boolean isResp) {
502        ByteBuffer b = ByteBuffer.allocate(fixed.length + 4);
503        b.order(ByteOrder.LITTLE_ENDIAN);
504        if (isResp) {
505            ctr |= 0x80000000;
506        }
507        b.putInt(ctr);
508        b.put(fixed);
509        return b.array();
510    }
511
512    private static byte[] genConfirm(byte[] secret, byte[] ownId, byte[] peerId, byte[] packetsHash) {
513        MessageDigest keyHash = null;
514        try {
515            keyHash = MessageDigest.getInstance(HASH_NAME);
516        } catch (NoSuchAlgorithmException e) {
517            throw new UnsupportedOperationException("Unsupported hash algorithm", e);
518        }
519        keyHash.update(secret);
520        keyHash.update("JPAKE_KC".getBytes());
521        Mac mac = null;
522        try {
523            SecretKeySpec keySpec = new SecretKeySpec(keyHash.digest(), MAC_NAME);
524            mac = Mac.getInstance(MAC_NAME);
525            mac.init(keySpec);
526        } catch (Exception e) {
527            throw new UnsupportedOperationException("Unsupported MAC algorithm", e);
528        }
529        mac.update("KC_1_U".getBytes());
530        mac.update(ownId);
531        mac.update(peerId);
532        mac.update(packetsHash);
533        return mac.doFinal();
534    }
