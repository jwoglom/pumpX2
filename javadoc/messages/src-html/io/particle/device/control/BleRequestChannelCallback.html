001package io.particle.device.control;
002
003/**
004 * An interface used to implement {@link BleRequestChannel} callbacks.
005 */
006public interface BleRequestChannelCallback {
007    /**
008     * The callback invoked when the channel is opened.
009     */
010    void onChannelOpen();
011    /**
012     * The callback invoked to send data to the device.
013     *
014     * @param data The packet data.
015     */
016    void onChannelWrite(byte[] data);
017    /**
018     * The callback invoked when a response is received.
019     *
020     * @param requestId The request ID.
021     * @param result The response result.
022     * @param data The response data.
023     */
024    void onRequestResponse(int requestId, int result, byte[] data);
025    /**
026     * The callback invoked when a request error occurs.
027     *
028     * @param requestId The request ID.
029     * @param error The request error.
030     */
031    void onRequestError(int requestId, RequestError error);
032}
