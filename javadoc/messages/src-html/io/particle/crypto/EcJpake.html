024import org.bouncycastle.jce.ECNamedCurveTable;
025import org.bouncycastle.jce.provider.BouncyCastleProvider;

027import org.bouncycastle.math.ec.ECPoint;

030import java.security.MessageDigest;
031import java.security.SecureRandom;
032import java.security.NoSuchAlgorithmException;
033import java.math.BigInteger;

040/**
041 * A class implementing the EC-JPAKE protocol as defined by the Thread specification.
042 */
043public class EcJpake {
044    /**
045     * Peer role.
046     */
047    public enum Role {
048        /**
049         * Client.
050         */
051        CLIENT,
052        /**
053         * Server.
054         */
055        SERVER
056    }
057
058    private class KeyPair {
059        public BigInteger priv;
060        public ECPoint pub;
061    }
062
063    private BigInteger xm1;
064    private ECPoint Xm1;
065    private BigInteger xm2;
066    private ECPoint Xm2;
067    private ECPoint Xp1;
068    private ECPoint Xp2;
069    private ECPoint Xp;
070    private BigInteger s;
071
072    private boolean hasPeerRound1;
073    private boolean hasPeerRound2;
074    private byte[] myRound1;
075    private byte[] myRound2;
076    private byte[] derivedSecret;
077
078    private Role role;
079    private byte[] myId;
080    private byte[] peerId;
081    private ECParameterSpec ec;
082    private MessageDigest hash;
083    private SecureRandom rand;
084
085    private static final String CURVE_NAME = "P-256";
086    private static final int CURVE_ID = 23; // RFC 4492, 5.1.1
087    private static final String HASH_NAME = "SHA256";
088    private static final byte[] CLIENT_ID = "client".getBytes();
089    private static final byte[] SERVER_ID = "server".getBytes();
090    private static final boolean ENCODED_COMPRESSED = false;
091
092    static {
093        // Remove builtin Android BouncyCastle provider and replace with newer version
094        if (Security.getProvider(BouncyCastleProvider.PROVIDER_NAME) != null) {
095            Security.removeProvider(BouncyCastleProvider.PROVIDER_NAME);
096        }
097        Security.addProvider(new BouncyCastleProvider());
098    }
099
100    /**
101     * Constructor.
102     *
103     * @param role The role of this peer.
104     * @param secret The pre-shared secret.
105     */
106    public EcJpake(Role role, byte[] secret) {
107        this(role, secret, new SecureRandom());
108    }
109
110    /**
111     * Constructor.
112     *
113     * @param role The role of this peer.
114     * @param secret The pre-shared secret.
115     * @param random The secure random number generator to use.
116     */
117    public EcJpake(Role role, byte[] secret, SecureRandom random) {
118        this.ec = ECNamedCurveTable.getParameterSpec(CURVE_NAME);
119        if (this.ec == null) {
120            throw new UnsupportedOperationException("Unsupported curve type");
121        }
122        try {
123            this.hash = MessageDigest.getInstance(HASH_NAME);
124        } catch (NoSuchAlgorithmException e) {
125            throw new UnsupportedOperationException("Unsupported hash algorithm", e);
126        }
127        this.rand = random;
128        this.role = role;
129        if (this.role == Role.CLIENT) {
130            this.myId = CLIENT_ID;
131            this.peerId = SERVER_ID;
132        } else {
133            this.myId = SERVER_ID;
134            this.peerId = CLIENT_ID;
135        }
136        this.s = new BigInteger(1, secret);
137        this.xm1 = null;
138        this.Xm1 = null;
139        this.xm2 = null;
140        this.Xm2 = null;
141        this.Xp1 = null;
142        this.Xp2 = null;
143        this.Xp = null;
144        this.hasPeerRound1 = false;
145        this.hasPeerRound2 = false;
146        this.myRound1 = null;
147        this.myRound2 = null;
148        this.derivedSecret = null;
149    }
150
151    /**
152     * Generate a message for the first round of the protocol.
153     *
154     * @return The message data.
155     */
156    public byte[] getRound1() {
157        if (this.myRound1 == null) {
158            ByteArrayOutputStream out = new ByteArrayOutputStream();
159            KeyPair kp = this.genKeyPair(this.ec.getG());
160            this.xm1 = kp.priv;
161            this.Xm1 = kp.pub;
162            this.writePoint(out, this.Xm1);
163            this.writeZkp(out, this.ec.getG(), this.xm1, this.Xm1, this.myId);
164            kp = this.genKeyPair(this.ec.getG());
165            this.xm2 = kp.priv;
166            this.Xm2 = kp.pub;
167            this.writePoint(out, this.Xm2);
168            this.writeZkp(out, this.ec.getG(), this.xm2, this.Xm2, this.myId);
169            this.myRound1 = out.toByteArray();
170        }
171        return this.myRound1;
172    }
173
174    /**
175     * Read a message generated by the peer for the first round of the protocol.
176     *
177     * @param data The message data.
178     * @return The number of bytes read.
179     */
180    public int readRound1(byte[] data) {
181        if (this.hasPeerRound1) {
182            throw new IllegalStateException("Invalid protocol state");
183        }
184        ByteArrayInputStream in = new ByteArrayInputStream(data);
185        this.Xp1 = this.readPoint(in);
186        this.readZkp(in, this.ec.getG(), this.Xp1, this.peerId);
187        this.Xp2 = this.readPoint(in);
188        this.readZkp(in, this.ec.getG(), this.Xp2, this.peerId);
189        this.hasPeerRound1 = true;
190        return data.length - in.available();
191    }
192
193    /**
194     * Generate a message for the second round of the protocol.
195     *
196     * @return The message data.
197     */
198    public byte[] getRound2() {
199        if (this.myRound2 == null) {
200            if (!this.hasPeerRound1 || this.myRound1 == null) {
201                throw new IllegalStateException("Invalid protocol state");
202            }
203            ByteArrayOutputStream out = new ByteArrayOutputStream();
204            ECPoint G = this.Xp1.add(this.Xp2).add(this.Xm1);
205            BigInteger xm = this.mulSecret(this.xm2, this.s, false /* negate */);
206            ECPoint Xm = G.multiply(xm);
207            if (this.role == Role.SERVER) {
208                this.writeCurveId(out);
209            }
210            this.writePoint(out, Xm);
211            this.writeZkp(out, G, xm, Xm, this.myId);
212            this.myRound2 = out.toByteArray();
213        }
214        return this.myRound2;
215    }
216
217    /**
218     * Read a message generated by the peer for the second round of the protocol.
219     *
220     * @param data The message data.
221     * @return The number of bytes read.
222     */
223    public int readRound2(byte[] data) {
224        if (!this.hasPeerRound1 || this.myRound1 == null || this.hasPeerRound2) {
225            throw new IllegalStateException("Invalid protocol state");
226        }
227        ByteArrayInputStream in = new ByteArrayInputStream(data);
228        if (this.role == Role.CLIENT) {
229            this.readCurveId(in);
230        }
231        ECPoint G = this.Xm1.add(this.Xm2).add(this.Xp1);
232        this.Xp = this.readPoint(in);
233        this.readZkp(in, G, this.Xp, this.peerId);
234        this.hasPeerRound2 = true;
235        return data.length - in.available();
236    }
237
238    /**
239     * Derive a shared secret.
240     *
241     * @return The shared secret.
242     */
243    public byte[] deriveSecret() {
244        if (this.derivedSecret == null) {
245            if (!this.hasPeerRound2) {
246                throw new IllegalStateException("Invalid protocol state");
247            }
248            BigInteger xm2s = this.mulSecret(this.xm2, this.s, true /* negate */);
249            ECPoint K = this.Xp.add(this.Xp2.multiply(xm2s)).multiply(this.xm2);
250            this.derivedSecret = this.hash.digest(BigIntegers.asUnsignedByteArray(K.normalize().getXCoord().toBigInteger()));
251        }
252        return this.derivedSecret;
253    }
254
255    private void readZkp(InputStream in, ECPoint G, ECPoint X, byte[] id) {
256        ECPoint V = this.readPoint(in);
257        BigInteger r = this.readNum(in);
258        BigInteger h = this.zkpHash(G, V, X, id);
259        ECPoint VV = G.multiply(r).add(X.multiply(h.mod(this.ec.getN())));
260        if (!VV.equals(V)) {
261            throw new RuntimeException("Validation failed");
262        }
263    }
264
265    private void writeZkp(OutputStream out, ECPoint G, BigInteger x, ECPoint X, byte[] id) {
266        KeyPair kp = this.genKeyPair(G);
267        BigInteger v = kp.priv;
268        ECPoint V = kp.pub;
269        BigInteger h = this.zkpHash(G, V, X, id);
270        BigInteger r = v.subtract(x.multiply(h)).mod(this.ec.getN());
271        this.writePoint(out, V);
272        this.writeNum(out, r);
273    }
274
275    private BigInteger zkpHash(ECPoint G, ECPoint V, ECPoint X, byte[] id) {
276        ByteArrayOutputStream out = new ByteArrayOutputStream();
277        this.writeZkpHashPoint(out, G);
278        this.writeZkpHashPoint(out, V);
279        this.writeZkpHashPoint(out, X);
280        Streams.writeUint32Be(out, id.length);
281        Streams.write(out, id);
282        byte[] hash = this.hash.digest(out.toByteArray());
283        BigInteger h = new BigInteger(1, hash);
284        return h.mod(this.ec.getN());
285    }
286
287    private void writeZkpHashPoint(OutputStream out, ECPoint point) {
288        byte[] encoded = point.getEncoded(ENCODED_COMPRESSED /* compressed */);
289        Streams.writeUint32Be(out, encoded.length);
290        Streams.write(out, encoded);
291    }
292
293    private BigInteger mulSecret(BigInteger X, BigInteger S, boolean negate) {
294        BigInteger b = new BigInteger(1, this.randBytes(16));
295        b = b.multiply(this.ec.getN()).add(S);
296        BigInteger R = X.multiply(b);
297        if (negate) {
298            R = R.negate();
299        }
300        return R.mod(this.ec.getN());
301    }
302
303    private ECPoint readPoint(InputStream in) {
304        int len = Streams.readUint8(in);
305        byte[] encoded = Streams.read(in, len);
306        return this.ec.getCurve().decodePoint(encoded);
307    }
308
309    private void writePoint(OutputStream out, ECPoint point) {
310        byte[] encoded = point.getEncoded(ENCODED_COMPRESSED /* compressed */);
311        if (encoded.length > 255) {
312            throw new RuntimeException("Encoded ECPoint is too long");
313        }
314        Streams.writeUint8(out, encoded.length);
315        Streams.write(out, encoded);
316    }
317
318    private BigInteger readNum(InputStream in) {
319        int len = Streams.readUint8(in);
320        byte[] encoded = Streams.read(in, len);
321        return new BigInteger(1, encoded);
322    }
323
324    private void writeNum(OutputStream out, BigInteger val) {
325        byte[] encoded = BigIntegers.asUnsignedByteArray(val);
326        if (encoded.length > 255) {
327            throw new RuntimeException("Encoded BigInteger is too long");
328        }
329        Streams.writeUint8(out, encoded.length);
330        Streams.write(out, encoded);
331    }
332
333    private void readCurveId(InputStream in) {
334        int type = Streams.readUint8(in);
335        if (type != 3) { // ECCurveType.named_curve
336            throw new RuntimeException("Invalid message");
337        }
338        int id = Streams.readUint16Be(in);
339        if (id != CURVE_ID) {
340            throw new RuntimeException("Unexpected curve type");
341        }
342    }
343
344    private void writeCurveId(OutputStream out) {
345        Streams.writeUint8(out, 3); // ECCurveType.named_curve
346        Streams.writeUint16Be(out, CURVE_ID);
347    }
348
349    private KeyPair genKeyPair(ECPoint G) {
350        KeyPair kp = new KeyPair();
351        kp.priv = BigIntegers.createRandomInRange(BigInteger.ONE, this.ec.getN().subtract(BigInteger.ONE), this.rand);
352        kp.pub = G.multiply(kp.priv);
353        return kp;
354    }
355
356    private byte[] randBytes(int bytes) {
357        byte[] b = new byte[bytes];
358        this.rand.nextBytes(b);
359        return b;
360    }
