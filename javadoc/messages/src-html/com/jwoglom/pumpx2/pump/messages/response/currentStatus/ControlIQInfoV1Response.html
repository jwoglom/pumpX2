001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.ControlIQInfoV1Request;
009
010@MessageProps(
011    opCode=105,
012    size=10,
013    type=MessageType.RESPONSE,
014    request= ControlIQInfoV1Request.class
015)
016public class ControlIQInfoV1Response extends ControlIQInfoAbstractResponse {
017    
018    private boolean closedLoopEnabled;
019    private int weight;
020    private int weightUnit;
021    private int totalDailyInsulin;
022    private int currentUserModeType;
023    private int byte6;
024    private int byte7;
025    private int byte8;
026    private int controlStateType;
027    
028    public ControlIQInfoV1Response() {}
029    
030    public ControlIQInfoV1Response(boolean closedLoopEnabled, int weight, int weightUnit, int totalDailyInsulin, int currentUserModeType, int byte6, int byte7, int byte8, int controlStateType) {
031        this.cargo = buildCargo(closedLoopEnabled, weight, weightUnit, totalDailyInsulin, currentUserModeType, byte6, byte7, byte8, controlStateType);
032        this.closedLoopEnabled = closedLoopEnabled;
033        this.weight = weight;
034        this.weightUnit = weightUnit;
035        this.totalDailyInsulin = totalDailyInsulin;
036        this.currentUserModeType = currentUserModeType;
037        this.byte6 = byte6;
038        this.byte7 = byte7;
039        this.byte8 = byte8;
040        this.controlStateType = controlStateType;
041        
042    }
043
044    public void parse(byte[] raw) {
045        Preconditions.checkArgument(raw.length == props().size());
046        this.cargo = raw;
047        this.closedLoopEnabled = raw[0] != 0;
048        this.weight = Bytes.readShort(raw, 1);
049        this.weightUnit = raw[3];
050        this.totalDailyInsulin = raw[4];
051        this.currentUserModeType = raw[5];
052        this.byte6 = raw[6];
053        this.byte7 = raw[7];
054        this.byte8 = raw[8];
055        this.controlStateType = raw[9];
056        
057    }
058
059    
060    public static byte[] buildCargo(boolean closedLoopEnabled, int weight, int weightUnit, int totalDailyInsulin, int currentUserModeType, int byte6, int byte7, int byte8, int controlStateType) {
061        return Bytes.combine(
062            new byte[]{ (byte) (closedLoopEnabled ? 1 : 0) },
063            Bytes.firstTwoBytesLittleEndian(weight), 
064            new byte[]{ (byte) weightUnit }, 
065            new byte[]{ (byte) totalDailyInsulin }, 
066            new byte[]{ (byte) currentUserModeType }, 
067            new byte[]{ (byte) byte6 }, 
068            new byte[]{ (byte) byte7 }, 
069            new byte[]{ (byte) byte8 }, 
070            new byte[]{ (byte) controlStateType });
071    }
072    
073    public boolean getClosedLoopEnabled() {
074        return closedLoopEnabled;
075    }
076    public int getWeight() {
077        return weight;
078    }
079    public int getWeightUnitId() {
080        return weightUnit;
081    }
082    public int getTotalDailyInsulin() {
083        return totalDailyInsulin;
084    }
085    public int getCurrentUserModeTypeId() {
086        return currentUserModeType;
087    }
088    public int getByte6() {
089        return byte6;
090    }
091    public int getByte7() {
092        return byte7;
093    }
094    public int getByte8() {
095        return byte8;
096    }
097    public int getControlStateType() {
098        return controlStateType;
099    }
100    
101}
