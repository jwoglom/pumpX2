001package com.jwoglom.pumpx2.pump.messages.request.control;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
010import com.jwoglom.pumpx2.pump.messages.response.control.CancelBolusResponse;
011
012/**
013 * also known as BolusTerminationRequest.
014 * Can be used on boluses which are initiated from the phone as well as the pump.
015 * Works for extended boluses.
016 */
017@MessageProps(
018    opCode=-96,
019    size=4, // 28 with 24 byte padding
020    type=MessageType.REQUEST,
021    characteristic=Characteristic.CONTROL,
022    response=CancelBolusResponse.class,
023    signed=true,
024    minApi=KnownApiVersion.API_V2_5
025)
026public class CancelBolusRequest extends Message { 
027    private int bolusId;
028    
029    public CancelBolusRequest() {}
030
031    public CancelBolusRequest(int bolusId) {
032        this.cargo = buildCargo(bolusId);
033        this.bolusId = bolusId;
034        
035    }
036
037    public void parse(byte[] raw) {
038        raw = removeSignedRequestHmacBytes(raw);
039        Preconditions.checkArgument(raw.length == props().size());
040        this.cargo = raw;
041        this.bolusId = Bytes.readShort(raw, 0);
042        
043    }
044    
045    public static byte[] buildCargo(int bolusId) {
046        return Bytes.combine(
047            Bytes.firstTwoBytesLittleEndian(bolusId),
048            new byte[]{0, 0}
049        );
050    }
051
052    public int getBolusId() {
053        return bolusId;
054    }
055}
