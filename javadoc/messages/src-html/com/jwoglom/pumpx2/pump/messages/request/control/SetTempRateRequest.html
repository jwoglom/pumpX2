001package com.jwoglom.pumpx2.pump.messages.request.control;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
010import com.jwoglom.pumpx2.pump.messages.models.SupportedDevices;
011import com.jwoglom.pumpx2.pump.messages.response.control.SetTempRateResponse;
012
013@MessageProps(
014    opCode=-92,
015    size=6, // 30 with signed
016    type=MessageType.REQUEST,
017    characteristic=Characteristic.CONTROL,
018    signed=true,
019    modifiesInsulinDelivery=true,
020    minApi=KnownApiVersion.MOBI_API_V3_5,
021    supportedDevices=SupportedDevices.MOBI_ONLY,
022    response=SetTempRateResponse.class
023)
024public class SetTempRateRequest extends Message {
025
026    private int minutes;
027    private int percent;
028    
029    public SetTempRateRequest() {}
030
031    public SetTempRateRequest(byte[] raw) {
032        parse(raw);
033    }
034
035    public SetTempRateRequest(int minutes, int percent) {
036        Preconditions.checkArgument(minutes >= 15 && minutes <= 72*60, "duration of temp rate must be between 15 and 4,320 minutes (72 hours)");
037        Preconditions.checkArgument(percent >= 0 && percent <= 250, "percent temp rate must be between 0-250%");
038        this.cargo = buildCargo(minutes, percent);
039        this.minutes = minutes;
040        this.percent = percent;
041    }
042
043    public void parse(byte[] raw) {
044        raw = this.removeSignedRequestHmacBytes(raw);
045        Preconditions.checkArgument(raw.length == props().size());
046        this.cargo = raw;
047        this.minutes = Math.toIntExact(Bytes.readUint32(raw, 0) / 1000 / 60);
048        this.percent = Bytes.readShort(raw, 4);
049    }
050
051    
052    public static byte[] buildCargo(int minutes, int percent) {
053        return Bytes.combine(
054                Bytes.toUint32(minutes * 1000L * 60L),
055                Bytes.firstTwoBytesLittleEndian(percent)
056        );
057    }
058
059
060    public int getMinutes() {
061        return minutes;
062    }
063
064    public int getPercent() {
065        return percent;
066    }
067}
