115        return Bytes.combine(
116            new byte[]{24, 1}, // 280 across 2 bytes
117            Bytes.toUint32(pumpTimeSec),
118            Bytes.toUint32(sequenceNum),
119            Bytes.firstTwoBytesLittleEndian(bolusID), 
120            new byte[]{ (byte) bolusDeliveryStatus }, 
121            new byte[]{ (byte) bolusType }, 
122            new byte[]{ (byte) bolusSource }, 
123            new byte[]{ (byte) reserved },
124            Bytes.firstTwoBytesLittleEndian(requestedNow),
125            Bytes.firstTwoBytesLittleEndian(requestedLater), 
126            Bytes.firstTwoBytesLittleEndian(correction), 
127            Bytes.firstTwoBytesLittleEndian(extendedDurationRequested), 
128            Bytes.firstTwoBytesLittleEndian(deliveredTotal));
129    }

192    public enum BolusType {
193        FOOD1(1), // used when there is carbs
194        CORRECTION(2), // used when there is a correction amount
195        EXTENDED(4),
196        FOOD2(8), // used when there is no carbs and just a distinct bolus amount
197        ;
198        private int mask;
199        BolusType(int mask) {
200            this.mask = mask;
201        }
202        public int mask() {
203            return mask;
204        }
205        public static Set<BolusType> fromBitmask(int bitmask) {
206            Set<BolusType> ret = new HashSet<>();
207            for (BolusType b : values()) {
208                if ((bitmask & b.mask()) != 0) {
209                    ret.add(b);
210                }
211            }
212            return ret;
213        }
214        public static int toBitmask(BolusType ...types) {
215            int bitmask = 0;
216            for (BolusType b : types) {
217                bitmask |= b.mask();
218            }
219            return bitmask;
220        }
