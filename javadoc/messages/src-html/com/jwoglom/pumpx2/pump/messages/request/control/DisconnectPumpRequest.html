003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;

008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;

011
012/*
013 * I don't actually think calling this command is required... since an app can just disconnect
014 * from the device over Bluetooth and forget the pairing key, and the user needs to re-trigger
015 * a pair in order to connect again anyway (either via menu on tslim X2 or entering "pairing mode"
016 * on the Mobi)
017 */
018@MessageProps(
019    opCode=-66,
020    size=0,
021    type=MessageType.REQUEST,
022    characteristic=Characteristic.CONTROL,
023    signed=true,
024    response=DisconnectPumpResponse.class,
025    minApi=KnownApiVersion.MOBI_API_V3_5
026)
027public class DisconnectPumpRequest extends Message { 
028    public DisconnectPumpRequest() {
029        this.cargo = EMPTY;
030    }
031
032    public void parse(byte[] raw) { 
033        raw = this.removeSignedRequestHmacBytes(raw);
034        Preconditions.checkArgument(raw.length == props().size());
035        this.cargo = raw;
036        
037    }
038
039    
040}
