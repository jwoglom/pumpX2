001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.PumpGlobalsRequest;
009
010@MessageProps(
011    opCode=87,
012    size=14,
013    type=MessageType.RESPONSE,
014    request=PumpGlobalsRequest.class
015)
016public class PumpGlobalsResponse extends Message {
017    
018    private int quickBolusEnabled;
019    private int quickBolusIncrementUnits;
020    private int quickBolusIncrementCarbs;
021    private int quickBolusEntryType;
022    private int quickBolusStatus;
023    private int buttonAnnun;
024    private int quickBolusAnnun;
025    private int bolusAnnun;
026    private int reminderAnnun;
027    private int alertAnnun;
028    private int alarmAnnun;
029    private int fillTubingAnnun;
030    
031    public PumpGlobalsResponse() {}
032    
033    public PumpGlobalsResponse(int quickBolusEnabled, int quickBolusIncrementUnits, int quickBolusIncrementCarbs, int quickBolusEntryType, int quickBolusStatus, int buttonAnnun, int quickBolusAnnun, int bolusAnnun, int reminderAnnun, int alertAnnun, int alarmAnnun, int fillTubingAnnun) {
034        this.cargo = buildCargo(quickBolusEnabled, quickBolusIncrementUnits, quickBolusIncrementCarbs, quickBolusEntryType, quickBolusStatus, buttonAnnun, quickBolusAnnun, bolusAnnun, reminderAnnun, alertAnnun, alarmAnnun, fillTubingAnnun);
035        this.quickBolusEnabled = quickBolusEnabled;
036        this.quickBolusIncrementUnits = quickBolusIncrementUnits;
037        this.quickBolusIncrementCarbs = quickBolusIncrementCarbs;
038        this.quickBolusEntryType = quickBolusEntryType;
039        this.quickBolusStatus = quickBolusStatus;
040        this.buttonAnnun = buttonAnnun;
041        this.quickBolusAnnun = quickBolusAnnun;
042        this.bolusAnnun = bolusAnnun;
043        this.reminderAnnun = reminderAnnun;
044        this.alertAnnun = alertAnnun;
045        this.alarmAnnun = alarmAnnun;
046        this.fillTubingAnnun = fillTubingAnnun;
047        
048    }
049
050    public void parse(byte[] raw) {
051        Preconditions.checkArgument(raw.length == props().size());
052        this.cargo = raw;
053        this.quickBolusEnabled = raw[0];
054        this.quickBolusIncrementUnits = Bytes.readShort(raw, 1);
055        this.quickBolusIncrementCarbs = Bytes.readShort(raw, 3);
056        this.quickBolusEntryType = raw[5];
057        this.quickBolusStatus = raw[6];
058        this.buttonAnnun = raw[7];
059        this.quickBolusAnnun = raw[8];
060        this.bolusAnnun = raw[9];
061        this.reminderAnnun = raw[10];
062        this.alertAnnun = raw[11];
063        this.alarmAnnun = raw[12];
064        this.fillTubingAnnun = raw[13];
065        
066    }
067
068    
069    public static byte[] buildCargo(int quickBolusEnabled, int quickBolusIncrementUnits, int quickBolusIncrementCarbs, int quickBolusEntryType, int quickBolusStatus, int buttonAnnun, int quickBolusAnnun, int bolusAnnun, int reminderAnnun, int alertAnnun, int alarmAnnun, int fillTubingAnnun) {
070        return Bytes.combine(
071            new byte[]{ (byte) quickBolusEnabled }, 
072            Bytes.firstTwoBytesLittleEndian(quickBolusIncrementUnits), 
073            Bytes.firstTwoBytesLittleEndian(quickBolusIncrementCarbs), 
074            new byte[]{ (byte) quickBolusEntryType }, 
075            new byte[]{ (byte) quickBolusStatus }, 
076            new byte[]{ (byte) buttonAnnun }, 
077            new byte[]{ (byte) quickBolusAnnun }, 
078            new byte[]{ (byte) bolusAnnun }, 
079            new byte[]{ (byte) reminderAnnun }, 
080            new byte[]{ (byte) alertAnnun }, 
081            new byte[]{ (byte) alarmAnnun }, 
082            new byte[]{ (byte) fillTubingAnnun });
083    }
084    
085    public int getQuickBolusEnabled() {
086        return quickBolusEnabled;
087    }
088    public int getQuickBolusIncrementUnits() {
089        return quickBolusIncrementUnits;
090    }
091    public int getQuickBolusIncrementCarbs() {
092        return quickBolusIncrementCarbs;
093    }
094    public int getQuickBolusEntryType() {
095        return quickBolusEntryType;
096    }
097    public int getQuickBolusStatus() {
098        return quickBolusStatus;
099    }
100    public AnnunciationEnum getButtonAnnun() {
101        return AnnunciationEnum.fromId(buttonAnnun);
102    }
103    public AnnunciationEnum getQuickBolusAnnun() {
104        return AnnunciationEnum.fromId(quickBolusAnnun);
105    }
106    public AnnunciationEnum getBolusAnnun() {
107        return AnnunciationEnum.fromId(bolusAnnun);
108    }
109    public AnnunciationEnum getReminderAnnun() {
110        return AnnunciationEnum.fromId(reminderAnnun);
111    }
112    public AnnunciationEnum getAlertAnnun() {
113        return AnnunciationEnum.fromId(alertAnnun);
114    }
115    public AnnunciationEnum getAlarmAnnun() {
116        return AnnunciationEnum.fromId(alarmAnnun);
117    }
118    public AnnunciationEnum getFillTubingAnnun() {
119        return AnnunciationEnum.fromId(fillTubingAnnun);
120    }
121
122    public enum AnnunciationEnum {
123        AUDIO_HIGH(0),
124        AUDIO_MEDIUM(1),
125        AUDIO_LOW(2),
126        VIBRATE(3),
127
128        ;
129
130        private final int id;
131        AnnunciationEnum(int id) {
132            this.id = id;
133        }
134
135        public int id() {
136            return id;
137        }
138
139        public static AnnunciationEnum fromId(int id) {
140            for (AnnunciationEnum a : AnnunciationEnum.values()) {
141                if (a.id() == id) {
142                    return a;
143                }
144            }
145            return null;
146        }
147    }
148    
149}
