001package com.jwoglom.pumpx2.pump.messages.request.authentication;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
008import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
009import com.jwoglom.pumpx2.pump.messages.response.authentication.PumpChallengeResponse;
010
011import java.util.Arrays;
012
013import kotlin.collections.ArraysKt;
014
015/**
016 * The second authorization message sent to the pump which contains the HMACed pairing code.
017 *
018 * When the Android libary is used, this message is invoked automatically by PumpX2 on Bluetooth
019 * connection initialization as part of performing the initial pump pairing process.
020 */
021@MessageProps(
022    opCode=18,
023    size=22,
024    type=MessageType.REQUEST,
025    characteristic=Characteristic.AUTHORIZATION,
026    response=PumpChallengeResponse.class
027)
028public class PumpChallengeRequest extends Message {
029    private int appInstanceId;
030    private byte[] pumpChallengeHash;
031
032    public PumpChallengeRequest() {}
033
034    public PumpChallengeRequest(int appInstanceId, byte[] pumpChallengeHash) {
035        parse(buildCargo(appInstanceId, pumpChallengeHash));
036        Preconditions.checkState(this.appInstanceId == appInstanceId);
037        Preconditions.checkState(Arrays.equals(this.pumpChallengeHash, pumpChallengeHash));
038    }
039
040    public int getAppInstanceId() {
041        return appInstanceId;
042    }
043
044    public byte[] getPumpChallengeHash() {
045        return pumpChallengeHash;
046    }
047
048    private static byte[] buildCargo(int appInstanceId, byte[] pumpChallengeHash) {
049        byte[] cargo = new byte[22];
050        System.arraycopy(Bytes.combine(Bytes.firstTwoBytesLittleEndian(appInstanceId), pumpChallengeHash), 0, cargo, 0, 22);
051
052        return cargo;
053    }
054
055    public void parse(byte[] raw) {
056        this.cargo = raw;
057        this.appInstanceId = Bytes.readShort(Arrays.copyOfRange(raw, 0, 2), 0);
058        this.pumpChallengeHash = Arrays.copyOfRange(raw, 2, raw.length);
059    }
060}
