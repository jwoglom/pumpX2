001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.jwoglom.pumpx2.pump.messages.Message;
004
005public abstract class ControlIQInfoAbstractResponse extends Message {
006    abstract boolean getClosedLoopEnabled();
007    abstract int getWeight();
008    abstract int getWeightUnitId();
009    abstract int getTotalDailyInsulin();
010    abstract int getCurrentUserModeTypeId();
011    abstract int getByte6();
012    abstract int getByte7();
013    abstract int getByte8();
014    abstract int getControlStateType();
015
016    public UserModeType getCurrentUserModeType() {
017        return UserModeType.fromId(getCurrentUserModeTypeId());
018    }
019
020    public enum UserModeType {
021        STANDARD(0),
022        SLEEP(1),
023        EXERCISE(2),
024
025        // NOT SUPPORTED IN CURRENT FIRMWARE (referenced in Tandem Source event schema)
026        NOT_SUPPORTED_IN_CURRENT_FIRMWARE__EATING_SOON(3),
027        ;
028
029        private final int id;
030        UserModeType(int id) {
031            this.id = id;
032        }
033
034        public int id() {
035            return id;
036        }
037
038        public static UserModeType fromId(int id) {
039            for (UserModeType t : values()) {
040                if (t.id == id) {
041                    return t;
042                }
043            }
044            return null;
045        }
046    }
047
048    public WeightUnit getWeightUnit() {
049        return WeightUnit.fromId(getWeightUnitId());
050    }
051
052    public enum WeightUnit {
053        KILOGRAMS(0),
054        POUNDS(1),
055        ;
056
057        private final int id;
058        WeightUnit(int id) {
059            this.id = id;
060        }
061
062        public int id() {
063            return id;
064        }
065
066        public static WeightUnit fromId(int id) {
067            for (WeightUnit t: values()) {
068                if (t.id == id) {
069                    return t;
070                }
071            }
072            return null;
073        }
074    }
075}
