060    public void parse(byte[] raw) {
061        Preconditions.checkArgument(raw.length == props().size());
062        this.cargo = raw;
063        this.isUnacked = raw[0] != 0;
064        this.correctionFactor = Bytes.readShort(raw, 1);
065        this.iob = Bytes.readUint32(raw, 3);
066        this.cartridgeRemainingInsulin = Bytes.readShort(raw, 7);
067        this.targetBg = Bytes.readShort(raw, 9);
068        this.isf = Bytes.readShort(raw, 11);
069        this.carbEntryEnabled = raw[13] != 0;
070        this.carbRatio = Bytes.readUint32(raw, 14);
071        this.maxBolusAmount = Bytes.readShort(raw, 18);
072        this.maxBolusHourlyTotal = Bytes.readUint32(raw, 20);
073        this.maxBolusEventsExceeded = raw[24] != 0;
074        this.maxIobEventsExceeded = raw[25] != 0;
075        this.isAutopopAllowed = raw[37] != 0;
076        
077    }

080    public static byte[] buildCargo(boolean isUnacked, int correctionFactor, long iob, int cartridgeRemainingInsulin, int targetBg, int isf, boolean carbEntryEnabled, long carbRatio, int maxBolusAmount, long maxBolusHourlyTotal, boolean maxBolusEventsExceeded, boolean maxIobEventsExceeded, boolean isAutopopAllowed, byte[] unknown11bytes, byte[] unknown8bytes) {
081        return Bytes.combine(
082            new byte[]{ (byte) (isUnacked ? 1 : 0) }, 
083            Bytes.firstTwoBytesLittleEndian(correctionFactor), 
084            Bytes.toUint32(iob), 
085            Bytes.firstTwoBytesLittleEndian(cartridgeRemainingInsulin), 
086            Bytes.firstTwoBytesLittleEndian(targetBg), 
087            Bytes.firstTwoBytesLittleEndian(isf), 
088            new byte[]{ (byte) (carbEntryEnabled ? 1 : 0) }, 
089            Bytes.toUint32(carbRatio), 
090            Bytes.firstTwoBytesLittleEndian(maxBolusAmount), 
091            Bytes.toUint32(maxBolusHourlyTotal), 
092            new byte[]{ (byte) (maxBolusEventsExceeded ? 1 : 0) }, 
093            new byte[]{ (byte) (maxIobEventsExceeded ? 1 : 0) },
094            unknown11bytes, // this contains unknown values
095            new byte[]{ (byte) (isAutopopAllowed ? 1 : 0) },
096            unknown8bytes); // this contains unknown values
097    }
