010import java.util.UUID;
011
012/**
013 * Enum version of CharacteristicUUID
014 *
015 * TODO: refactor all uses to this class
016 */
017public enum Characteristic {
018    CURRENT_STATUS(CURRENT_STATUS_CHARACTERISTICS),
019    HISTORY_LOG(HISTORY_LOG_CHARACTERISTICS),
020    AUTHORIZATION(AUTHORIZATION_CHARACTERISTICS),
021    CONTROL(CONTROL_CHARACTERISTICS),
022    CONTROL_STREAM(CONTROL_STREAM_CHARACTERISTICS),
023    ;
024
025    private final UUID uuid;
026    Characteristic(UUID uuid) {
027        this.uuid = uuid;
028    }
029
030    public static Characteristic of(UUID uuid) {
031        for (Characteristic c : values()) {
032            if (c.uuid.toString().equals(uuid.toString())) {
033                return c;
034            }
035        }
036        return null;
037    }
038
039    public UUID getUuid() {
040        return uuid;
041    }
042}
