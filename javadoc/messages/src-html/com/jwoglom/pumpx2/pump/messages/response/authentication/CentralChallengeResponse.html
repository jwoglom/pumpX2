001package com.jwoglom.pumpx2.pump.messages.response.authentication;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.MessageType;
005import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
006import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.request.authentication.CentralChallengeRequest;
009
010import java.util.Arrays;
011
012@MessageProps(
013    opCode=17,
014    size=30,
015    type=MessageType.RESPONSE,
016    characteristic=Characteristic.AUTHORIZATION,
017    request=CentralChallengeRequest.class
018)
019public class CentralChallengeResponse extends AbstractCentralChallengeResponse {
020    private int appInstanceId;
021    private byte[] centralChallengeHash;
022    private byte[] hmacKey;
023
024    public CentralChallengeResponse() {}
025
026    public CentralChallengeResponse(int appInstanceId, byte[] centralChallengeHash, byte[] hmacKey) {
027        parse(buildCargo(appInstanceId, centralChallengeHash, hmacKey));
028        Preconditions.checkState(this.appInstanceId == appInstanceId);
029        Preconditions.checkState(Arrays.equals(this.centralChallengeHash, centralChallengeHash));
030        Preconditions.checkState(Arrays.equals(this.hmacKey, hmacKey));
031    }
032
033    public void parse(byte[] raw) {
034        Preconditions.checkArgument(raw.length == props().size());
035        this.cargo = raw;
036        appInstanceId = Bytes.readShort(raw, 0);
037        centralChallengeHash = Arrays.copyOfRange(raw, 2, 22); // len=10 == Request.centralChallenge
038        hmacKey = Arrays.copyOfRange(raw, 22, 30); // len=8
039    }
040
041    public static byte[] buildCargo(int byte0short, byte[] bytes2to22, byte[] bytes22to30) {
042        return Bytes.combine(Bytes.firstTwoBytesLittleEndian(byte0short), Bytes.combine(bytes2to22, bytes22to30));
043    }
044
045    public int getAppInstanceId() {
046        return appInstanceId;
047    }
048
049    public byte[] getCentralChallengeHash() {
050        return centralChallengeHash;
051    }
052
053    public byte[] getHmacKey() {
054        return hmacKey;
055    }
056}
