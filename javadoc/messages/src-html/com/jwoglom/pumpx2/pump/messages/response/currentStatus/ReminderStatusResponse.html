014@MessageProps(
015    opCode=73,

020public class ReminderStatusResponse extends Message {
021    
022    private BigInteger intMap;
023    
024    public ReminderStatusResponse() {}
025    
026    public ReminderStatusResponse(BigInteger reminderBitmask) {
027        this.cargo = buildCargo(reminderBitmask);
028        this.intMap = reminderBitmask;
029        
030    }
031
032    public void parse(byte[] raw) {
033        Preconditions.checkArgument(raw.length == props().size());
034        this.cargo = raw;
035        this.intMap = Bytes.readUint64(raw, 0);
036        
037    }
038
039    
040    public static byte[] buildCargo(BigInteger reminderBitmask) {
041        return Bytes.combine(
042            Bytes.toUint64(reminderBitmask.longValue()));
043    }
044    
045    public BigInteger getIntMap() {
046        return intMap;
047    }
048    public long getBitMap() {
049        return intMap.longValue();
050    }
051
052
053    public Set<ReminderType> getReminders() {
054        return ReminderType.fromBitmask(getIntMap());
055    }

124        private final int id;
125        ReminderType(int id) {
126            this.id = id;
127        }
128
129        public int id() {
130            return id;
131        }
132
133        public static Set<ReminderType> fromBitmask(BigInteger bitmask) {
134            Set<ReminderType> set = new HashSet<>();
135            for (ReminderType a : values()) {
136                if (bitmask.testBit(a.id())) {
137                    set.add(a);
138                }
139            }
140            return set;
141        }
142
143        public static BigInteger toBitmask(ReminderType...alerts) {
144            BigInteger i = BigInteger.ZERO;
145            for (ReminderType a : alerts) {
146                i = i.setBit(a.id());
147            }
148            return i;
149        }
