001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007import java.math.BigInteger;
008
009@HistoryLogProps(
010    opCode = 2,
011    displayName = "Temp Basal Rate Activated",
012    usedByTidepool = true
013)
014public class TempRateActivatedHistoryLog extends HistoryLog {
015    
016    private float percent;
017    private float duration;
018    private int tempRateId;
019    
020    public TempRateActivatedHistoryLog() {}
021    public TempRateActivatedHistoryLog(long pumpTimeSec, long sequenceNum, float percent, float duration, int tempRateId) {
022        super(pumpTimeSec, sequenceNum);
023        this.cargo = buildCargo(pumpTimeSec, sequenceNum, percent, duration, tempRateId);
024        this.percent = percent;
025        this.duration = duration;
026        this.tempRateId = tempRateId;
027        
028    }
029
030    public TempRateActivatedHistoryLog(float percent, float duration, int tempRateId) {
031        this(0, 0, percent, duration, tempRateId);
032    }
033
034    public int typeId() {
035        return 2;
036    }
037
038    /**
039     * <table border="1">
040     * <caption>Parsed fields for TempRateActivatedHistoryLog:</caption>
041     * <tr><td><code>percent</code></td><td><code>Bytes.readFloat(raw, 10)</code></td></tr>
042     * <tr><td><code>duration</code></td><td><code>Bytes.readFloat(raw, 14)</code></td></tr>
043     * <tr><td><code>tempRateId</code></td><td><code>Bytes.readShort(raw, 20)</code></td></tr>
044     * </table>
045     */
046    public void parse(byte[] raw) {
047        Preconditions.checkArgument(raw.length == 26);
048        this.cargo = raw;
049        parseBase(raw);
050        this.percent = Bytes.readFloat(raw, 10);
051        this.duration = Bytes.readFloat(raw, 14);
052        this.tempRateId = Bytes.readShort(raw, 20);
053        
054    }
055
056    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, float percent, float duration, int tempRateId) {
057        return HistoryLog.fillCargo(Bytes.combine(
058            new byte[]{2, 0},
059            Bytes.toUint32(pumpTimeSec),
060            Bytes.toUint32(sequenceNum),
061            Bytes.toFloat(percent), 
062            Bytes.toFloat(duration), 
063            Bytes.firstTwoBytesLittleEndian(tempRateId)));
064    }
065    public float getPercent() {
066        return percent;
067    }
068    public float getDuration() {
069        return duration;
070    }
071    public int getTempRateId() {
072        return tempRateId;
073    }
074    
075}
