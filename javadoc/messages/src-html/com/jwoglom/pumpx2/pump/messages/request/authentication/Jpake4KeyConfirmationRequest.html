001package com.jwoglom.pumpx2.pump.messages.request.authentication;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
008import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
009import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
010import com.jwoglom.pumpx2.pump.messages.response.authentication.Jpake4KeyConfirmationResponse;
011
012import java.util.Arrays;
013
014/**
015 * The fifth authorization message sent to the pump with V2 authorization style.
016 *
017 * When the Android libary is used, this message is invoked automatically by PumpX2 on Bluetooth
018 * connection initialization as part of performing the initial pump pairing process.
019 */
020@MessageProps(
021    opCode=40,
022    size=50,
023    type=MessageType.REQUEST,
024    minApi=KnownApiVersion.API_V3_2,
025    characteristic=Characteristic.AUTHORIZATION,
026    response=Jpake4KeyConfirmationResponse.class
027)
028public class Jpake4KeyConfirmationRequest extends Message {
029    private int appInstanceId;
030    private byte[] hashDigest;
031    private byte[] reserved;
032    private byte[] nonce;
033
034    public static byte[] RESERVED = new byte[]{0, 0, 0, 0, 0, 0, 0, 0};
035
036    public Jpake4KeyConfirmationRequest() {}
037
038
039    public Jpake4KeyConfirmationRequest(int appInstanceId, byte[] nonce, byte[] reserved, byte[] hashDigest) {
040        this.cargo = buildCargo(appInstanceId, nonce, reserved, hashDigest);
041        this.appInstanceId = appInstanceId;
042        this.nonce = nonce;
043        this.reserved = reserved;
044        this.hashDigest = hashDigest;
045    }
046
047    public Jpake4KeyConfirmationRequest(byte[] rawCargo) {
048        parse(rawCargo);
049    }
050
051    public int getAppInstanceId() {
052        return appInstanceId;
053    }
054
055    public byte[] getNonce() {
056        return nonce;
057    }
058
059    public byte[] getHashDigest() {
060        return hashDigest;
061    }
062
063    public byte[] getReserved() {
064        return reserved;
065    }
066
067    private static byte[] buildCargo(int appInstanceId, byte[] nonce, byte[] reserved, byte[] hashDigest) {
068        Preconditions.checkArgument(nonce.length == 8, "nonce was " + nonce.length + " not 8");
069        Preconditions.checkArgument(reserved.length == 8);
070        Preconditions.checkArgument(hashDigest.length == 32, "hashDigest was " + hashDigest.length + " not 32");
071        byte[] cargo = new byte[50];
072        System.arraycopy(Bytes.combine(
073                Bytes.firstTwoBytesLittleEndian(appInstanceId),
074                nonce,
075                reserved,
076                hashDigest
077        ), 0, cargo, 0, 50);
078
079        return cargo;
080    }
081
082    public void parse(byte[] raw) {
083        this.cargo = raw;
084        this.appInstanceId = Bytes.readShort(Arrays.copyOfRange(raw, 0, 2), 0);
085        this.nonce = Arrays.copyOfRange(raw, 2, 10);
086        this.reserved = Arrays.copyOfRange(raw, 10, 18);
087        this.hashDigest = Arrays.copyOfRange(raw, 18, 50); // 32
088    }
089}
