001package com.jwoglom.pumpx2.pump.messages.builders.crypto;
002
003import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
004
005import java.security.InvalidKeyException;
006import java.security.Key;
007import java.security.NoSuchAlgorithmException;
008import java.util.Arrays;
009
010import javax.crypto.Mac;
011import javax.crypto.spec.SecretKeySpec;
012
013public class Hkdf {
014    public static byte[] build(byte[] nonce, byte[] keyMaterial) {
015        return expandKey(updateMacWithKeyMaterial(nonce, keyMaterial), new byte[0], 32);
016    }
017
018    static byte[] expandKey(byte[] keyMaterial, byte[] nonce, int keyLength) {
019        Mac mac = initHmacSha256(keyMaterial);
020        if (mac == null) {
021            return null;
022        }
023
024        byte[] previousBlock = new byte[0];
025        byte[] resultKey = new byte[0];
026        int numBlocks = (int) Math.ceil(keyLength / 32.0);
027        for (int blockIndex = 0; blockIndex < numBlocks; blockIndex++) {
028            previousBlock = updateMac(Bytes.combine(previousBlock, nonce, decodeHex(Integer.toHexString(blockIndex + 1))), mac);
029            resultKey = Bytes.combine(resultKey, previousBlock);
030        }
031        return Arrays.copyOfRange(resultKey, 0, keyLength);
032    }
033
034    private static byte[] updateMacWithKeyMaterial(byte[] keyMaterial, byte[] data) {
035        return updateMac(data, initHmacSha256(keyMaterial));
036    }
037
038    private static Mac initHmacSha256(byte[] keyMaterial) {
039        Mac mac;
040        try {
041            mac = Mac.getInstance("hmacSHA256");
042            mac.init(newSecretKeySpec(keyMaterial));
043        } catch (NoSuchAlgorithmException | InvalidKeyException e) {
044            return null;
045        }
046        return mac;
047    }
048
049    private static Key newSecretKeySpec(byte[] keyMaterial) {
050        if (keyMaterial.length == 0) {
051            keyMaterial = new byte[32];
052        }
053        return new SecretKeySpec(keyMaterial, "HmacSHA256");
054    }
055
056    private static byte[] updateMac(byte[] bArr, Mac mac) {
057        mac.update(bArr);
058        byte[] doFinal = mac.doFinal();
059        mac.reset();
060        return doFinal;
061    }
062
063    private static byte[] decodeHex(String str) {
064        if (str.length() % 2 == 1) {
065            str = "0" + str;
066        }
067
068        int length = str.length() / 2;
069        byte[] out = new byte[length];
070        for (int i = 0; i < length; i++) {
071            out[i] = (byte) Integer.parseInt(str.substring(i*2, (i*2) + 2), 16);
072        }
073        return out;
074    }
075}
