001package com.jwoglom.pumpx2.pump.messages.request.authentication;
002
003import com.jwoglom.pumpx2.pump.messages.MessageType;
004import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
005import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
006import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
007import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
008import com.jwoglom.pumpx2.pump.messages.response.authentication.Jpake1aResponse;
009
010import java.util.Arrays;
011
012/**
013 * The first message sent on connection to a Tandem pump which begins the authorization process
014 * using the 6-character pairing code -- the first half of the first JPAKE authorization client request.
015 *
016 * When the Android libary is used, this message is invoked automatically by PumpX2 on Bluetooth
017 * connection initialization as part of performing the initial pump pairing process.
018 */
019@MessageProps(
020    opCode=32,
021    size=167,
022    type=MessageType.REQUEST,
023    minApi=KnownApiVersion.API_V3_2,
024    characteristic=Characteristic.AUTHORIZATION,
025    response=Jpake1aResponse.class
026)
027public class Jpake1aRequest extends AbstractCentralChallengeRequest {
028    private int appInstanceId;
029    private byte[] centralChallenge;
030
031    public Jpake1aRequest() {}
032
033    public Jpake1aRequest(int appInstanceId, byte[] centralChallenge) {
034        this.cargo = buildCargo(appInstanceId, centralChallenge);
035        this.appInstanceId = appInstanceId;
036        this.centralChallenge = centralChallenge; // 165
037    }
038
039    public Jpake1aRequest(byte[] rawCargo) {
040        parse(rawCargo);
041    }
042
043    public int getAppInstanceId() {
044        return appInstanceId;
045    }
046
047    public byte[] getCentralChallenge() {
048        return centralChallenge;
049    }
050
051    private static byte[] buildCargo(int appInstanceId, byte[] centralChallenge) {
052        byte[] cargo = new byte[167];
053        System.arraycopy(Bytes.combine(Bytes.firstTwoBytesLittleEndian(appInstanceId), centralChallenge), 0, cargo, 0, 167);
054
055        return cargo;
056    }
057
058    public void parse(byte[] raw) {
059        this.cargo = raw;
060        this.appInstanceId = Bytes.readShort(Arrays.copyOfRange(raw, 0, 2), 0);
061        this.centralChallenge = Arrays.copyOfRange(raw, 2, raw.length);
062    }
063}
