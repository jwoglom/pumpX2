001package com.jwoglom.pumpx2.pump.messages.calculator;
002
003import com.google.common.collect.ImmutableList;
004import com.jwoglom.pumpx2.pump.messages.response.currentStatus.BolusCalcDataSnapshotResponse;
005import com.jwoglom.pumpx2.pump.messages.response.currentStatus.LastBGResponse;
006
007import java.util.ArrayList;
008import java.util.List;
009import java.util.Optional;
010import java.util.Set;
011
012/**
013 * A builder object for filling data into {@link BolusCalculator} when response messages are received
014 * from the pump.
015 */
016public class BolusCalculatorBuilder {
017    // arguments
018    private BolusCalcDataSnapshotResponse dataSnapshotResponse;
019    private LastBGResponse lastBGResponse;
020
021    private BolusCalcDataSnapshot dataSnapshot;
022    private BolusCalcLastBG lastBG;
023    private BolusParameters userInputParameters = new BolusParameters();
024
025    /**
026     * When no messages have been received
027     */
028    public BolusCalculatorBuilder() {
029    }
030
031    /**
032     * When a BolusCalcDataSnapshotResponse and LastBGResponse have been received
033     */
034    public BolusCalculatorBuilder(BolusCalcDataSnapshotResponse dataSnapshotResponse, LastBGResponse lastBGResponse) {
035        this.dataSnapshotResponse = dataSnapshotResponse;
036        this.lastBGResponse = lastBGResponse;
037        this.dataSnapshot = new BolusCalcDataSnapshot(dataSnapshotResponse);
038        this.lastBG = new BolusCalcLastBG(dataSnapshotResponse, lastBGResponse);
039    }
040
041    /**
042     * When a BolusCalcDataSnapshotResponse is received
043     */
044    public void onBolusCalcDataSnapshotResponse(BolusCalcDataSnapshotResponse resp) {
045        dataSnapshotResponse = resp;
046        dataSnapshot = new BolusCalcDataSnapshot(dataSnapshotResponse);
047        if (lastBG == null && lastBGResponse != null) {
048            lastBG = new BolusCalcLastBG(dataSnapshotResponse, lastBGResponse);
049        }
050    }
051
052    /**
053     * When a LastBGResponse is received
054     */
055    public void onLastBGResponse(LastBGResponse resp) {
056        lastBGResponse = resp;
057        if (dataSnapshotResponse != null) {
058            lastBG = new BolusCalcLastBG(dataSnapshotResponse, lastBGResponse);
059        }
060    }
061
062    /**
063     * @return the set BG if filled by the user, otherwise the filled-in BG if able
064     */
065    public Optional<Integer> getGlucoseMgdl() {
066        if (userInputParameters.getGlucoseMgdl() != null) {
067            return Optional.of(userInputParameters.getGlucoseMgdl());
068        }
069
070        if (lastBG == null) {
071            return Optional.empty();
072        }
073
074        return Optional.ofNullable(lastBG.determineLastBG().getLeft());
075    }
076
077    /**
078     * @param glucoseMgdl the BG value in mg/dL set by the user
079     */
080    public void setGlucoseMgdl(Integer glucoseMgdl) {
081        userInputParameters = userInputParameters.withGlucoseMgdl(glucoseMgdl);
082    }
083
084    /**
085     * @return the set number of carbs in grams if filled by the user
086     */
087    public Optional<Integer> getCarbsValueGrams() {
088        return Optional.ofNullable(userInputParameters.getCarbsGrams());
089    }
090
091    /**
092     * @param carbsGrams the number of carbs in grams set by the user
093     */
094    public Optional<BolusCalcCondition> setCarbsValueGrams(Integer carbsGrams) {
095        if (dataSnapshot != null && !dataSnapshot.carbEntryEnabled) {
096            return Optional.of(new BolusCalcCondition.FailedPrecondition("carb entry disabled"));
097        }
098
099        userInputParameters = userInputParameters.withCarbsGrams(carbsGrams);
100        return Optional.empty();
101    }
102
103    /**
104     * @return a {@link BolusCalcUnits} object representing the total insulin units calculated
105     */
106    public BolusCalcUnits getInsulinUnits() {
107        return new BolusCalculator(userInputParameters, dataSnapshot, lastBG).getUnits();
108    }
109
110    /**
111     * @return a set of {@link BolusCalcCondition} objects describing why the bolus calculator
112     * decision was made.
113     */
114    public Set<BolusCalcCondition> getConditions() {
115        return new BolusCalculator(userInputParameters, dataSnapshot, lastBG).getConditions();
116    }
117
118    /**
119     * @param insulinUnits the override number of units of insulin set by the user
120     */
121    public void setInsulinUnits(Double insulinUnits) {
122        userInputParameters = userInputParameters.withUnits(insulinUnits);
123    }
124
125    /**
126     * @return a {@link BolusCalculator} which can be used to get units and conditions
127     */
128    public BolusCalculator build() {
129        return new BolusCalculator(userInputParameters, dataSnapshot, lastBG);
130    }
131
132    public String toString() {
133        return "BolusCalculatorBuilder(userInputParameters="+userInputParameters+")";
134    }
135}
