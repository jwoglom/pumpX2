001package com.jwoglom.pumpx2.pump.messages;
002
003import com.google.common.base.Preconditions;
004import com.google.common.collect.ImmutableSet;
005import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
006import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
009import com.jwoglom.pumpx2.pump.messages.models.SupportedDevices;
010import com.jwoglom.pumpx2.shared.Hex;
011import com.jwoglom.pumpx2.shared.JavaHelpers;
012
013import org.json.JSONObject;
014
015import java.util.Set;
016
017public abstract class Message {
018    public static final byte[] EMPTY = new byte[]{};
019    protected byte[] cargo = null;
020
021    public abstract void parse(byte[] raw);
022
023    public MessageProps props() {
024        return getClass().getAnnotation(MessageProps.class);
025    }
026
027    public byte opCode() {
028        return props().opCode();
029    }
030
031    public boolean signed() {
032        return props().signed();
033    }
034
035    public boolean stream() {
036        return props().stream();
037    }
038
039    public MessageType type() {
040        return props().type();
041    }
042
043    public Class<? extends Message> getResponseClass() {
044        Preconditions.checkState(type() == MessageType.REQUEST);
045        return props().response();
046    }
047
048    public MessageProps getResponseProps() {
049        Preconditions.checkState(type() == MessageType.REQUEST);
050        return props().response().getAnnotation(MessageProps.class);
051    }
052
053    public byte getResponseOpCode() {
054        if (type() == MessageType.REQUEST) {
055            return getResponseProps().opCode();
056        }
057        return opCode();
058    }
059
060    public int getResponseSize() {
061        if (type() == MessageType.REQUEST) {
062            return getResponseProps().size();
063        }
064        return props().size();
065    }
066
067    public Class<? extends Message> getRequestClass() {
068        Preconditions.checkState(type() == MessageType.RESPONSE);
069        return props().request();
070    }
071
072    public MessageProps getRequestProps() {
073        Preconditions.checkState(type() == MessageType.RESPONSE);
074        return props().request().getAnnotation(MessageProps.class);
075    }
076
077    public Characteristic getCharacteristic() {
078        return props().characteristic();
079    }
080
081    public byte[] getCargo() {
082        return this.cargo;
083    }
084
085    public void fillWithEmptyCargo() {
086        this.cargo = EMPTY;
087    }
088
089    protected byte[] removeSignedRequestHmacBytes(byte[] raw) {
090        if (signed() && raw.length == props().size() + 24) {
091            return Bytes.dropLastN(raw, 24);
092        }
093        return raw;
094    }
095
096    private static final Set<String> IGNORED_PROPERTY_NAMES = ImmutableSet.of("requestClass", "requestProps", "historyLogStreamBytes", "intMap");
097
098    public String toString() {
099        return JavaHelpers.autoToString(this, IGNORED_PROPERTY_NAMES);
100    }
101
102    public String verboseToString() {
103        return JavaHelpers.autoToStringVerbose(this, IGNORED_PROPERTY_NAMES);
104    }
105
106    public String messageName() {
107        return getClass().getName().replace("com.jwoglom.pumpx2.pump.messages.", "");
108    }
109
110    public String jsonToString() {
111        String generatedToString = JavaHelpers.autoToStringJson(this, IGNORED_PROPERTY_NAMES);
112        JSONObject obj = new JSONObject();
113        obj.put("name", messageName());
114        obj.put("messageProps", new JSONObject(propsJsonToString()));
115        obj.put("params", new JSONObject(generatedToString));
116        obj.put("cargoHex", Hex.encodeHexString(getCargo()));
117
118        return obj.toString(0);
119    }
120
121    private String propsJsonToString() {
122        JSONObject props = new JSONObject();
123        props.putOpt("opCode", props().opCode());
124        props.putOpt("size", props().size());
125        props.putOpt("variableSize", props().variableSize());
126        props.putOpt("stream", props().stream());
127        props.putOpt("signed", props().signed());
128        props.putOpt("type", props().type());
129        if (!props().response().equals(UndefinedMessage.class)) {
130            props.putOpt("responseName", props().response().getName().replace("com.jwoglom.pumpx2.pump.messages.", ""));
131            props.putOpt("responseOpCode", props().response().getAnnotation(MessageProps.class).opCode());
132        }
133        if (!props().request().equals(UndefinedMessage.class)) {
134            props.putOpt("requestName", props().request().getName().replace("com.jwoglom.pumpx2.pump.messages.", ""));
135            props.putOpt("requestOpCode", props().request().getAnnotation(MessageProps.class).opCode());
136        }
137        props.putOpt("minApi", props().minApi());
138        props.putOpt("supportedDevices", props().supportedDevices());
139        props.putOpt("characteristicUuid", props().characteristic().getUuid());
140        props.putOpt("characteristic", props().characteristic().name());
141        props.putOpt("modifiesInsulinDelivery", props().modifiesInsulinDelivery());
142        return props.toString(0);
143    }
144}
