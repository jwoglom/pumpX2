001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.helpers.Dates;
009import com.jwoglom.pumpx2.pump.messages.request.currentStatus.LastBGRequest;
010
011import java.time.Instant;
012
013/**
014 * This message ONLY returns BGs entered via the Bolus Calculator, and NOT the most recent CGM
015 * reading.
016 */
017@MessageProps(
018    opCode=51,
019    size=7,
020    type=MessageType.RESPONSE,
021    request=LastBGRequest.class
022)
023public class LastBGResponse extends Message {
024    
025    private long bgTimestamp;
026    private int bgValue;
027    private int bgSourceId;
028    
029    public LastBGResponse() {}
030    
031    public LastBGResponse(long bgTimestamp, int bgValue, int bgSourceId) {
032        this.cargo = buildCargo(bgTimestamp, bgValue, bgSourceId);
033        this.bgTimestamp = bgTimestamp;
034        this.bgValue = bgValue;
035        this.bgSourceId = bgSourceId;
036        
037    }
038
039    public void parse(byte[] raw) {
040        Preconditions.checkArgument(raw.length == props().size());
041        this.cargo = raw;
042        this.bgTimestamp = Bytes.readUint32(raw, 0);
043        this.bgValue = Bytes.readShort(raw, 4);
044        this.bgSourceId = raw[6];
045        
046    }
047
048    
049    public static byte[] buildCargo(long bgTimestamp, int bgValue, int bgSource) {
050        return Bytes.combine(
051            Bytes.toUint32(bgTimestamp), 
052            Bytes.firstTwoBytesLittleEndian(bgValue), 
053            new byte[]{ (byte) bgSource });
054    }
055    
056    public long getBgTimestamp() {
057        return bgTimestamp;
058    }
059
060    public Instant getBgTimestampInstant() {
061        return Dates.fromJan12008EpochSecondsToDate(bgTimestamp);
062    }
063
064    public int getBgValue() {
065        return bgValue;
066    }
067    public int getBgSourceId() {
068        return bgSourceId;
069    }
070    public BgSource getBgSource() {
071        return BgSource.fromId(bgSourceId);
072    }
073
074    public enum BgSource {
075        // TODO: confirm
076        MANUAL(0),
077        CGM(1),
078        ;
079
080        private final int id;
081        BgSource(int id) {
082            this.id = id;
083        }
084
085        public int getId() {
086            return id;
087        }
088
089        public static BgSource fromId(int id) {
090            for (BgSource b : values()) {
091                if (b.id == id) {
092                    return b;
093                }
094            }
095            return null;
096        }
097    }
098    
099}
