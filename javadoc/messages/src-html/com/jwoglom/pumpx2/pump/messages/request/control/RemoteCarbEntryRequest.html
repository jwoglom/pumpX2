001package com.jwoglom.pumpx2.pump.messages.request.control;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.helpers.Dates;
010import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
011import com.jwoglom.pumpx2.pump.messages.response.control.BolusPermissionResponse;
012import com.jwoglom.pumpx2.pump.messages.response.control.RemoteCarbEntryResponse;
013import com.jwoglom.pumpx2.pump.messages.response.currentStatus.TimeSinceResetResponse;
014
015/**
016 * Saves the entered carbs as a history log entry which populates it in the pump's graph.
017 * If not called, then the bolus appears in the mobile app / t:connect without the carb amount.
018 *
019 * pumpTime is the ACTUAL pump time seconds since boot, and can be returned from a TimeSinceResetResponse
020 * via {@link TimeSinceResetResponse#getPumpTimeSecondsSinceReset()}
021 *
022 * This should be called before {@link InitiateBolusRequest}
023 */
024@MessageProps(
025    opCode=-14,
026    size=9, // 33 with 24 byte padding
027    type=MessageType.REQUEST,
028    characteristic=Characteristic.CONTROL,
029    response=RemoteCarbEntryResponse.class,
030    signed=true,
031    minApi=KnownApiVersion.API_V2_5
032)
033public class RemoteCarbEntryRequest extends Message {
034    private int carbs;
035    private int unknown;
036    private long pumpTime;
037    private int bolusId;
038
039    public RemoteCarbEntryRequest() {}
040
041    /**
042     * Creates a request to affix information on the given number of carbs (in grams) to the
043     * bolus ID (which must be in progress; in a state between calling {@link BolusPermissionRequest}
044     * and {@link InitiateBolusRequest})
045     * @param carbs the number of carbs in grams
046     * @param pumpTimeSecondsSinceBoot the output of {@link TimeSinceResetResponse#getPumpTimeSecondsSinceReset()}
047     * @param bolusId the bolus ID returned from {@link BolusPermissionResponse#getBolusId()}
048     */
049    public RemoteCarbEntryRequest(int carbs, long pumpTimeSecondsSinceBoot, int bolusId) {
050        this(carbs, 1, pumpTimeSecondsSinceBoot, bolusId);
051    }
052
053    public RemoteCarbEntryRequest(int carbs, int unknown, long pumpTimeSecondsSinceBoot, int bolusId) {
054        Preconditions.checkArgument(pumpTimeSecondsSinceBoot < Dates.JANUARY_1_2008_UNIX_EPOCH, "pumpTimeSecondsSinceBoot ("+pumpTimeSecondsSinceBoot+") should be seconds since boot; a unix epoch was provided instead");
055        this.cargo = buildCargo(carbs, unknown, pumpTimeSecondsSinceBoot, bolusId);
056        this.carbs = carbs;
057        this.unknown = unknown; // from examples, always 1
058        this.pumpTime = pumpTimeSecondsSinceBoot;
059        this.bolusId = bolusId;
060    }
061
062    public static byte[] buildCargo(int carbs, int unknown, long pumpTimeSecondsSinceBoot, int bolusId) {
063        return Bytes.combine(
064                Bytes.firstTwoBytesLittleEndian(carbs),
065                new byte[]{(byte) unknown},
066                Bytes.toUint32(pumpTimeSecondsSinceBoot),
067                Bytes.firstTwoBytesLittleEndian(bolusId)
068        );
069    }
070
071    public void parse(byte[] raw) {
072        raw = removeSignedRequestHmacBytes(raw);
073        Preconditions.checkArgument(raw.length == props().size(), "size " + raw.length);
074        this.cargo = raw;
075        this.carbs = Bytes.readShort(raw, 0);
076        this.unknown = raw[2];
077        this.pumpTime = Bytes.readUint32(raw, 3);
078        this.bolusId = Bytes.readShort(raw, 7);
079    }
080
081    public int getCarbs() {
082        return carbs;
083    }
084
085    public long getPumpTime() {
086        return pumpTime;
087    }
088
089    public int getBolusId() {
090        return bolusId;
091    }
092}
