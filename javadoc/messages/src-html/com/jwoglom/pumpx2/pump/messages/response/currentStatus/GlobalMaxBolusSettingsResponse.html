001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.GlobalMaxBolusSettingsRequest;
009
010@MessageProps(
011    opCode=-115,
012    size=4,
013    type=MessageType.RESPONSE,
014    request=GlobalMaxBolusSettingsRequest.class
015)
016public class GlobalMaxBolusSettingsResponse extends Message {
017    
018    private int maxBolus;
019    private int maxBolusDefault;
020    
021    public GlobalMaxBolusSettingsResponse() {}
022    
023    public GlobalMaxBolusSettingsResponse(int maxBolus, int maxBolusDefault) {
024        this.cargo = buildCargo(maxBolus, maxBolusDefault);
025        this.maxBolus = maxBolus;
026        this.maxBolusDefault = maxBolusDefault;
027        
028    }
029
030    public void parse(byte[] raw) {
031        Preconditions.checkArgument(raw.length == props().size());
032        this.cargo = raw;
033        this.maxBolus = Bytes.readShort(raw, 0);
034        this.maxBolusDefault = Bytes.readShort(raw, 2);
035        
036    }
037
038    
039    public static byte[] buildCargo(int maxBolus, int maxBolusDefault) {
040        return Bytes.combine(
041            Bytes.firstTwoBytesLittleEndian(maxBolus), 
042            Bytes.firstTwoBytesLittleEndian(maxBolusDefault));
043    }
044    
045    public int getMaxBolus() {
046        return maxBolus;
047    }
048    public int getMaxBolusDefault() {
049        return maxBolusDefault;
050    }
051    
052}
