001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.helpers.Dates;
009import com.jwoglom.pumpx2.pump.messages.request.currentStatus.TimeSinceResetRequest;
010
011import java.time.Instant;
012
013/**
014 * Returns information on the pump's current internal time, which is different from the
015 * human-visible time and date which is stored in the history log. The currentTime is the
016 * user-set clock time.
017 */
018@MessageProps(
019    opCode=55,
020    size=8,
021    type=MessageType.RESPONSE,
022    request=TimeSinceResetRequest.class
023)
024public class TimeSinceResetResponse extends Message {
025    
026    private long currentTime;
027    private long pumpTimeSinceReset;
028    
029    public TimeSinceResetResponse() {}
030    
031    public TimeSinceResetResponse(long currentTime, long pumpTimeSinceReset) {
032        this.cargo = buildCargo(currentTime, pumpTimeSinceReset);
033        this.currentTime = currentTime;
034        this.pumpTimeSinceReset = pumpTimeSinceReset;
035        
036    }
037
038    public void parse(byte[] raw) {
039        Preconditions.checkArgument(raw.length == props().size());
040        this.cargo = raw;
041        this.currentTime = Bytes.readUint32(raw, 0);
042        this.pumpTimeSinceReset = Bytes.readUint32(raw, 4);
043    }
044
045    
046    public static byte[] buildCargo(long timeSinceReset, long pumpTime) {
047        return Bytes.combine(
048            Bytes.toUint32(timeSinceReset), 
049            Bytes.toUint32(pumpTime));
050    }
051
052    /**
053     * @return the pump's internal realtime clock which is an actual date,
054     * instead of the internal epoch time
055     */
056    public long getCurrentTime() {
057        return currentTime;
058    }
059
060    /**
061     * @return the evaluation of timeSinceReset against the epoch
062     */
063    public Instant getCurrentTimeInstant() {
064        return Dates.fromJan12008EpochSecondsToDate(currentTime);
065    }
066
067    /**
068     * @return the number of seconds the pump has been on since it was last reset
069     */
070    public long getPumpTimeSecondsSinceReset() {
071        return pumpTimeSinceReset;
072    }
073    
074}
