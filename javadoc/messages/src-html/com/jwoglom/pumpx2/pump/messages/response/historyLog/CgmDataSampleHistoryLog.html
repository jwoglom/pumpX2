001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007import java.math.BigInteger;
008
009@HistoryLogProps(
010    opCode = 151,
011    displayName = "CGM Data Sample",
012    usedByTidepool = true
013)
014public class CgmDataSampleHistoryLog extends HistoryLog {
015    
016    private int status;
017    private int value;
018    
019    public CgmDataSampleHistoryLog() {}
020    public CgmDataSampleHistoryLog(long pumpTimeSec, long sequenceNum, int status, int value) {
021        super(pumpTimeSec, sequenceNum);
022        this.cargo = buildCargo(pumpTimeSec, sequenceNum, status, value);
023        this.status = status;
024        this.value = value;
025        
026    }
027
028    public CgmDataSampleHistoryLog(int status, int value) {
029        this(0, 0, status, value);
030    }
031
032    public int typeId() {
033        return 151;
034    }
035
036    public void parse(byte[] raw) {
037        Preconditions.checkArgument(raw.length == 26);
038        this.cargo = raw;
039        parseBase(raw);
040        this.status = Bytes.readShort(raw, 10);
041        this.value = Bytes.readShort(raw, 19);
042        
043    }
044
045    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int status, int value) {
046        return HistoryLog.fillCargo(Bytes.combine(
047            new byte[]{-105, 0}, // (byte) 151
048            Bytes.toUint32(pumpTimeSec),
049            Bytes.toUint32(sequenceNum),
050            Bytes.firstTwoBytesLittleEndian(status), 
051            Bytes.firstTwoBytesLittleEndian(value)));
052    }
053    public int getStatus() {
054        return status;
055    }
056    public int getValue() {
057        return value;
058    }
059    
060}
