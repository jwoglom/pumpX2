001package com.jwoglom.pumpx2.pump.messages.bluetooth;
002
003import java.util.UUID;
004
005public class ServiceUUID {
006    // Pump-specific UUID used for all known functions
007    public static final UUID PUMP_SERVICE_UUID = UUID.fromString("0000fdfb-0000-1000-8000-00805f9b34fb");
008
009    // Bluetooth Device Information service (DIS)
010    public static final UUID DIS_SERVICE_UUID = UUID.fromString("0000180A-0000-1000-8000-00805f9b34fb");
011
012    // Generic Access service
013    public static final UUID GENERIC_ACCESS_SERVICE_UUID = UUID.fromString("00001800-0000-1000-8000-00805f9b34fb");
014
015    // Generic Attribute service
016    public static final UUID GENERIC_ATTRIBUTE_SERVICE_UUID = UUID.fromString("00001801-0000-1000-8000-00805f9b34fb");
017
018    public static String which(UUID uuid) {
019        if (uuid.equals(PUMP_SERVICE_UUID)) {
020            return "PUMP_SERVICE";
021        } else if (uuid.equals(DIS_SERVICE_UUID)) {
022            return "DIS_SERVICE";
023        } else if (uuid.equals(GENERIC_ACCESS_SERVICE_UUID)) {
024            return "GENERIC_ACCESS_SERVICE";
025        } else if (uuid.equals(GENERIC_ATTRIBUTE_SERVICE_UUID)) {
026            return "GENERIC_ATTRIBUTE_SERVICE";
027        } else {
028            return "unknown (" + uuid + ")";
029        }
030    }
031}
