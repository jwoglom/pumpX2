001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.MessageType;
005import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
006import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
007import com.jwoglom.pumpx2.pump.messages.request.currentStatus.PumpFeaturesV1Request;
008
009import java.math.BigInteger;
010import java.util.HashSet;
011import java.util.Set;
012
013@MessageProps(
014    opCode=79,
015    size=8,
016    type=MessageType.RESPONSE,
017    request= PumpFeaturesV1Request.class
018)
019public class PumpFeaturesV1Response extends PumpFeaturesAbstractResponse {
020    
021    private BigInteger intMap;
022    private Set<PumpFeatureType> features;
023
024    public PumpFeaturesV1Response() {}
025
026    public PumpFeaturesV1Response(BigInteger intMap) {
027        this.cargo = buildCargo(intMap);
028        this.intMap = intMap;
029        this.features = PumpFeatureType.fromBitmask(intMap);
030    }
031
032    public void parse(byte[] raw) {
033        Preconditions.checkArgument(raw.length == props().size());
034        this.cargo = raw;
035        this.intMap = Bytes.readUint64(raw, 0);
036        this.features = PumpFeatureType.fromBitmask(intMap);
037    }
038
039    
040    public static byte[] buildCargo(BigInteger intMap) {
041        return Bytes.combine(
042            Bytes.toUint64(intMap.longValue())
043        );
044    }
045    
046    public BigInteger getIntMap() {
047        return intMap;
048    }
049
050    public enum PumpFeatureType {
051        DEXCOM_G5_SUPPORTED(1),                        // position: 0
052        DEXCOM_G6_SUPPORTED(2),                        // position: 1
053        BASAL_IQ_SUPPORTED(4),                         // position: 2
054        CONTROL_IQ_SUPPORTED(1024),                    // position: 10
055        WOMBAT_SUPPORTED(65536),                       // position: 16
056        BASAL_LIMIT_SUPPORTED(262144),                 // position: 18
057        AUTO_POP_SUPPORTED(33554432),                  // position: 25
058        BLE_PUMP_CONTROL_SUPPORTED(268435456),         // position: 28
059        PUMP_SETTINGS_IN_IDP_GUI_SUPPORTED(536870912), // position: 29
060
061        ;
062
063        private final long bitmask;
064        PumpFeatureType(long bitmask) {
065            this.bitmask = bitmask;
066        }
067
068        public BigInteger bitmask() {
069            return BigInteger.valueOf(bitmask);
070        }
071
072        public String toString() {
073            return name();
074        }
075
076        public static BigInteger build(PumpFeatureType ...types) {
077            BigInteger ret = BigInteger.ZERO;
078            for (PumpFeatureType type : types) {
079                ret = ret.setBit(type.bitmask().getLowestSetBit());
080            }
081
082            return ret;
083        }
084
085        public static Set<PumpFeatureType> fromBitmask(BigInteger intMap) {
086            Set<PumpFeatureType> current = new HashSet<>();
087            for (PumpFeatureType type : PumpFeatureType.values()) {
088                if (!intMap.and(type.bitmask()).equals(BigInteger.ZERO)) {
089                    current.add(type);
090                }
091            }
092
093            return current;
094        }
095    }
096
097    public Set<PumpFeatureType> getPrimaryFeatures() {
098        return features;
099    }
100}
