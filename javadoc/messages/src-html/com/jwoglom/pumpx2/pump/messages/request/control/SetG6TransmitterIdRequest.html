001package com.jwoglom.pumpx2.pump.messages.request.control;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
010import com.jwoglom.pumpx2.pump.messages.response.control.SetG6TransmitterIdResponse;
011
012@MessageProps(
013    opCode=-80,
014    size=16,
015    type=MessageType.REQUEST,
016    characteristic=Characteristic.CONTROL,
017    signed=true,
018    response=SetG6TransmitterIdResponse.class,
019    minApi=KnownApiVersion.MOBI_API_V3_5
020)
021public class SetG6TransmitterIdRequest extends Message {
022    public static final int TXID_LENGTH = 6;
023
024    private String txId;
025
026    public SetG6TransmitterIdRequest() {
027        this.cargo = EMPTY;
028    }
029
030    public SetG6TransmitterIdRequest(byte[] raw) {
031        this.cargo = raw;
032        parse(raw);
033    }
034
035    public SetG6TransmitterIdRequest(String txId) {
036        this.cargo = buildCargo(txId);
037        this.txId = txId;
038    }
039
040    public void parse(byte[] raw) { 
041        raw = this.removeSignedRequestHmacBytes(raw);
042        Preconditions.checkArgument(raw.length == props().size());
043        this.cargo = raw;
044        this.txId = Bytes.readString(raw, 0, TXID_LENGTH);
045    }
046
047
048    public static byte[] buildCargo(String txId) {
049        return Bytes.combine(
050                Bytes.writeString(txId, 6),
051                new byte[10]
052        );
053    }
054
055    public String getTxId() {
056        return txId;
057    }
058}
