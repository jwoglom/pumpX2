061    public static UUID determine(Message message) {
062        if (message.getCharacteristic() == Characteristic.AUTHORIZATION) {
063            return AUTHORIZATION_CHARACTERISTICS;
064        }
065
066        if (message instanceof HistoryLogStreamResponse) {
067            return HISTORY_LOG_CHARACTERISTICS;
068        }
069
070        if (message.getCharacteristic() == Characteristic.CONTROL) {
071            return CONTROL_CHARACTERISTICS;
072        }
073
074        if (message.getCharacteristic() == Characteristic.CONTROL_STREAM) {
075            return CONTROL_STREAM_CHARACTERISTICS;
076        }
077
078        if (message.signed()) {
079            return CONTROL_CHARACTERISTICS;
080        }
081
082        return CURRENT_STATUS_CHARACTERISTICS;
083    }
