001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007@HistoryLogProps(
008    opCode = 66,
009    displayName = "Bolus Requested 3/3",
010    usedByAndroid = true,
011    usedByTidepool = true
012)
013public class BolusRequestedMsg3HistoryLog extends HistoryLog {
014    
015    private int bolusId;
016    private int spare;
017    private float foodBolusSize;
018    private float correctionBolusSize;
019    private float totalBolusSize;
020    
021    public BolusRequestedMsg3HistoryLog() {}
022    
023    public BolusRequestedMsg3HistoryLog(long pumpTimeSec, long sequenceNum, int bolusId, int spare, float foodBolusSize, float correctionBolusSize, float totalBolusSize) {
024        this.cargo = buildCargo(pumpTimeSec, sequenceNum, bolusId, spare, foodBolusSize, correctionBolusSize, totalBolusSize);
025        this.pumpTimeSec = pumpTimeSec;
026        this.sequenceNum = sequenceNum;
027        this.bolusId = bolusId;
028        this.spare = spare;
029        this.foodBolusSize = foodBolusSize;
030        this.correctionBolusSize = correctionBolusSize;
031        this.totalBolusSize = totalBolusSize;
032    }
033
034    public BolusRequestedMsg3HistoryLog(int bolusId, int spare, float foodBolusSize, float correctionBolusSize, float totalBolusSize) {
035        this(0, 0, bolusId, spare, foodBolusSize, correctionBolusSize, totalBolusSize);
036    }
037
038    public int typeId() {
039        return 66;
040    }
041
042    public void parse(byte[] raw) {
043        Preconditions.checkArgument(raw.length == 26);
044        this.cargo = raw;
045        parseBase(raw);
046        this.bolusId = Bytes.readShort(raw, 10);
047        this.spare = Bytes.readShort(raw, 12);
048        this.foodBolusSize = Bytes.readFloat(raw, 14);
049        this.correctionBolusSize = Bytes.readFloat(raw, 18);
050        this.totalBolusSize = Bytes.readFloat(raw, 22);
051        
052    }
053
054    
055    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int bolusId, int spare, float foodBolusSize, float correctionBolusSize, float totalBolusSize) {
056        return Bytes.combine(
057            new byte[] { (byte) 66, 0},
058            Bytes.toUint32(pumpTimeSec),
059            Bytes.toUint32(sequenceNum),
060            Bytes.firstTwoBytesLittleEndian(bolusId), 
061            Bytes.firstTwoBytesLittleEndian(spare), 
062            Bytes.toFloat(foodBolusSize), 
063            Bytes.toFloat(correctionBolusSize), 
064            Bytes.toFloat(totalBolusSize));
065    }
066
067    /**
068     * @return bolus ID
069     */
070    public int getBolusId() {
071        return bolusId;
072    }
073    public int getSpare() {
074        return spare;
075    }
076
077    /**
078     * @return the portion of the bolus amount relating to food from carbs
079     */
080    public float getFoodBolusSize() {
081        return foodBolusSize;
082    }
083
084    /**
085     * @return the portion of the bolus amount relating to the correction bolus from BG+IOB
086     */
087    public float getCorrectionBolusSize() {
088        return correctionBolusSize;
089    }
090
091    /**
092     * @return the total bolus amount which is scheduled to be delivered following this message.
093     * note that this may not be correctionBolusSize+foodBolusSize if the amounts are overridden,
094     * or due to ieee float arithmetic
095     */
096    public float getTotalBolusSize() {
097        return totalBolusSize;
098    }
099}
