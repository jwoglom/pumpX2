001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.NonControlIQIOBRequest;
009
010@MessageProps(
011        opCode=39,
012        size=12,
013        type=MessageType.RESPONSE,
014        request=NonControlIQIOBRequest.class
015)
016public class NonControlIQIOBResponse extends Message {
017    private long iob;
018    private long timeRemaining;
019    private long totalIOB;
020
021    public NonControlIQIOBResponse() {}
022
023    public NonControlIQIOBResponse(long iob, long timeRemaining, long totalIOB) {
024        this.cargo = buildCargo(iob, timeRemaining, totalIOB);
025        this.iob = iob;
026        this.timeRemaining = timeRemaining;
027        this.totalIOB = totalIOB;
028    }
029
030    private static byte[] buildCargo(long iob, long timeRemaining, long totalIOB) {
031        return Bytes.combine(
032                Bytes.toUint32(iob),
033                Bytes.toUint32(timeRemaining),
034                Bytes.toUint32(totalIOB));
035    }
036
037    public void parse(byte[] raw) {
038        Preconditions.checkArgument(raw.length == props().size());
039        iob = Bytes.readUint32(raw, 0);
040        timeRemaining = Bytes.readUint32(raw, 4);
041        totalIOB = Bytes.readUint32(raw, 8);
042        cargo = raw;
043    }
044
045    public long getIOB() {
046        return iob;
047    }
048
049    public long getTimeRemaining() {
050        return timeRemaining;
051    }
052
053    public long getTotalIOB() {
054        return totalIOB;
055    }
056}
