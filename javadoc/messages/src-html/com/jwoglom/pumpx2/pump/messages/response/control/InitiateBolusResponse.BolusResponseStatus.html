001package com.jwoglom.pumpx2.pump.messages.response.control;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.request.control.InitiateBolusRequest;
010
011@MessageProps(
012    opCode=-97,
013    size=6, // 30 with 24 byte hmac padding
014    type=MessageType.RESPONSE,
015    characteristic=Characteristic.CONTROL,
016    request=InitiateBolusRequest.class,
017    signed=true,
018    modifiesInsulinDelivery=true
019)
020public class InitiateBolusResponse extends Message {
021    
022    private int status;
023    private int bolusId;
024    private int statusTypeId;
025    
026    public InitiateBolusResponse() {}
027    
028    public InitiateBolusResponse(int status, int bolusId, int statusTypeId) {
029        this.cargo = buildCargo(status, bolusId, statusTypeId);
030        this.status = status;
031        this.bolusId = bolusId;
032        this.statusTypeId = statusTypeId;
033        
034    }
035
036    public void parse(byte[] raw) {
037        raw = removeSignedRequestHmacBytes(raw);
038        Preconditions.checkArgument(raw.length == props().size());
039        this.cargo = raw;
040        this.status = raw[0];
041        this.bolusId = Bytes.readShort(raw, 1);
042        this.statusTypeId = raw[5];
043        
044    }
045
046    
047    public static byte[] buildCargo(int status, int bolusId, int statusType) {
048        return Bytes.combine(
049            new byte[]{ (byte) status }, 
050            Bytes.firstTwoBytesLittleEndian(bolusId),
051            new byte[]{ 0, 0},
052            new byte[]{ (byte) statusType });
053    }
054    
055    public int getStatus() {
056        return status;
057    }
058    public int getBolusId() {
059        return bolusId;
060    }
061    public int getStatusTypeId() {
062        return statusTypeId;
063    }
064    public BolusResponseStatus getStatusType() {
065        return BolusResponseStatus.fromId(statusTypeId);
066    }
067
068
069    // identical to ChangeReason??
070    public enum BolusResponseStatus {
071        SUCCESS(0), // 0 = successBolusIsPending
072        REVOKED_PRIORITY(2)
073
074        ;
075
076        private final int id;
077        BolusResponseStatus(int id) {
078            this.id = id;
079        }
080
081        public static BolusResponseStatus fromId(int id) {
082            for (BolusResponseStatus s : values()) {
083                if (s.id == id) {
084                    return s;
085                }
086            }
087            return null;
088        }
089    }
090
091    /**
092     * @return true when bolus was initiated
093     */
094    public boolean wasBolusInitiated() {
095        return status == 0 && getStatusType() == BolusResponseStatus.SUCCESS;
096    }
097}
