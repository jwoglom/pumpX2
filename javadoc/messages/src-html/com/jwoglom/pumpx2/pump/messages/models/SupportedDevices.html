001package com.jwoglom.pumpx2.pump.messages.models;
002
003import com.google.common.collect.ImmutableList;
004import com.google.common.collect.ImmutableSet;
005
006import java.util.Arrays;
007import java.util.HashSet;
008import java.util.List;
009import java.util.Set;
010
011public enum SupportedDevices {
012    ALL(KnownDeviceModel.TSLIM_X2, KnownDeviceModel.MOBI),
013    TSLIM_X2_ONLY(KnownDeviceModel.TSLIM_X2),
014    MOBI_ONLY(KnownDeviceModel.MOBI),
015    ;
016
017    private final Set<KnownDeviceModel> models;
018    SupportedDevices(KnownDeviceModel... models) {
019        this.models = new HashSet<>();
020        this.models.addAll(Arrays.asList(models));
021    }
022
023    public Set<KnownDeviceModel> getModels() {
024        return models;
025    }
026
027    public boolean supports(KnownDeviceModel model) {
028        return this.models.stream().anyMatch(m -> m == model);
029    }
030}
