001package com.jwoglom.pumpx2.pump.messages.response.controlStream;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.Messages;
006import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
007
008public class ControlStreamMessages {
009    public static Message determineRequestMessage(byte[] rawBtValue) throws InstantiationException, IllegalAccessException {
010        Preconditions.checkArgument(rawBtValue.length >= 3);
011        byte opCode = rawBtValue[2];
012        return Messages.fromOpcode(opCode, Characteristic.CONTROL_STREAM).newInstance();
013    }
014}
