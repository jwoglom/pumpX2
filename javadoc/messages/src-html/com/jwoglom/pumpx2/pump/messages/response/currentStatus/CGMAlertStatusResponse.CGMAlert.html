024    public CGMAlertStatusResponse() {}
025    
026    public CGMAlertStatusResponse(long cgmAlertBitmask) {
027        this.cargo = buildCargo(cgmAlertBitmask);
028        this.intMap = BigInteger.valueOf(cgmAlertBitmask);
029        
030    }
031
032    public void parse(byte[] raw) {
033        Preconditions.checkArgument(raw.length == props().size());
034        this.cargo = raw;
035        this.intMap = Bytes.readUint64(raw, 0);
036        
037    }
038
039    
040    public static byte[] buildCargo(long cgmAlertBitmask) {
041        return Bytes.combine(
042            Bytes.toUint64(cgmAlertBitmask));
043    }
044    
045    public BigInteger getIntMap() {
046        return intMap;
047    }
048    public long getBitMap() {
049        return intMap.longValue();
050    }
051
052    public Set<CGMAlert> getCgmAlerts() {
053        return CGMAlert.fromBitmask(intMap);
054    }

123        private final int id;
124        CGMAlert(int id) {
125            this.id = id;
126        }

128        public int id() {
129            return id;
130        }
131
132        public static Set<CGMAlert> fromBitmask(BigInteger bitmask) {
133            Set<CGMAlert> set = new HashSet<>();
134            for (CGMAlert a : values()) {
135                if (bitmask.testBit(a.id())) {
136                    set.add(a);
137                }
138            }
139            return set;
140        }
141
142        public static BigInteger toBitmask(CGMAlert ...alerts) {
143            BigInteger i = BigInteger.ZERO;
144            for (CGMAlert a : alerts) {
145                i.setBit(a.id());
146            }
147            return i;
148        }
