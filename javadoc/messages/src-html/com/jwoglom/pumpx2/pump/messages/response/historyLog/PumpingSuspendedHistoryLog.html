001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007@HistoryLogProps(
008    opCode = 11,
009    displayName = "Pumping Suspended",
010    usedByTidepool = true
011)
012public class PumpingSuspendedHistoryLog extends HistoryLog {
013    
014    private int insulinAmount;
015    private int reasonId;
016    
017    public PumpingSuspendedHistoryLog() {}
018    public PumpingSuspendedHistoryLog(long pumpTimeSec, long sequenceNum, int insulinAmount, int reasonId) {
019        super(pumpTimeSec, sequenceNum);
020        this.cargo = buildCargo(pumpTimeSec, sequenceNum, insulinAmount, reasonId);
021        this.insulinAmount = insulinAmount;
022        this.reasonId = reasonId;
023        
024    }
025
026    public PumpingSuspendedHistoryLog(int insulinAmount, int reasonId) {
027        this(0, 0, insulinAmount, reasonId);
028    }
029
030    public int typeId() {
031        return 11;
032    }
033
034    /**
035     * <table border="1">
036     * <caption>Parsed fields for PumpingSuspendedHistoryLog:</caption>
037     * <tr><td><code>insulinAmount</code></td><td><code>Bytes.readShort(raw, 14)</code></td></tr>
038     * <tr><td><code>reasonId</code></td><td><code>raw[16]</code></td></tr>
039     * </table>
040     */
041    public void parse(byte[] raw) {
042        Preconditions.checkArgument(raw.length == 26);
043        this.cargo = raw;
044        parseBase(raw);
045        this.insulinAmount = Bytes.readShort(raw, 14);
046        this.reasonId = raw[16];
047        
048    }
049
050    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int insulinAmount, int reason) {
051        return HistoryLog.fillCargo(Bytes.combine(
052            new byte[]{11, 0},
053            Bytes.toUint32(pumpTimeSec),
054            Bytes.toUint32(sequenceNum),
055            Bytes.firstTwoBytesLittleEndian(insulinAmount), 
056            new byte[]{ (byte) reason }));
057    }
058
059    /**
060     * @return insulin amount in milliunits at the time pumping was suspended (I think this matches
061     * with the IOB, needs confirmation)
062     */
063    public int getInsulinAmount() {
064        return insulinAmount;
065    }
066    public int getReasonId() {
067        return reasonId;
068    }
069    public SuspendReason getReason() {
070        return SuspendReason.fromId(reasonId);
071    }
072
073    enum SuspendReason {
074        USER_ABORTED(0),
075        ALARM(1),
076        MALFUNCTION(2),
077        AUTO_SUSPEND_PREDICTIVE_LOW_GLUCOSE(6),
078
079        ;
080        private final int id;
081        SuspendReason(int id) {
082            this.id = id;
083        }
084
085        static SuspendReason fromId(int id) {
086            for (SuspendReason r : values()) {
087                if (r.id == id) {
088                    return r;
089                }
090            }
091            return null;
092        }
093    }
094    
095}
