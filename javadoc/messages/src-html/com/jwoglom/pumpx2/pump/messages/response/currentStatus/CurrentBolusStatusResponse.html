001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.helpers.Dates;
009import com.jwoglom.pumpx2.pump.messages.request.currentStatus.CurrentBolusStatusRequest;
010import com.jwoglom.pumpx2.pump.messages.response.historyLog.BolusDeliveryHistoryLog;
011
012import java.time.Instant;
013import java.util.Set;
014
015@MessageProps(
016    opCode=45,
017    size=15,
018    type=MessageType.RESPONSE,
019    request=CurrentBolusStatusRequest.class
020)
021public class CurrentBolusStatusResponse extends Message {
022    
023    private int statusId;
024    private int bolusId;
025    private long timestamp;
026    private long requestedVolume;
027    private int bolusSourceId;
028    private int bolusTypeBitmask;
029    
030    public CurrentBolusStatusResponse() {}
031    
032    public CurrentBolusStatusResponse(int statusId, int bolusId, long timestamp, long requestedVolume, int bolusSourceId, int bolusTypeBitmask) {
033        this.cargo = buildCargo(statusId, bolusId, timestamp, requestedVolume, bolusSourceId, bolusTypeBitmask);
034        this.statusId = statusId;
035        this.bolusId = bolusId;
036        this.timestamp = timestamp;
037        this.requestedVolume = requestedVolume;
038        this.bolusSourceId = bolusSourceId;
039        this.bolusTypeBitmask = bolusTypeBitmask;
040        
041    }
042
043    public void parse(byte[] raw) {
044        Preconditions.checkArgument(raw.length == props().size());
045        this.cargo = raw;
046        this.statusId = raw[0];
047        this.bolusId = Bytes.readShort(raw, 1);
048        this.timestamp = Bytes.readUint32(raw, 5);
049        this.requestedVolume = Bytes.readUint32(raw, 9);
050        this.bolusSourceId = raw[13];
051        this.bolusTypeBitmask = raw[14];
052        
053    }
054
055    
056    public static byte[] buildCargo(int status, int bolusId, long timestamp, long requestedVolume, int bolusSource, int bolusTypeBitmask) {
057        return Bytes.combine(
058            new byte[]{ (byte) status }, 
059            Bytes.firstTwoBytesLittleEndian(bolusId),
060            new byte[]{ 0, 0 },
061            Bytes.toUint32(timestamp), 
062            Bytes.toUint32(requestedVolume), 
063            new byte[]{ (byte) bolusSource }, 
064            new byte[]{ (byte) bolusTypeBitmask });
065    }
066    
067    public int getStatusId() {
068        return statusId;
069    }
070    public CurrentBolusStatus getStatus() {
071        return CurrentBolusStatus.fromId(statusId);
072    }
073
074    public enum CurrentBolusStatus {
075        REQUESTING(2),
076        DELIVERING(1),
077        ALREADY_DELIVERED_OR_INVALID(0),
078        ;
079
080        private final int id;
081        CurrentBolusStatus(int id) {
082            this.id = id;
083        }
084
085        public int getId() {
086            return id;
087        }
088
089        public static CurrentBolusStatus fromId(int id) {
090            for (CurrentBolusStatus s : values()) {
091                if (s.id == id) {
092                    return s;
093                }
094            }
095            return null;
096        }
097    }
098
099    public int getBolusId() {
100        return bolusId;
101    }
102
103    public long getTimestamp() {
104        return timestamp;
105    }
106
107    /**
108     * @return the {@link Instant} representing the timestamp in which the current bolus state
109     * ({@link CurrentBolusStatus}) was entered, according to the pump. i.e., if the status is in
110     * DELIVERING, then the timestamp the delivery started.
111     */
112    public Instant getTimestampInstant() {
113        return Dates.fromJan12008EpochSecondsToDate(timestamp);
114    }
115
116    /**
117     * @return the requested insulin volume in milliunits. To convert to human-readable format,
118     * use {@link com.jwoglom.pumpx2.pump.messages.models.InsulinUnit#from1000To1(Long)}).
119     */
120    public long getRequestedVolume() {
121        return requestedVolume;
122    }
123
124    public int getBolusSourceId() {
125        return bolusSourceId;
126    }
127    public BolusDeliveryHistoryLog.BolusSource getBolusSource() {
128        return BolusDeliveryHistoryLog.BolusSource.fromId(bolusSourceId);
129    }
130
131    public int getBolusTypeBitmask() {
132        return bolusTypeBitmask;
133    }
134    public Set<BolusDeliveryHistoryLog.BolusType> getBolusTypes() {
135        return BolusDeliveryHistoryLog.BolusType.fromBitmask(bolusTypeBitmask);
136    }
137
138    /**
139     * @return whether data is filled and this output can be used. If not valid, then the bolus
140     * is no longer "current" and {@link com.jwoglom.pumpx2.pump.messages.builders.LastBolusStatusRequestBuilder} must be used.
141     */
142    public boolean isValid() {
143        return !(getStatus() == CurrentBolusStatus.ALREADY_DELIVERED_OR_INVALID && getBolusId() == 0 && getTimestamp() == 0);
144
145    }
146    
147}
