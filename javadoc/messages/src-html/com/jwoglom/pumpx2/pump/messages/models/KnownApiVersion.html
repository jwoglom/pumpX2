001package com.jwoglom.pumpx2.pump.messages.models;
002
003public enum KnownApiVersion {
004    // v2.1 is the API version used by software v7.1 and v7.4. It is the earliest known API version,
005    // previous pump firmware did not have Bluetooth connection compatibility.
006    API_V2_1(2, 1),
007    // v2.5 is the API version used by software v7.6 and includes remote bolus
008    API_V2_5(2, 5),
009    // stub version for all V3
010    API_V3(3, 0),
011    // v3.2 is the API version used by software v7.7 and utilizes a 6-character numeric pairing PIN
012    API_V3_2(3, 2),
013    // v3.4 is the API version used by software v7.8
014    API_V3_4(3, 4),
015    // Tandem Mobi initial release
016    MOBI_API_V3_5(3, 5),
017    // Represents known messages from the app that we can't parse with known firmware
018    API_FUTURE(99, 99)
019    ;
020
021    private int major;
022    private int minor;
023    KnownApiVersion(int major, int minor) {
024        this.major = major;
025        this.minor = minor;
026    }
027
028    public ApiVersion get() {
029        return new ApiVersion(major, minor);
030    }
031}
