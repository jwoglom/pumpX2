001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007@HistoryLogProps(
008    opCode = 3,
009    displayName = "Basal Rate Change",
010    usedByTidepool = true
011)
012public class BasalRateChangeHistoryLog extends HistoryLog {
013    
014    private float commandBasalRate;
015    private float baseBasalRate;
016    private float maxBasalRate;
017    private int insulinDeliveryProfile;
018    private int changeTypeId;
019    
020    public BasalRateChangeHistoryLog() {}
021    public BasalRateChangeHistoryLog(long pumpTimeSec, long sequenceNum, float commandBasalRate, float baseBasalRate, float maxBasalRate, int insulinDeliveryProfile, int changeTypeId) {
022        super(pumpTimeSec, sequenceNum);
023        this.cargo = buildCargo(pumpTimeSec, sequenceNum, commandBasalRate, baseBasalRate, maxBasalRate, insulinDeliveryProfile, changeTypeId);
024        this.commandBasalRate = commandBasalRate;
025        this.baseBasalRate = baseBasalRate;
026        this.maxBasalRate = maxBasalRate;
027        this.insulinDeliveryProfile = insulinDeliveryProfile;
028        this.changeTypeId = changeTypeId;
029        
030    }
031
032    public BasalRateChangeHistoryLog(float commandBasalRate, float baseBasalRate, float maxBasalRate, int insulinDeliveryProfile, int changeTypeId) {
033        this(0, 0, commandBasalRate, baseBasalRate, maxBasalRate, insulinDeliveryProfile, changeTypeId);
034    }
035
036    public int typeId() {
037        return 3;
038    }
039
040    public void parse(byte[] raw) {
041        Preconditions.checkArgument(raw.length == 26);
042        this.cargo = raw;
043        parseBase(raw);
044        this.commandBasalRate = Bytes.readFloat(raw, 10);
045        this.baseBasalRate = Bytes.readFloat(raw, 14);
046        this.maxBasalRate = Bytes.readFloat(raw, 18);
047        this.insulinDeliveryProfile = Bytes.readShort(raw, 22);
048        this.changeTypeId = raw[24];
049        
050    }
051
052    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, float commandBasalRate, float baseBasalRate, float maxBasalRate, int insulinDeliveryProfile, int changeType) {
053        return Bytes.combine(
054            new byte[]{3, 0},
055            Bytes.toUint32(pumpTimeSec),
056            Bytes.toUint32(sequenceNum),
057            Bytes.toFloat(commandBasalRate), 
058            Bytes.toFloat(baseBasalRate), 
059            Bytes.toFloat(maxBasalRate), 
060            Bytes.firstTwoBytesLittleEndian(insulinDeliveryProfile), 
061            new byte[]{ (byte) changeType, 0 });
062    }
063
064    /**
065     * @return the basal rate which is being changed to
066     */
067    public float getCommandBasalRate() {
068        return commandBasalRate;
069    }
070
071    /**
072     * @return the basal rate which is default for the current profile segment
073     */
074    public float getBaseBasalRate() {
075        return baseBasalRate;
076    }
077
078    /**
079     * @return the maximum allowed basal rate in the profile settings
080     */
081    public float getMaxBasalRate() {
082        return maxBasalRate;
083    }
084
085    /**
086     * @return the ID of the Insulin Delivery Profile (IDP)
087     */
088    public int getInsulinDeliveryProfile() {
089        return insulinDeliveryProfile;
090    }
091
092    public int getChangeTypeId() {
093        return changeTypeId;
094    }
095
096    enum ChangeType {
097        TIMED_SEGMENT(1),
098        NEW_PROFILE(2),
099        TEMP_RATE_START(4),
100        TEMP_RATE_END(8),
101        PUMP_SUSPENDED(16),
102        PUMP_RESUMED(32),
103        PUMP_SHUT_DOWN(64),
104        ;
105
106        final int id;
107        ChangeType(int id) {
108            this.id = id;
109        }
110
111        public static ChangeType fromId(int id) {
112            for (ChangeType c : values()) {
113                if (c.id == id) {
114                    return c;
115                }
116            }
117            return null;
118        }
119    }
120
121    /**
122     * @return the reason that the basal rate was changed
123     */
124    public ChangeType getChangeType() {
125        return ChangeType.fromId(changeTypeId);
126    }
127    
128}
