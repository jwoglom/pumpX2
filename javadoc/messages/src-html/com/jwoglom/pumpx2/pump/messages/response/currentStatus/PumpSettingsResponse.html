001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.PumpSettingsRequest;
009
010@MessageProps(
011    opCode=83,
012    size=9,
013    type=MessageType.RESPONSE,
014    request=PumpSettingsRequest.class
015)
016public class PumpSettingsResponse extends Message {
017    
018    private int lowInsulinThreshold;
019    private int cannulaPrimeSize;
020    private int autoShutdownEnabled;
021    private int autoShutdownDuration;
022    private int featureLock;
023    private int oledTimeout;
024    private int status;
025    
026    public PumpSettingsResponse() {}
027    
028    public PumpSettingsResponse(int lowInsulinThreshold, int cannulaPrimeSize, int autoShutdownEnabled, int autoShutdownDuration, int featureLock, int oledTimeout, int status) {
029        this.cargo = buildCargo(lowInsulinThreshold, cannulaPrimeSize, autoShutdownEnabled, autoShutdownDuration, featureLock, oledTimeout, status);
030        this.lowInsulinThreshold = lowInsulinThreshold;
031        this.cannulaPrimeSize = cannulaPrimeSize;
032        this.autoShutdownEnabled = autoShutdownEnabled;
033        this.autoShutdownDuration = autoShutdownDuration;
034        this.featureLock = featureLock;
035        this.oledTimeout = oledTimeout;
036        this.status = status;
037        
038    }
039
040    public void parse(byte[] raw) {
041        Preconditions.checkArgument(raw.length == props().size());
042        this.cargo = raw;
043        this.lowInsulinThreshold = raw[0];
044        this.cannulaPrimeSize = raw[1];
045        this.autoShutdownEnabled = raw[2];
046        this.autoShutdownDuration = Bytes.readShort(raw, 3);
047        this.featureLock = raw[5];
048        this.oledTimeout = raw[6];
049        this.status = Bytes.readShort(raw, 7);
050        
051    }
052
053    
054    public static byte[] buildCargo(int lowInsulinThreshold, int cannulaPrimeSize, int autoShutdownEnabled, int autoShutdownDuration, int featureLock, int oledTimeout, int status) {
055        return Bytes.combine(
056            new byte[]{ (byte) lowInsulinThreshold }, 
057            new byte[]{ (byte) cannulaPrimeSize }, 
058            new byte[]{ (byte) autoShutdownEnabled }, 
059            Bytes.firstTwoBytesLittleEndian(autoShutdownDuration), 
060            new byte[]{ (byte) featureLock }, 
061            new byte[]{ (byte) oledTimeout }, 
062            Bytes.firstTwoBytesLittleEndian(status));
063    }
064    
065    public int getLowInsulinThreshold() {
066        return lowInsulinThreshold;
067    }
068    public int getCannulaPrimeSize() {
069        return cannulaPrimeSize;
070    }
071    public int getAutoShutdownEnabled() {
072        return autoShutdownEnabled;
073    }
074    public int getAutoShutdownDuration() {
075        return autoShutdownDuration;
076    }
077    public int getFeatureLock() {
078        return featureLock;
079    }
080    public int getOledTimeout() {
081        return oledTimeout;
082    }
083    public int getStatus() {
084        return status;
085    }
086    
087}
