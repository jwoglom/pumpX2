001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.ApiVersionRequest;
009
010/**
011 * Returns the CGM transmitter ID.
012 */
013@MessageProps(
014        opCode=97,
015        size=17,
016        type=MessageType.RESPONSE,
017        request=ApiVersionRequest.class
018)
019public class CGMHardwareInfoResponse extends Message {
020    private String hardwareInfoString;
021    private int lastByte;
022
023    public CGMHardwareInfoResponse() {}
024
025    public CGMHardwareInfoResponse(String hardwareInfoString, int lastByte) {
026        this.cargo = buildCargo(hardwareInfoString, lastByte);
027        this.hardwareInfoString = hardwareInfoString;
028        this.lastByte = lastByte;
029    }
030
031    private static byte[] buildCargo(String hardwareInfoString, int lastByte) {
032        return Bytes.combine(
033                Bytes.writeString(hardwareInfoString, 16),
034                new byte[]{(byte) lastByte});
035    }
036
037    public void parse(byte[] raw) {
038        Preconditions.checkArgument(raw.length == props().size());
039        hardwareInfoString = Bytes.readString(raw, 0, 16);
040        lastByte = raw[16];
041        cargo = raw;
042    }
043
044    public String getHardwareInfoString() {
045        return hardwareInfoString;
046    }
047
048    public int getLastByte() {
049        return lastByte;
050    }
051}
