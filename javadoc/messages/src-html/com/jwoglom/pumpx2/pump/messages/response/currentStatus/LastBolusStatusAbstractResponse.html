001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.jwoglom.pumpx2.pump.messages.Message;
004import com.jwoglom.pumpx2.pump.messages.response.historyLog.BolusDeliveryHistoryLog;
005
006import java.time.Instant;
007import java.util.Set;
008
009/**
010 * Contains everything shared between {@link LastBolusStatusResponse} and {@link LastBolusStatusV2Response}
011 * except that the latter contains requestedVolume
012 */
013public abstract class LastBolusStatusAbstractResponse extends Message {
014    public abstract int getStatus();
015    public abstract int getBolusId();
016    public abstract long getTimestamp();
017    public abstract Instant getTimestampInstant();
018    public abstract long getDeliveredVolume();
019    public abstract int getBolusStatusId();
020    public enum BolusStatus {
021        // TODO: this is guesswork and is incomplete
022        STOPPED(0),
023        COMPLETE(3),
024        ;
025
026        private final int id;
027        BolusStatus(int id) {
028            this.id = id;
029        }
030
031        public int id() {
032            return id;
033        }
034
035        public static BolusStatus fromId(int id) {
036            for (BolusStatus s : values()) {
037                if (s.id() == id) {
038                    return s;
039                }
040            }
041            return null;
042        }
043    }
044
045    public BolusStatus getBolusStatus() {
046        return BolusStatus.fromId(getBolusStatusId());
047    }
048    public abstract int getBolusSourceId();
049    public BolusDeliveryHistoryLog.BolusSource getBolusSource() {
050        return BolusDeliveryHistoryLog.BolusSource.fromId(getBolusSourceId());
051    }
052    public abstract int getBolusTypeBitmask();
053    public Set<BolusDeliveryHistoryLog.BolusType> getBolusType() {
054        return BolusDeliveryHistoryLog.BolusType.fromBitmask(getBolusTypeBitmask());
055    }
056    public abstract long getExtendedBolusDuration();
057}
