034    private String pairingCode;
035    private List<Message> sentMessages;

051    public JpakeAuthBuilder(String pairingCode, JpakeStep step, byte[] clientRound1, byte[] serverRound1, byte[] clientRound2, byte[] serverRound2, byte[] derivedSecret, SecureRandom rand) {
052        this.pairingCode = pairingCode;
053        this.cli = new EcJpake(EcJpake.Role.CLIENT, pairingCodeToBytes(pairingCode), rand);
054        this.sentMessages = new ArrayList<>();
055        this.receivedMessages = new ArrayList<>();
056
057        this.step = step;
058        this.clientRound1 = clientRound1;
059        this.serverRound1 = serverRound1;
060        this.clientRound2 = clientRound2;
061        this.serverRound2 = serverRound2;
062        this.derivedSecret = derivedSecret;
063        this.rand = rand;
064    }

066    public JpakeAuthBuilder(String pairingCode) {
067        this(pairingCode, decideInitialStep(null), null, null, null, null, null, new SecureRandom());
068    }
069
070    public JpakeAuthBuilder(String pairingCode, byte[] derivedSecret) {
071        this(pairingCode, decideInitialStep(derivedSecret), null, null, null, null, derivedSecret, new SecureRandom());
072    }
073
074    public JpakeAuthBuilder(String pairingCode, byte[] derivedSecret, SecureRandom random) {
075        this(pairingCode, decideInitialStep(derivedSecret), null, null, null, null, derivedSecret, random);
076    }
077
078    static JpakeStep decideInitialStep(byte[] derivedSecret) {
079        if (derivedSecret == null || derivedSecret.length == 0) {
080            return JpakeStep.BOOTSTRAP_INITIAL;
081        } else {
082            return JpakeStep.CONFIRM_INITIAL;
083        }
084    }
085
086    static byte[] pairingCodeToBytes(String pairingCode) {
087        //return pairingCode.getBytes(StandardCharsets.UTF_8);
088        byte[] ret = new byte[pairingCode.length()]; // should always be 6
089        for (int i=0; i<pairingCode.length(); i++) {
090            ret[i] = charCode(pairingCode.charAt(i));
091        }
092        return ret;
093    }
094
095    static byte charCode(char c) {
096        if (c == '0') return 48;
097        if (c == '1') return 49;
098        if (c == '2') return 50;
099        if (c == '3') return 51;
100        if (c == '4') return 52;
101        if (c == '5') return 53;
102        if (c == '6') return 54;
103        if (c == '7') return 55;
104        if (c == '8') return 56;
105        if (c == '9') return 57;
106        return -1;
107    }

111        if (INSTANCE == null || !INSTANCE.pairingCode.equals(pairingCode)) {
112            INSTANCE = new JpakeAuthBuilder(pairingCode, null);

125        }

127    }

133    public Message nextRequest() {
134        Message request;
135        if (step == JpakeStep.BOOTSTRAP_INITIAL) {
136            this.clientRound1 = this.cli.getRound1();
137            byte[] challenge = Arrays.copyOfRange(this.clientRound1, 0, 165);
138
139            L.i(TAG, "Req1a: " + Hex.encodeHexString(challenge));
140            request = new Jpake1aRequest(0, challenge);
141
142            step = JpakeStep.ROUND_1A_SENT;
143        } else if (step == JpakeStep.ROUND_1A_RECEIVED) {
144            byte[] challenge = Arrays.copyOfRange(this.clientRound1, 165, 330);
145
146            L.i(TAG, "Req1b: " + Hex.encodeHexString(challenge));
147            request = new Jpake1bRequest(0, challenge);
148
149            step = JpakeStep.ROUND_1B_SENT;
150        } else if (step == JpakeStep.ROUND_1B_RECEIVED) {
151            this.clientRound2 = this.cli.getRound2();
152            byte[] challenge = Arrays.copyOfRange(this.clientRound2, 0, 165);
153            L.i(TAG, "Req2: " + Hex.encodeHexString(challenge));
154            request = new Jpake2Request(0, challenge);
155
156            step = JpakeStep.ROUND_2_SENT;
157        } else if (step == JpakeStep.ROUND_2_RECEIVED) {
158            request = new Jpake3SessionKeyRequest(0);
159
160            this.derivedSecret = this.cli.deriveSecret();
161            L.i(TAG, "Req3 DerivedSecret=" + Hex.encodeHexString(derivedSecret));
162
163            step = JpakeStep.CONFIRM_3_SENT;
164        } else if (step == JpakeStep.CONFIRM_INITIAL) {
165            request = new Jpake3SessionKeyRequest(0);
166
167            step = JpakeStep.CONFIRM_3_SENT;
168        } else if (step == JpakeStep.CONFIRM_3_RECEIVED) {
169            this.clientNonce4 = generateNonce();
170            byte[] hashDigest3 = HmacSha256.hmacSha256(clientNonce4, Hkdf.build(serverNonce3, derivedSecret));
171
172            L.i(TAG, "Req4: clientNonce4=" + Hex.encodeHexString(clientNonce4));
173            L.i(TAG, "Req4: derivedSecret=" + Hex.encodeHexString(derivedSecret));
174            L.i(TAG, "Req4: serverNonce3=" + Hex.encodeHexString(serverNonce3));
175            L.i(TAG, "Req4: hashDigest3=" + Hex.encodeHexString(hashDigest3));
176            request = new Jpake4KeyConfirmationRequest(0,
177                    clientNonce4,
178                    Jpake4KeyConfirmationRequest.RESERVED,
179                    hashDigest3
180            );
181
182            step = JpakeStep.CONFIRM_4_SENT;
183        } else if (step == JpakeStep.CONFIRM_4_RECEIVED) {
184            byte[] hashDigest4 = HmacSha256.hmacSha256(serverNonce4, Hkdf.build(serverNonce3, derivedSecret));
185            if (Hex.encodeHexString(this.serverHashDigest4).equals(Hex.encodeHexString(hashDigest4))) {
186                L.i(TAG, "JpakeAuthBuilder HMAC SECRET VALIDATES");
187                step = JpakeStep.COMPLETE;
188            } else {
189                L.w(TAG, "JpakeAuthBuilder HMAC SECRET DOES NOT VALIDATE hashDigest=" + Hex.encodeHexString(hashDigest4));
190                L.w(TAG, "serverNonce4=" + Hex.encodeHexString(this.serverNonce4));
191                L.w(TAG, "derivedSecret=" + Hex.encodeHexString(this.derivedSecret));
192                L.w(TAG, "serverHashDigest4=" + Hex.encodeHexString(this.serverHashDigest4));
193                step = JpakeStep.INVALID;
194            }
195
196            return null;
197        } else {
198            return null;
199        }
200
201        this.sentMessages.add(request);
202        return request;
203    }

205    public void processResponse(Message response) {
206        this.receivedMessages.add(response);
207        if (response instanceof Jpake1aResponse) {
208            Jpake1aResponse m = (Jpake1aResponse) response;
209            L.i(TAG, "Res1a: " + Hex.encodeHexString(m.getCentralChallengeHash()));
210            this.serverRound1 = m.getCentralChallengeHash();
211
212            step = JpakeStep.ROUND_1A_RECEIVED;
213        } else if (response instanceof Jpake1bResponse) {
214            Jpake1bResponse m = (Jpake1bResponse) response;
215            L.i(TAG, "Res1b: " + Hex.encodeHexString(m.getCentralChallengeHash()));
216            byte[] fullServerRound1 = Bytes.combine(this.serverRound1, m.getCentralChallengeHash());
217            L.i(TAG, "FULL_ROUND_1_RESPONSE: " + Hex.encodeHexString(fullServerRound1));
218            this.serverRound1 = fullServerRound1;
219
220            this.cli.readRound1(fullServerRound1);
221            step = JpakeStep.ROUND_1B_RECEIVED;
222        } else if (response instanceof Jpake2Response) {
223            Jpake2Response m = (Jpake2Response) response;
224            L.i(TAG, "Res2: " + Hex.encodeHexString(m.getCentralChallengeHash()));
225            this.serverRound2 = m.getCentralChallengeHash();
226            L.i(TAG, "FULL_ROUND_2_RESPONSE: " + Hex.encodeHexString(this.serverRound2));
227
228            this.cli.readRound2(this.serverRound2);
229            step = JpakeStep.ROUND_2_RECEIVED;
230        } else if (response instanceof Jpake3SessionKeyResponse) {
231            Jpake3SessionKeyResponse m = (Jpake3SessionKeyResponse) response;
232            L.i(TAG, "Res3: nonce=" + Hex.encodeHexString(m.getDeviceKeyNonce()));
233            L.i(TAG, "Res3: reserved=" + Hex.encodeHexString(m.getDeviceKeyReserved()));
234
235            this.serverNonce3 = m.getDeviceKeyNonce();
236            step = JpakeStep.CONFIRM_3_RECEIVED;
237        } else if (response instanceof Jpake4KeyConfirmationResponse) {
238            Jpake4KeyConfirmationResponse m = (Jpake4KeyConfirmationResponse) response;
239            L.i(TAG, "JpakeAuthBuilder Res4: nonce=" + Hex.encodeHexString(m.getHashDigest()) + " hashDigest=" + Hex.encodeHexString(m.getNonce()));
240
241            this.serverNonce4 = m.getNonce();
242            this.serverHashDigest4 = m.getHashDigest();
243            step = JpakeStep.CONFIRM_4_RECEIVED;
244        }
245    }
246
247    byte[] generateNonce() {
248        byte[] nonce = new byte[8];
249        this.rand.nextBytes(nonce);
250        return nonce;
251    }
252
253    public enum JpakeStep {
254        BOOTSTRAP_INITIAL,
255        ROUND_1A_SENT,
256        ROUND_1A_RECEIVED,
257        ROUND_1B_SENT,
258        ROUND_1B_RECEIVED,
259        ROUND_2_SENT,
260        ROUND_2_RECEIVED,
261        CONFIRM_INITIAL,
262        CONFIRM_3_SENT,
263        CONFIRM_3_RECEIVED,
264        CONFIRM_4_SENT,
265        CONFIRM_4_RECEIVED,
266        COMPLETE,
267        INVALID
268    }
269
270    public boolean done() {
271        return step == JpakeStep.COMPLETE;
272    }
273
274    public boolean invalid() {
275        return step == JpakeStep.INVALID;
276    }
277
278    public byte[] getDerivedSecret() {
279        return this.derivedSecret;
280    }
281
282    public byte[] getServerNonce() {
283        return this.serverNonce3;
284    }
