001package com.jwoglom.pumpx2.pump.messages.calculator;
002
003import com.google.common.collect.ImmutableList;
004import com.jwoglom.pumpx2.pump.messages.response.currentStatus.BolusCalcDataSnapshotResponse;
005import com.jwoglom.pumpx2.pump.messages.response.currentStatus.LastBGResponse;
006
007import org.apache.commons.lang3.tuple.Pair;
008
009import java.time.Instant;
010import java.time.temporal.ChronoUnit;
011import java.util.List;
012
013public class BolusCalcLastBG {
014    private final BolusCalcDataSnapshotResponse dataSnapshot;
015    private final LastBGResponse lastBGResponse;
016
017
018    BolusCalcLastBG(BolusCalcDataSnapshotResponse dataSnapshot, LastBGResponse lastBGResponse) {
019        this.dataSnapshot = dataSnapshot;
020        this.lastBGResponse = lastBGResponse;
021    }
022
023    public Pair<Integer, List<BolusCalcCondition>> determineLastBG() {
024        Integer bgValue = null;
025
026        // Correction factor from BolusCalcDataSnapshot, aka the most recent CGM reading if it qualifies
027        if (dataSnapshot != null) {
028            if (dataSnapshot.getCorrectionFactor() > 0 && dataSnapshot.getIsAutopopAllowed()) {
029                bgValue = dataSnapshot.getCorrectionFactor();
030            }
031        }
032
033        // If the data from LastBGResponse is empty, then we rely on the correctionFactor.
034        if (lastBGResponse == null || lastBGResponse.getBgValue() == 0) {
035            if (bgValue == null) {
036                return Pair.of(null, ImmutableList.of(new BolusCalcCondition.DataDecision("not filling BG", "not present in data snapshot or LastBGResponse")));
037            } else {
038                return Pair.of(bgValue, ImmutableList.of(new BolusCalcCondition.DataDecision("filling BG from CGM")));
039            }
040        }
041
042        long durationMins = ChronoUnit.MINUTES.between(lastBGResponse.getBgTimestampInstant(), Instant.now());
043        if (bgValue == null && durationMins >= 15) {
044            return Pair.of(null, ImmutableList.of(new BolusCalcCondition.DataDecision("not filling BG", "no BG in data snapshot and LastBGResponse BG is too old")));
045        }
046
047        if (bgValue == null) {
048            if (lastBGResponse.getBgValue() >= 40 && lastBGResponse.getBgValue() <= 400) {
049                return Pair.of(lastBGResponse.getBgValue(), ImmutableList.of(new BolusCalcCondition.DataDecision("filling BG from " + lastBGResponse.getBgSource().name())));
050            } else {
051                return Pair.of(null, ImmutableList.of(new BolusCalcCondition.FailedSanityCheck("LastBGResponse BG not within interval")));
052            }
053        }
054
055        // Default to using BolusCalcDataSnapshot correctionFactor if it exists
056        return Pair.of(bgValue, ImmutableList.of(new BolusCalcCondition.DataDecision("filling BG from CGM")));
057    }
058}
