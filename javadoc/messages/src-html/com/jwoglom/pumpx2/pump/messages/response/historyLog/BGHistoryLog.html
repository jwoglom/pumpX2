001package com.jwoglom.pumpx2.pump.messages.response.historyLog;

003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.response.currentStatus.LastBGResponse;

008@HistoryLogProps(
009    opCode = 16,
010    displayName = "BG Taken",
011    usedByAndroid = true,
012    usedByTidepool = true
013)
014public class BGHistoryLog extends HistoryLog {
015
016    private int bg;
017    private int cgmCalibration;
018    private int bgSourceId;
019    private float iob;
020    private int targetBG;
021    private int isf;
022    private long spare;
023    
024    public BGHistoryLog() {}
025    
026    public BGHistoryLog(long pumpTimeSec, long sequenceNum, int bg, int cgmCalibration, int bgSourceId, float iob, int targetBG, int isf, long spare) {
027        this.cargo = buildCargo(pumpTimeSec, sequenceNum, bg, cgmCalibration, bgSourceId, iob, targetBG, isf, spare);
028        this.pumpTimeSec = pumpTimeSec;
029        this.sequenceNum = sequenceNum;
030        this.bg = bg;
031        this.cgmCalibration = cgmCalibration;
032        this.bgSourceId = bgSourceId;
033        this.iob = iob;
034        this.targetBG = targetBG;
035        this.isf = isf;
036        this.spare = spare;
037        
038    }
039
040    public int typeId() {
041        return 16;
042    }
043
044    /**
045     * <table border="1">
046     * <caption>Parsed fields for BGHistoryLog:</caption>
047     * <tr><td><code>bg</code></td><td><code>Bytes.readShort(raw, 10)</code></td></tr>
048     * <tr><td><code>cgmCalibration</code></td><td><code>raw[12]</code></td></tr>
049     * <tr><td><code>bgSourceId</code></td><td><code>raw[13]</code></td></tr>
050     * <tr><td><code>iob</code></td><td><code>Bytes.readFloat(raw, 14)</code></td></tr>
051     * <tr><td><code>targetBG</code></td><td><code>Bytes.readShort(raw, 18)</code></td></tr>
052     * <tr><td><code>isf</code></td><td><code>Bytes.readShort(raw, 20)</code></td></tr>
053     * <tr><td><code>spare</code></td><td><code>Bytes.readUint32(raw, 22)</code></td></tr>
054     * </table>
055     */
056    public void parse(byte[] raw) {
057        Preconditions.checkArgument(raw.length == 26);
058        parseBase(raw);
059        this.cargo = raw;
060        this.bg = Bytes.readShort(raw, 10);
061        this.cgmCalibration = raw[12];
062        this.bgSourceId = raw[13];
063        this.iob = Bytes.readFloat(raw, 14);
064        this.targetBG = Bytes.readShort(raw, 18);
065        this.isf = Bytes.readShort(raw, 20);
066        this.spare = Bytes.readUint32(raw, 22);
067        
068    }
069
070    
071    /**
072     * <table border="1">
073     * <caption>Byte layout for BGHistoryLog.buildCargo(...):</caption>
074     * <tr><th>Offset</th><th>Description</th></tr>
075     * <tr><td>0</td><td><code>new byte[]{...} [element 0]</code></td></tr>
076     * <tr><td>1</td><td><code>new byte[]{...} [element 1]</code></td></tr>
077     * <tr><td>2</td><td><code>toUint32(pumpTimeSec) [byte 0]</code></td></tr>
078     * <tr><td>3</td><td><code>toUint32(pumpTimeSec) [byte 1]</code></td></tr>
079     * <tr><td>4</td><td><code>toUint32(pumpTimeSec) [byte 2]</code></td></tr>
080     * <tr><td>5</td><td><code>toUint32(pumpTimeSec) [byte 3]</code></td></tr>
081     * <tr><td>6</td><td><code>toUint32(sequenceNum) [byte 0]</code></td></tr>
082     * <tr><td>7</td><td><code>toUint32(sequenceNum) [byte 1]</code></td></tr>
083     * <tr><td>8</td><td><code>toUint32(sequenceNum) [byte 2]</code></td></tr>
084     * <tr><td>9</td><td><code>toUint32(sequenceNum) [byte 3]</code></td></tr>
085     * <tr><td>10</td><td><code>firstTwoBytesLittleEndian(bg) [byte 0]</code></td></tr>
086     * <tr><td>11</td><td><code>firstTwoBytesLittleEndian(bg) [byte 1]</code></td></tr>
087     * <tr><td>12</td><td><code>new byte[]{...} [element 0]</code></td></tr>
088     * <tr><td>13</td><td><code>new byte[]{...} [element 0]</code></td></tr>
089     * <tr><td>14</td><td><code>toFloat(iob) [byte 0]</code></td></tr>
090     * <tr><td>15</td><td><code>toFloat(iob) [byte 1]</code></td></tr>
091     * <tr><td>16</td><td><code>toFloat(iob) [byte 2]</code></td></tr>
092     * <tr><td>17</td><td><code>toFloat(iob) [byte 3]</code></td></tr>
093     * <tr><td>18</td><td><code>firstTwoBytesLittleEndian(targetBG) [byte 0]</code></td></tr>
094     * <tr><td>19</td><td><code>firstTwoBytesLittleEndian(targetBG) [byte 1]</code></td></tr>
095     * <tr><td>20</td><td><code>firstTwoBytesLittleEndian(isf) [byte 0]</code></td></tr>
096     * <tr><td>21</td><td><code>firstTwoBytesLittleEndian(isf) [byte 1]</code></td></tr>
097     * <tr><td>22</td><td><code>toUint32(spare) [byte 0]</code></td></tr>
098     * <tr><td>23</td><td><code>toUint32(spare) [byte 1]</code></td></tr>
099     * <tr><td>24</td><td><code>toUint32(spare) [byte 2]</code></td></tr>
100     * <tr><td>25</td><td><code>toUint32(spare) [byte 3]</code></td></tr>
101     * </table>
102     */
103    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int bg, int cgmCalibration, int bgSource, float iob, int targetBG, int isf, long spare) {
104        return Bytes.combine(
105            new byte[]{ 16, 0 },
106            Bytes.toUint32(pumpTimeSec),
107            Bytes.toUint32(sequenceNum),
108            Bytes.firstTwoBytesLittleEndian(bg), 
109            new byte[]{ (byte) cgmCalibration }, 
110            new byte[]{ (byte) bgSource }, 
111            Bytes.toFloat(iob), 
112            Bytes.firstTwoBytesLittleEndian(targetBG), 
113            Bytes.firstTwoBytesLittleEndian(isf), 
114            Bytes.toUint32(spare));
115    }
