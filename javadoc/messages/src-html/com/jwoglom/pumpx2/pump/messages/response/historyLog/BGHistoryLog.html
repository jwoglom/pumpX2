001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.response.currentStatus.LastBGResponse;
007
008@HistoryLogProps(
009    opCode = 16,
010    displayName = "BG Taken",
011    usedByAndroid = true,
012    usedByTidepool = true
013)
014public class BGHistoryLog extends HistoryLog {
015
016    private int bg;
017    private int cgmCalibration;
018    private int bgSourceId;
019    private float iob;
020    private int targetBG;
021    private int isf;
022    private long spare;
023    
024    public BGHistoryLog() {}
025    
026    public BGHistoryLog(long pumpTimeSec, long sequenceNum, int bg, int cgmCalibration, int bgSourceId, float iob, int targetBG, int isf, long spare) {
027        this.cargo = buildCargo(pumpTimeSec, sequenceNum, bg, cgmCalibration, bgSourceId, iob, targetBG, isf, spare);
028        this.pumpTimeSec = pumpTimeSec;
029        this.sequenceNum = sequenceNum;
030        this.bg = bg;
031        this.cgmCalibration = cgmCalibration;
032        this.bgSourceId = bgSourceId;
033        this.iob = iob;
034        this.targetBG = targetBG;
035        this.isf = isf;
036        this.spare = spare;
037        
038    }
039
040    public int typeId() {
041        return 16;
042    }
043
044    public void parse(byte[] raw) {
045        Preconditions.checkArgument(raw.length == 26);
046        parseBase(raw);
047        this.cargo = raw;
048        this.bg = Bytes.readShort(raw, 10);
049        this.cgmCalibration = raw[12];
050        this.bgSourceId = raw[13];
051        this.iob = Bytes.readFloat(raw, 14);
052        this.targetBG = Bytes.readShort(raw, 18);
053        this.isf = Bytes.readShort(raw, 20);
054        this.spare = Bytes.readUint32(raw, 22);
055        
056    }
057
058    
059    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int bg, int cgmCalibration, int bgSource, float iob, int targetBG, int isf, long spare) {
060        return Bytes.combine(
061            new byte[]{ 16, 0 },
062            Bytes.toUint32(pumpTimeSec),
063            Bytes.toUint32(sequenceNum),
064            Bytes.firstTwoBytesLittleEndian(bg), 
065            new byte[]{ (byte) cgmCalibration }, 
066            new byte[]{ (byte) bgSource }, 
067            Bytes.toFloat(iob), 
068            Bytes.firstTwoBytesLittleEndian(targetBG), 
069            Bytes.firstTwoBytesLittleEndian(isf), 
070            Bytes.toUint32(spare));
071    }
072
073    /**
074     * @return the BG entry in mg/dL
075     */
076    public int getBg() {
077        return bg;
078    }
079
080    /**
081     * @return TODO(unknown)
082     */
083    public int getCgmCalibration() {
084        return cgmCalibration;
085    }
086
087    public int getBgSourceId() {
088        return bgSourceId;
089    }
090
091    /**
092     * @return the source of the BG entry (CGM or manual)
093     */
094    public LastBGResponse.BgSource getBgSource() {
095        return LastBGResponse.BgSource.fromId(bgSourceId);
096    }
097
098    /**
099     * @return the current insulin on board
100     */
101    public float getIob() {
102        return iob;
103    }
104
105    /**
106     * @return the target BG in the pump profile
107     */
108    public int getTargetBG() {
109        return targetBG;
110    }
111
112    /**
113     * @return the insulin sensitivity factor in the pump profile
114     */
115    public int getIsf() {
116        return isf;
117    }
118
119    /**
120     * @return TODO(unknown): always 1?
121     */
122    public long getSpare() {
123        return spare;
124    }
125    
126}
