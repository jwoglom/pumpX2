001package com.jwoglom.pumpx2.pump.messages.request.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
005import com.jwoglom.pumpx2.pump.messages.Message;
006import com.jwoglom.pumpx2.pump.messages.MessageType;
007import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
008import com.jwoglom.pumpx2.pump.messages.response.currentStatus.IDPSettingsResponse;
009
010/**
011 * Given an insulin delivery profile ID -- NOT the slot index -- makes a request for the settings
012 * (including number of profile segments) for that profile.
013 */
014@MessageProps(
015    opCode=64,
016    size=1,
017    type=MessageType.REQUEST,
018    response=IDPSettingsResponse.class
019)
020public class IDPSettingsRequest extends Message { 
021    private int idpId;
022    
023    public IDPSettingsRequest() {}
024
025    public IDPSettingsRequest(int idpId) {
026        this.cargo = buildCargo(idpId);
027        this.idpId = idpId;
028        
029    }
030
031    public void parse(byte[] raw) {
032        Preconditions.checkArgument(raw.length == props().size());
033        this.cargo = raw;
034        this.idpId = raw[0];
035        
036    }
037
038    
039    public static byte[] buildCargo(int idpId) {
040        return Bytes.combine(
041            Bytes.firstByteLittleEndian(idpId)
042        );
043    }
044    public int getIdpId() {
045        return idpId;
046    }
047    
048    
049}
