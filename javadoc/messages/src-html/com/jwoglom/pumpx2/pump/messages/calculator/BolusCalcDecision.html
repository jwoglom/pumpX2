001package com.jwoglom.pumpx2.pump.messages.calculator;
002
003import java.util.List;
004import java.util.Set;
005
006/**
007 * Represents data about the completed Bolus calculation, with the insulin amounts via {@link BolusCalcUnits}
008 * and the conditions which resulted in those insulin amounts being decided.
009 */
010public class BolusCalcDecision {
011    private final BolusCalcUnits units;
012    private final Set<BolusCalcCondition> conditions;
013
014    BolusCalcDecision(BolusCalcUnits units, Set<BolusCalcCondition> conditions) {
015        this.units = units;
016        this.conditions = conditions;
017    }
018
019    /**
020     * @return the {@link BolusCalcUnits} representing the bolus calculator insulin decision
021     */
022    public BolusCalcUnits getUnits() {
023        return units;
024    }
025
026    /**
027     * @return the {@link BolusCalcCondition}s which resulted in this amount of insulin being decided
028     */
029    public Set<BolusCalcCondition> getConditions() {
030        return conditions;
031    }
032
033    @Override
034    public String toString() {
035        return "<" + units + ">: " + conditions;
036    }
037}
