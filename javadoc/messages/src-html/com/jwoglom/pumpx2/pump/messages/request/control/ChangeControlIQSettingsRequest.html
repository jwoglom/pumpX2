001package com.jwoglom.pumpx2.pump.messages.request.control;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.response.control.ChangeControlIQSettingsResponse;
010
011@MessageProps(
012    opCode=-54,
013    size=6,
014    type=MessageType.REQUEST,
015    characteristic=Characteristic.CONTROL,
016    signed=true,
017    response=ChangeControlIQSettingsResponse.class
018)
019public class ChangeControlIQSettingsRequest extends Message {
020
021    private boolean enabled;
022    private int weightLbs;
023    private int totalDailyInsulinUnits;
024
025    public ChangeControlIQSettingsRequest() {
026        this.cargo = EMPTY;
027    }
028
029    public ChangeControlIQSettingsRequest(byte[] raw) {
030        parse(raw);
031    }
032
033    public ChangeControlIQSettingsRequest(boolean enabled, int weightLbs, int totalDailyInsulinUnits) {
034        parse(buildCargo(enabled, weightLbs, totalDailyInsulinUnits));
035    }
036
037    public void parse(byte[] raw) { 
038        raw = this.removeSignedRequestHmacBytes(raw);
039        Preconditions.checkArgument(raw.length == props().size());
040        this.cargo = raw;
041        this.enabled = raw[0] == 1;
042        this.weightLbs = Bytes.readShort(raw, 1);
043        this.totalDailyInsulinUnits = raw[4];
044    }
045
046
047    public static byte[] buildCargo(boolean enabled, int weightLbs, int totalDailyInsulinUnits) {
048        return Bytes.combine(
049                new byte[]{(byte)(enabled ? 1 : 0)},
050                Bytes.firstTwoBytesLittleEndian(weightLbs),
051                new byte[]{
052                        1,
053                        (byte) totalDailyInsulinUnits,
054                        1
055                }
056        );
057    }
058
059
060    public boolean isEnabled() {
061        return enabled;
062    }
063
064    public int getWeightLbs() {
065        return weightLbs;
066    }
067
068    public int getTotalDailyInsulinUnits() {
069        return totalDailyInsulinUnits;
070    }
071}
