001package com.jwoglom.pumpx2.pump.messages.request.authentication;
002
003import com.jwoglom.pumpx2.pump.messages.Message;
004import com.jwoglom.pumpx2.pump.messages.MessageType;
005import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
006import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
009import com.jwoglom.pumpx2.pump.messages.response.authentication.Jpake1bResponse;
010
011import java.util.Arrays;
012
013/**
014 * The second authorization message sent to the pump with V2 authorization style which contains the
015 * second half of the first round client output of EcJpake.
016 * The message structure is identical to {@link Jpake1aRequest}.
017 *
018 * When the Android libary is used, this message is invoked automatically by PumpX2 on Bluetooth
019 * connection initialization as part of performing the initial pump pairing process.
020 */
021@MessageProps(
022    opCode=34,
023    size=167,
024    type=MessageType.REQUEST,
025    minApi=KnownApiVersion.API_V3_2,
026    characteristic=Characteristic.AUTHORIZATION,
027    response=Jpake1bResponse.class
028)
029public class Jpake1bRequest extends Message {
030    private int appInstanceId;
031    private byte[] centralChallenge;
032
033    public Jpake1bRequest() {}
034
035    public Jpake1bRequest(int appInstanceId, byte[] centralChallenge) {
036        this.cargo = buildCargo(appInstanceId, centralChallenge);
037        this.appInstanceId = appInstanceId;
038        this.centralChallenge = centralChallenge; // 165
039    }
040
041    public Jpake1bRequest(byte[] rawCargo) {
042        parse(rawCargo);
043    }
044
045    public int getAppInstanceId() {
046        return appInstanceId;
047    }
048
049    public byte[] getCentralChallenge() {
050        return centralChallenge;
051    }
052
053    private static byte[] buildCargo(int appInstanceId, byte[] centralChallenge) {
054        byte[] cargo = new byte[167];
055        System.arraycopy(Bytes.combine(Bytes.firstTwoBytesLittleEndian(appInstanceId), centralChallenge), 0, cargo, 0, 167);
056
057        return cargo;
058    }
059
060    public void parse(byte[] raw) {
061        this.cargo = raw;
062        this.appInstanceId = Bytes.readShort(Arrays.copyOfRange(raw, 0, 2), 0);
063        this.centralChallenge = Arrays.copyOfRange(raw, 2, raw.length);
064    }
065}
