073    public static Map<Integer, Class<? extends HistoryLog>> LOG_MESSAGE_IDS = new HashMap<>();
074
075    static {
076        for (Class<? extends HistoryLog> clazz : LOG_MESSAGE_TYPES) {
077            try {
078                LOG_MESSAGE_IDS.put(clazz.newInstance().typeId(), clazz);
079            } catch (IllegalAccessException|InstantiationException e) {
080                L.e(TAG, String.format("could not instantiate %s", clazz), e);
081                e.printStackTrace();
082            }
083        }
084    }
085
086    public static HistoryLog parse(byte[] rawStream) {
087        int typeId = Bytes.readShort(rawStream, 0) & 4095;
088//        if (typeId % 256 != typeId) {
089//            L.w(TAG, "typeId "+typeId+" is being corrected to "+(typeId % 256));
090//            typeId = typeId % 256;
091//        }
092        HistoryLog ret = parseWithTypeId(rawStream, typeId);
093        if (ret instanceof UnknownHistoryLog) {
094            L.w(TAG, "retry HistoryLog parse on typeId " + typeId + " => " + ((byte) typeId));
095            HistoryLog two = parseWithTypeId(rawStream, (byte) typeId);
096            if (two instanceof UnknownHistoryLog) {
097                return ret;
098            }
099            return two;
100        }
101        return ret;
102    }
103
104    private static HistoryLog parseWithTypeId(byte[] rawStream, int typeId) {
105        HistoryLog historyLog = null;
106        if (!LOG_MESSAGE_IDS.containsKey(typeId)) {
107            L.w(TAG, "unknown HistoryLog typeId "+typeId+": "+ Hex.encodeHexString(rawStream));
108            historyLog = new UnknownHistoryLog();
109            historyLog.parse(rawStream);
110            L.i(TAG, String.format("Processed embedded HistoryLog: UnknownHistoryLog (%d): %s", typeId, Hex.encodeHexString(rawStream)));
111            return historyLog;
112        }
113
114        try {
115            historyLog = LOG_MESSAGE_IDS.get(typeId).newInstance();
116        } catch (IllegalAccessException|InstantiationException e) {
117            L.e(TAG, "could not instantiate "+typeId, e);
118            e.printStackTrace();
119            return null;
120        }
121
122        String name = MessageHelpers.lastTwoParts(historyLog.getClass().getName());
123        L.d(TAG, "found matching "+name+" HistoryLog typeId "+typeId+": "+ Hex.encodeHexString(rawStream));
124        historyLog.parse(rawStream);
125        L.i(TAG, String.format("Processed embedded HistoryLog: %s (%d): %s", name, typeId, Hex.encodeHexString(rawStream)));
126        return historyLog;
127    }
