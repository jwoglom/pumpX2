001package com.jwoglom.pumpx2.pump.messages.request.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.response.currentStatus.LastBGResponse;
008
009/**
010 * Returns the most recent BG entry which was entered in the Bolus Calculator. Does NOT return
011 * the most recent CGM reading -- you must browse through the HistoryLog to find that.
012 */
013@MessageProps(
014    opCode=50,
015    size=0,
016    type=MessageType.REQUEST,
017    response=LastBGResponse.class
018)
019public class LastBGRequest extends Message { 
020    public LastBGRequest() {
021        this.cargo = EMPTY;
022    }
023
024    public void parse(byte[] raw) {
025        Preconditions.checkArgument(raw.length == props().size());
026        this.cargo = raw;
027    }
028
029    
030}
