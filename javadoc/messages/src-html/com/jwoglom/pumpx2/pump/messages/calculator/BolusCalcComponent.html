001package com.jwoglom.pumpx2.pump.messages.calculator;
002
003import java.util.List;
004
005/**
006 * Represents data about a component of the total Bolus calculation, with its insulin amount and
007 * conditions which resulted in that insulin amount being decided.
008 */
009public class BolusCalcComponent {
010    private final Double units;
011    private final List<BolusCalcCondition> conditions;
012
013    BolusCalcComponent(Double units, List<BolusCalcCondition> conditions) {
014        this.units = units;
015        this.conditions = conditions;
016    }
017
018    /**
019     * @return the insulin amount in units for this component
020     */
021    public Double getUnits() {
022        return units;
023    }
024
025    /**
026     * @return the {@link BolusCalcCondition}s which resulted in this amount of insulin being decided
027     */
028    public List<BolusCalcCondition> getConditions() {
029        return conditions;
030    }
031
032    @Override
033    public String toString() {
034        return this.units+"u: " + conditions;
035    }
036}
