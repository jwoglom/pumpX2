001package com.jwoglom.pumpx2.pump.messages.calculator;
002
003import java.util.Objects;
004
005import javax.annotation.Nullable;
006
007public interface BolusCalcCondition {
008    BolusCalcCondition POSITIVE_BG_CORRECTION = new Decision("Adding positive BG correction", "above BG target",
009            new BolusCalcPrompt("Your BG is above target: add correction bolus?", "Correction bolus added", "Correction bolus declined"));
010    BolusCalcCondition NO_POSITIVE_BG_CORRECTION = new NonActionDecision("Not adding positive BG correction", "active IOB is greater than correction bolus while above target");
011    BolusCalcCondition SET_ZERO_INSULIN = new Decision("Setting zero insulin", "negative correction greater than carb amount",
012            new BolusCalcPrompt("Your BG is below target: reduce bolus calculation to zero?", "Bolus set to zero", "Zero bolus declined"));
013    BolusCalcCondition NEGATIVE_BG_CORRECTION = new Decision("Adding negative BG correction", "below BG target",
014            new BolusCalcPrompt("Your BG is below target: reduce bolus calculation?", "Bolus reduced", "Bolus reduction declined"));
015
016    class FailedPrecondition extends Condition implements BolusCalcCondition {
017        public final String reason;
018        FailedPrecondition(String reason) {
019            super(reason);
020            this.reason = reason;
021        }
022
023        FailedPrecondition(String reason, Object was, Object expected) {
024            this(reason + ": was " + was + ", expected " + expected);
025        }
026    }
027
028    class FailedSanityCheck extends Condition implements BolusCalcCondition {
029        public final String reason;
030        FailedSanityCheck(String reason) {
031            super("Sanity check failed: " + reason);
032            this.reason = reason;
033        }
034    }
035
036    class WaitingOnPrecondition extends Condition implements BolusCalcCondition {
037        public final String thing;
038        WaitingOnPrecondition(String thing) {
039            super("Waiting for: " + thing);
040            this.thing = thing;
041        }
042    }
043
044    class Decision extends Condition implements BolusCalcCondition {
045        Decision(String decision) {
046            super(decision);
047        }
048
049        Decision(String decision, String because) {
050            super(decision, because);
051        }
052
053        Decision(String decision, String because, BolusCalcPrompt prompt) {
054            super(decision, because, prompt);
055        }
056    }
057
058
059    class DataDecision extends Decision implements BolusCalcCondition {
060        DataDecision(String decision) {
061            super(decision);
062        }
063
064        DataDecision(String decision, String reason) {
065            super(decision, reason);
066        }
067    }
068
069    class NonActionDecision extends Decision implements BolusCalcCondition {
070        NonActionDecision(String decision, String reason) {
071            super(decision, reason);
072        }
073    }
074
075    class IgnoredDecision extends Decision implements BolusCalcCondition {
076        IgnoredDecision(BolusCalcCondition decision) {
077            super("Ignoring '" + decision.getMsg() + "'", "ignored by user");
078        }
079    }
080
081    class BolusCalcPrompt {
082        private final String promptMessage;
083        private final String whenAcceptedNotice;
084        private final String whenIgnoredNotice;
085        BolusCalcPrompt(String promptMessage, String whenAcceptedNotice, String whenIgnoredNotice) {
086            this.promptMessage = promptMessage;
087            this.whenAcceptedNotice = whenAcceptedNotice;
088            this.whenIgnoredNotice = whenIgnoredNotice;
089        }
090
091        public String getPromptMessage() {
092            return promptMessage;
093        }
094
095        public String getWhenAcceptedNotice() {
096            return whenAcceptedNotice;
097        }
098
099        public String getWhenIgnoredNotice() {
100            return whenIgnoredNotice;
101        }
102
103        @Override
104        public boolean equals(Object o) {
105            if (this == o) return true;
106            if (o == null || getClass() != o.getClass()) return false;
107            BolusCalcPrompt that = (BolusCalcPrompt) o;
108            return Objects.equals(promptMessage, that.promptMessage) && Objects.equals(whenAcceptedNotice, that.whenAcceptedNotice) && Objects.equals(whenIgnoredNotice, that.whenIgnoredNotice);
109        }
110
111        @Override
112        public int hashCode() {
113            return Objects.hash(promptMessage, whenAcceptedNotice, whenIgnoredNotice);
114        }
115    }
116
117    class Condition extends Exception {
118        private final String msg;
119        private final String reason;
120        private final BolusCalcPrompt prompt;
121        Condition(String msg) {
122            super(msg);
123            this.msg = msg;
124            this.reason = null;
125            this.prompt = null;
126        }
127
128        Condition(String msg, String reason) {
129            super(msg + " because " + reason);
130            this.msg = msg;
131            this.reason = reason;
132            this.prompt = null;
133        }
134
135        Condition(String msg, String reason, BolusCalcPrompt prompt) {
136            super(msg + " because " + reason);
137            this.msg = msg;
138            this.reason = reason;
139            this.prompt = prompt;
140        }
141
142        @Override
143        public boolean equals(Object o) {
144            if (this == o) return true;
145            if (!(o instanceof Condition)) return false;
146            Condition condition = (Condition) o;
147            return msg.equals(condition.msg);
148        }
149
150        @Override
151        public int hashCode() {
152            return Objects.hash(msg);
153        }
154
155        public String getMsg() {
156            return msg;
157        }
158
159        public @Nullable String getReason() {
160            return reason;
161        }
162
163        public @Nullable BolusCalcPrompt getPrompt() {
164            return prompt;
165        }
166    }
167
168    String getMsg();
169    @Nullable String getReason();
170    @Nullable BolusCalcPrompt getPrompt();
171}
