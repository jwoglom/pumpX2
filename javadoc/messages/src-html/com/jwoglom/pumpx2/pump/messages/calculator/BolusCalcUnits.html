001package com.jwoglom.pumpx2.pump.messages.calculator;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.shared.L;
005
006import java.math.RoundingMode;
007import java.text.DecimalFormat;
008import java.util.Objects;
009
010/**
011 * An object representing the units for each component of the bolus calculation. The total amount
012 * is the amount which is actually suggested by the pump; all other values are helpers which can
013 * be used to display the subcomponents of the calculation. Note that the total is NOT ALWAYS the
014 * sum of each other component, since in some scenarios components will be ignored in the calculation.
015 */
016public class BolusCalcUnits {
017
018    private final double total;
019    private final double fromCarbs;
020    private final double fromBG;
021    private final double fromIOB;
022    private final double fromUser;
023
024    BolusCalcUnits(double total, double fromCarbs, double fromBG, double fromIOB, double fromUser) {
025        this.total = doublePrecision(total);
026        this.fromCarbs = doublePrecision(fromCarbs);
027        this.fromBG = doublePrecision(fromBG);
028        this.fromIOB = doublePrecision(fromIOB);
029        this.fromUser = doublePrecision(fromUser);
030    }
031
032    BolusCalcUnits(double total, double fromCarbs, double fromBG, double fromIOB) {
033        this(total, fromCarbs, fromBG, fromIOB, 0.0);
034    }
035
036    /**
037     * @param fromUser the amount of insulin entered by the user which overrides all other amounts
038     * @return a {@link BolusCalcUnits} object which only contains the user-provided input
039     */
040    public static BolusCalcUnits fromUser(double fromUser) {
041        return new BolusCalcUnits(fromUser, 0.0, 0.0, 0.0, fromUser);
042    }
043
044    /**
045     * @return the total amount of insulin which will be filled in the units input.
046     * This is NOT ALWAYS the sum of all of the individual components.
047     */
048    public double getTotal() {
049        return total;
050    }
051
052    public double getFromCarbs() {
053        return fromCarbs;
054    }
055
056    public double getFromBG() {
057        return fromBG;
058    }
059
060    public double getFromIOB() {
061        return fromIOB;
062    }
063
064    public double getFromUser() {
065        return fromUser;
066    }
067
068    @Override
069    public boolean equals(Object o) {
070        if (this == o) return true;
071        if (o == null || getClass() != o.getClass()) return false;
072        BolusCalcUnits that = (BolusCalcUnits) o;
073        return Double.compare(that.total, total) == 0 && Double.compare(that.fromCarbs, fromCarbs) == 0 && Double.compare(that.fromBG, fromBG) == 0 && Double.compare(that.fromIOB, fromIOB) == 0 && Double.compare(that.fromUser, fromUser) == 0;
074    }
075
076    @Override
077    public int hashCode() {
078        return Objects.hash(total, fromCarbs, fromBG, fromIOB, fromUser);
079    }
080
081    @Override
082    public String toString() {
083        return "BolusCalcUnits(total: " + total + " = fromCarbs: " + fromCarbs + " + bg: " + fromBG + " + fromIOB: " + fromIOB + " + fromUser: " + fromUser + ")";
084    }
085
086
087    private static final DecimalFormat decFormat = new DecimalFormat("0.00");
088    static {
089        decFormat.setRoundingMode(RoundingMode.HALF_UP);
090    }
091    static double doublePrecision(double val) {
092        return Double.parseDouble(decFormat.format(val));
093    }
094}
