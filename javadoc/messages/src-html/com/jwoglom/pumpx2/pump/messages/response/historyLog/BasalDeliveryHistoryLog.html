001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007import java.math.BigInteger;
008
009@HistoryLogProps(
010    opCode = 279,
011    displayName = "Basal Delivery",
012    usedByTidepool = true
013)
014public class BasalDeliveryHistoryLog extends HistoryLog {
015    
016    private int commandedRateSource;
017    private int commandedRate;
018    private int profileBasalRate;
019    private int algorithmRate;
020    private int tempRate;
021    
022    public BasalDeliveryHistoryLog() {}
023    public BasalDeliveryHistoryLog(long pumpTimeSec, long sequenceNum, int commandedRateSource, int commandedRate, int profileBasalRate, int algorithmRate, int tempRate) {
024        super(pumpTimeSec, sequenceNum);
025        this.cargo = buildCargo(pumpTimeSec, sequenceNum, commandedRateSource, commandedRate, profileBasalRate, algorithmRate, tempRate);
026        this.commandedRateSource = commandedRateSource;
027        this.commandedRate = commandedRate;
028        this.profileBasalRate = profileBasalRate;
029        this.algorithmRate = algorithmRate;
030        this.tempRate = tempRate;
031        
032    }
033
034    public BasalDeliveryHistoryLog(int commandedRateSource, int commandedRate, int profileBasalRate, int algorithmRate, int tempRate) {
035        this(0, 0, commandedRateSource, commandedRate, profileBasalRate, algorithmRate, tempRate);
036    }
037
038    public int typeId() {
039        return 279;
040    }
041
042    public void parse(byte[] raw) {
043        Preconditions.checkArgument(raw.length == 26);
044        this.cargo = raw;
045        parseBase(raw);
046        this.commandedRateSource = Bytes.readShort(raw, 10);
047        this.commandedRate = Bytes.readShort(raw, 14);
048        this.profileBasalRate = Bytes.readShort(raw, 16);
049        this.algorithmRate = Bytes.readShort(raw, 18);
050        this.tempRate = Bytes.readShort(raw, 20);
051        
052    }
053
054    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int commandedRateSource, int commandedRate, int profileBasalRate, int algorithmRate, int tempRate) {
055        return HistoryLog.fillCargo(Bytes.combine(
056            new byte[]{23, 0}, // (byte) 279
057            Bytes.toUint32(pumpTimeSec),
058            Bytes.toUint32(sequenceNum),
059            Bytes.firstTwoBytesLittleEndian(commandedRateSource), 
060            Bytes.firstTwoBytesLittleEndian(commandedRate), 
061            Bytes.firstTwoBytesLittleEndian(profileBasalRate), 
062            Bytes.firstTwoBytesLittleEndian(algorithmRate), 
063            Bytes.firstTwoBytesLittleEndian(tempRate)));
064    }
065    public int getCommandedRateSource() {
066        return commandedRateSource;
067    }
068    public int getCommandedRate() {
069        return commandedRate;
070    }
071    public int getProfileBasalRate() {
072        return profileBasalRate;
073    }
074    public int getAlgorithmRate() {
075        return algorithmRate;
076    }
077    public int getTempRate() {
078        return tempRate;
079    }
080    
081}
