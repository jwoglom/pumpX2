001package com.jwoglom.pumpx2.pump.messages.request.authentication;
002
003import com.jwoglom.pumpx2.pump.messages.Message;
004import com.jwoglom.pumpx2.pump.messages.MessageType;
005import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
006import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
009import com.jwoglom.pumpx2.pump.messages.response.authentication.Jpake3SessionKeyResponse;
010
011import java.util.Arrays;
012
013/**
014 * The fourth authorization message sent to the pump with V2 authorization style which triggers
015 * Jpake session validation. Input is always '0'
016 *
017 * When the Android libary is used, this message is invoked automatically by PumpX2 on Bluetooth
018 * connection initialization as part of performing the initial pump pairing process.
019 */
020@MessageProps(
021    opCode=38,
022    size=2,
023    type=MessageType.REQUEST,
024    minApi=KnownApiVersion.API_V3_2,
025    characteristic=Characteristic.AUTHORIZATION,
026    response= Jpake3SessionKeyResponse.class
027)
028public class Jpake3SessionKeyRequest extends Message {
029    private int challengeParam;
030
031    public Jpake3SessionKeyRequest() {}
032
033    public Jpake3SessionKeyRequest(int challengeParam) {
034        this.cargo = buildCargo(challengeParam);
035        this.challengeParam = challengeParam;
036    }
037
038    public Jpake3SessionKeyRequest(byte[] rawCargo) {
039        parse(rawCargo);
040    }
041
042    public int getChallengeParam() {
043        return challengeParam;
044    }
045
046    private static byte[] buildCargo(int challengeParam) {
047        byte[] cargo = new byte[2];
048        System.arraycopy(Bytes.firstTwoBytesLittleEndian(challengeParam), 0, cargo, 0, 2);
049
050        return cargo;
051    }
052
053    public void parse(byte[] raw) {
054        this.cargo = raw;
055        this.challengeParam = Bytes.readShort(Arrays.copyOfRange(raw, 0, 2), 0);
056    }
057}
