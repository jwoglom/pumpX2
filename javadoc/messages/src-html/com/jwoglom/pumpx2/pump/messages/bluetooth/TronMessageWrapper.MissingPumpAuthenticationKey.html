001package com.jwoglom.pumpx2.pump.messages.bluetooth;
002
003import com.jwoglom.pumpx2.pump.messages.Message;
004import com.jwoglom.pumpx2.pump.messages.MessageType;
005import com.jwoglom.pumpx2.pump.messages.PacketArrayList;
006import com.jwoglom.pumpx2.pump.messages.Packetize;
007import com.jwoglom.pumpx2.pump.messages.bluetooth.models.Packet;
008import com.jwoglom.pumpx2.shared.L;
009
010import java.util.List;
011
012public class TronMessageWrapper {
013
014    private final Message message;
015    private final List<Packet> packets;
016
017    public TronMessageWrapper(Message requestMessage, byte currentTxId) {
018        this.message = requestMessage;
019
020        byte[] authKey = new byte[0];
021        if (requestMessage.signed()) {
022            if (PumpStateSupplier.authenticationKey == null) {
023                throw new MissingPumpAuthenticationKey("PumpStateSupplier.authenticationKey is not set for signed message", requestMessage);
024            }
025            try {
026                authKey = PumpStateSupplier.authenticationKey.get();
027            } catch (IllegalStateException e) {
028                throw new MissingPumpAuthenticationKey("PumpStateSupplier.authenticationKey returned " + e.getMessage(), requestMessage);
029            }
030        }
031        this.packets = Packetize.packetize(requestMessage, authKey, currentTxId);
032    }
033
034    // When viewing BT responses in Wireshark, what would be split into two packets has been seen as one packet.
035    // The maxChunkSize overrides the default 18-size
036    public TronMessageWrapper(Message requestMessage, byte currentTxId, int maxChunkSize) {
037        this.message = requestMessage;
038
039        byte[] authKey = new byte[0];
040        if (requestMessage.signed()) {
041            authKey = PumpStateSupplier.authenticationKey.get();
042        }
043        this.packets = Packetize.packetize(requestMessage, authKey, currentTxId, maxChunkSize);
044    }
045
046
047    public List<Packet> packets() {
048        return packets;
049    }
050
051    public Packet mergeIntoSinglePacket() {
052        Packet packet = null;
053        for (Packet pkt : packets()) {
054            if (packet == null) {
055                packet = pkt;
056            } else {
057                packet = packet.merge(pkt);
058            }
059        }
060        return packet;
061    }
062
063    public Message message() {
064        return message;
065    }
066
067    public PacketArrayList buildPacketArrayList(MessageType type) {
068        int opCode = message.getResponseOpCode();
069        int size = message.getResponseSize();
070
071        // When parsing a request message, set the expected response opcode/size
072        // to the request's size.
073        if (type == MessageType.REQUEST) {
074            opCode = message.opCode();
075            size = message.props().size();
076
077            // This reflects the size += 24 in Packetize.java which adds a hmacsha1
078            if (message.signed()) {
079                size += 24;
080            }
081        }
082
083        return PacketArrayList.build(
084                (byte) opCode,
085                message.getCharacteristic(),
086                (byte) size,
087                packets.get(0).transactionId(),
088                message.signed()
089        );
090    }
091
092    public static class MissingPumpAuthenticationKey extends RuntimeException {
093        MissingPumpAuthenticationKey(String reason, Message requestMessage) {
094            super("Missing pump authentication key when parsing "+requestMessage+": "+reason);
095        }
096    }
097}
