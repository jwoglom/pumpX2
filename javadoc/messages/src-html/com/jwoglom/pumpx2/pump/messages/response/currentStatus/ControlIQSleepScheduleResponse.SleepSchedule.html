001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.models.MinsTime;
009import com.jwoglom.pumpx2.pump.messages.models.MultiDay;
010import com.jwoglom.pumpx2.pump.messages.request.currentStatus.ControlIQSleepScheduleRequest;
011import com.jwoglom.pumpx2.shared.JavaHelpers;
012
013import java.util.Set;
014
015@MessageProps(
016    opCode=107,
017    size=24,
018    type=MessageType.RESPONSE,
019    request=ControlIQSleepScheduleRequest.class
020)
021public class ControlIQSleepScheduleResponse extends Message {
022    
023    private SleepSchedule schedule0;
024    private SleepSchedule schedule1;
025    private SleepSchedule schedule2;
026    private SleepSchedule schedule3;
027    
028    public ControlIQSleepScheduleResponse() {}
029    
030    public ControlIQSleepScheduleResponse(SleepSchedule schedule0, SleepSchedule schedule1, SleepSchedule schedule2, SleepSchedule schedule3) {
031        this.cargo = buildCargo(schedule0, schedule1, schedule2, schedule3);
032        this.schedule0 = schedule0;
033        this.schedule1 = schedule1;
034        this.schedule2 = schedule2;
035        this.schedule3 = schedule3;
036        
037    }
038
039    public void parse(byte[] raw) {
040        Preconditions.checkArgument(raw.length == props().size());
041        this.cargo = raw;
042        this.schedule0 = new SleepSchedule(raw, 0);
043        this.schedule1 = new SleepSchedule(raw, 6);
044        this.schedule2 = new SleepSchedule(raw, 12);
045        this.schedule3 = new SleepSchedule(raw, 18);
046    }
047
048    
049    public static byte[] buildCargo(SleepSchedule schedule0, SleepSchedule schedule1, SleepSchedule schedule2, SleepSchedule schedule3) {
050        return Bytes.combine(
051            schedule0.build(),
052            schedule1.build(),
053            schedule2.build(),
054            schedule3.build());
055    }
056    
057    public SleepSchedule getSchedule0() {
058        return schedule0;
059    }
060    public SleepSchedule getSchedule1() {
061        return schedule1;
062    }
063    public SleepSchedule getSchedule2() {
064        return schedule2;
065    }
066    public SleepSchedule getSchedule3() {
067        return schedule3;
068    }
069
070    public static class SleepSchedule {
071        private final int enabled;
072        private final int activeDays;
073        private final int startTime;
074        private final int endTime;
075
076        public SleepSchedule(byte[] cargo, int i) {
077            this.enabled = cargo[i];
078            this.activeDays = cargo[i+1];
079            this.startTime = Bytes.readShort(cargo, i+2);
080            this.endTime = Bytes.readShort(cargo, i+4);
081        }
082
083        public SleepSchedule(int enabled, int activeDays, int startTime, int endTime) {
084            this.enabled = enabled;
085            this.activeDays = activeDays;
086            this.startTime = startTime;
087            this.endTime = endTime;
088        }
089
090        public int getEnabled() {
091            return enabled;
092        }
093
094        public Set<MultiDay> activeDays() {
095            return MultiDay.fromBitmask(activeDays);
096        }
097
098        public MinsTime startTime() {
099            return new MinsTime(startTime);
100        }
101
102        public MinsTime endTime() {
103            return new MinsTime(endTime);
104        }
105
106        public byte[] build() {
107            return Bytes.combine(
108                new byte[]{ (byte) enabled },
109                new byte[]{ (byte) activeDays },
110                Bytes.firstTwoBytesLittleEndian(startTime),
111                Bytes.firstTwoBytesLittleEndian(endTime));
112        }
113
114        public String toString() {
115            return JavaHelpers.autoToString(this, null);
116        }
117    }
118}
