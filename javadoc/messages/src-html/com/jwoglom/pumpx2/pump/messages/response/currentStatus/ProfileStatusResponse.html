001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.google.common.collect.ImmutableList;
005import com.jwoglom.pumpx2.pump.messages.Message;
006import com.jwoglom.pumpx2.pump.messages.MessageType;
007import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
008import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
009import com.jwoglom.pumpx2.pump.messages.request.currentStatus.ProfileStatusRequest;
010
011import java.util.List;
012
013/**
014 * The returned idp (insulin delivery profile) slots have a value of -1 if the slot is not present.
015 * Otherwise, an IDPSettingsRequest with that IDP id will return the details of that slot.
016 * The IDP slot 0 ALWAYS contains the currently selected insulin delivery profile, unless it is
017 * -1 in which case no delivery profiles exist.
018 *
019 * @see com.jwoglom.pumpx2.pump.messages.builders.InsulinDeliveryProfileRequestBuilder
020 */
021@MessageProps(
022    opCode=63,
023    size=8,
024    type=MessageType.RESPONSE,
025    request=ProfileStatusRequest.class
026)
027public class ProfileStatusResponse extends Message {
028    
029    private int numberOfProfiles;
030    private int idpSlot0Id;
031    private int idpSlot1Id;
032    private int idpSlot2Id;
033    private int idpSlot3Id;
034    private int idpSlot4Id;
035    private int idpSlot5Id;
036    private int activeSegmentIndex;
037    
038    public ProfileStatusResponse() {}
039    
040    public ProfileStatusResponse(int numberOfProfiles, int idpSlot0Id, int idpSlot1Id, int idpSlot2Id, int idpSlot3Id, int idpSlot4Id, int idpSlot5Id, int activeSegmentIndex) {
041        this.cargo = buildCargo(numberOfProfiles, idpSlot0Id, idpSlot1Id, idpSlot2Id, idpSlot3Id, idpSlot4Id, idpSlot5Id, activeSegmentIndex);
042        this.numberOfProfiles = numberOfProfiles;
043        this.idpSlot0Id = idpSlot0Id;
044        this.idpSlot1Id = idpSlot1Id;
045        this.idpSlot2Id = idpSlot2Id;
046        this.idpSlot3Id = idpSlot3Id;
047        this.idpSlot4Id = idpSlot4Id;
048        this.idpSlot5Id = idpSlot5Id;
049        this.activeSegmentIndex = activeSegmentIndex;
050        
051    }
052
053    public void parse(byte[] raw) {
054        Preconditions.checkArgument(raw.length == props().size());
055        this.cargo = raw;
056        this.numberOfProfiles = raw[0];
057        this.idpSlot0Id = raw[1];
058        this.idpSlot1Id = raw[2];
059        this.idpSlot2Id = raw[3];
060        this.idpSlot3Id = raw[4];
061        this.idpSlot4Id = raw[5];
062        this.idpSlot5Id = raw[6];
063        this.activeSegmentIndex = raw[7];
064        
065    }
066
067    
068    public static byte[] buildCargo(int numberOfProfiles, int idpSlot0Id, int idpSlot1Id, int idpSlot2Id, int idpSlot3Id, int idpSlot4Id, int idpSlot5Id, int activeSegmentIndex) {
069        return Bytes.combine(
070            new byte[]{ (byte) numberOfProfiles }, 
071            new byte[]{ (byte) idpSlot0Id }, 
072            new byte[]{ (byte) idpSlot1Id }, 
073            new byte[]{ (byte) idpSlot2Id }, 
074            new byte[]{ (byte) idpSlot3Id }, 
075            new byte[]{ (byte) idpSlot4Id }, 
076            new byte[]{ (byte) idpSlot5Id }, 
077            new byte[]{ (byte) activeSegmentIndex });
078    }
079    
080    public int getNumberOfProfiles() {
081        return numberOfProfiles;
082    }
083    public int getIdpSlot0Id() {
084        return idpSlot0Id;
085    }
086    public int getIdpSlot1Id() {
087        return idpSlot1Id;
088    }
089    public int getIdpSlot2Id() {
090        return idpSlot2Id;
091    }
092    public int getIdpSlot3Id() {
093        return idpSlot3Id;
094    }
095    public int getIdpSlot4Id() {
096        return idpSlot4Id;
097    }
098    public int getIdpSlot5Id() {
099        return idpSlot5Id;
100    }
101    public int getActiveSegmentIndex() {
102        return activeSegmentIndex;
103    }
104
105    /**
106     * @return the active insulin delivery profile ID, which is always slot 0
107     */
108    public int getActiveIdpSlotId() {
109        return getIdpSlot0Id();
110    }
111
112    /**
113     * @return all IDP slots in order restricted to the total number of profiles.
114     */
115    public List<Integer> getIdpSlotIds() {
116        return ImmutableList.of(
117                getIdpSlot0Id(),
118                getIdpSlot1Id(),
119                getIdpSlot2Id(),
120                getIdpSlot3Id(),
121                getIdpSlot4Id(),
122                getIdpSlot5Id()
123        ).subList(0, getNumberOfProfiles());
124    }
125    
126}
