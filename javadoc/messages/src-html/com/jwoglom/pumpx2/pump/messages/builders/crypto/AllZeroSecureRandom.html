001package com.jwoglom.pumpx2.pump.messages.builders.crypto;
002
003import java.security.NoSuchAlgorithmException;
004import java.security.SecureRandom;
005import java.security.SecureRandomSpi;
006import java.util.Arrays;
007
008public class AllZeroSecureRandom extends SecureRandom {
009
010    private static class CustomSpi extends SecureRandomSpi {
011        CustomSpi() {}
012
013        @Override
014        protected void engineSetSeed(byte[] seed) {
015        }
016
017        @Override
018        protected void engineNextBytes(byte[] bytes) {
019            Arrays.fill(bytes, (byte) 0);
020        }
021
022        @Override
023        protected byte[] engineGenerateSeed(int numBytes) {
024            return null;
025        }
026
027    }
028
029    public AllZeroSecureRandom()
030    {
031        super(createSPI(), null);
032    }
033
034    static SecureRandomSpi createSPI()
035    {
036        return new CustomSpi();
037    }
038
039    public static void main(String... args) {
040        AllZeroSecureRandom secRnd = new AllZeroSecureRandom();
041
042        byte[] rndBytes = new byte[17];
043        secRnd.nextBytes(rndBytes);
044
045        for (byte r : rndBytes)
046        {
047            System.out.print(String.format("%02x ", r & 0xFF));
048        }
049        System.out.println();
050    }
051
052    private static final long serialVersionUID = -7550189683373026294L;
053
054}
