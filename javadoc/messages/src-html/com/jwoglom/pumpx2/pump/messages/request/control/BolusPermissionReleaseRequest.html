001package com.jwoglom.pumpx2.pump.messages.request.control;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
010import com.jwoglom.pumpx2.pump.messages.response.control.BolusPermissionReleaseResponse;
011
012/**
013 * This occurs after BolusCalcDataSnapshotRequest and has an input of a bolusId which was returned
014 * by BolusPermissionRequest. It returns a status of 0 if the bolus permission was successfully
015 * released (i.e. relinquished), otherwise 1 if the bolus isn't an ID which was requested.
016 */
017@MessageProps(
018    opCode=-16,
019    size=4, // plus 24 byte padding
020    type=MessageType.REQUEST,
021    characteristic=Characteristic.CONTROL,
022    response= BolusPermissionReleaseResponse.class,
023    signed=true,
024    minApi=KnownApiVersion.API_V2_5
025)
026public class BolusPermissionReleaseRequest extends Message {
027    private long bolusId;
028
029    public BolusPermissionReleaseRequest() {}
030
031    public BolusPermissionReleaseRequest(long bolusId) {
032        this.cargo = buildCargo(bolusId);
033        this.bolusId = bolusId;
034    }
035
036    public static byte[] buildCargo(long bolusId) {
037        return Bytes.combine(
038                Bytes.toUint32(bolusId)
039        );
040    }
041
042    public void parse(byte[] raw) {
043        raw = removeSignedRequestHmacBytes(raw);
044        Preconditions.checkArgument(raw.length == props().size());
045        this.cargo = raw;
046        this.bolusId = Bytes.readUint32(raw, 0);
047    }
048
049    public long getBolusId() {
050        return bolusId;
051    }
052}
