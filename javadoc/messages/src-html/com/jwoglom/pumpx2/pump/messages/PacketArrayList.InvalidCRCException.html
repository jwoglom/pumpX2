022public class PacketArrayList {
023    protected static final String TAG = "PacketArrayList";
024    public static final String IGNORE_INVALID_HMAC = "IGNORE_HMAC_SIGNATURE_EXCEPTION";
025    public static boolean ignoreInvalidTxId = false;
026
027    protected byte expectedOpCode;
028    protected byte expectedCargoSize;
029    protected int actualExpectedCargoSize;
030    protected byte expectedTxId;
031    protected boolean isSigned;
032    protected byte[] fullCargo;
033    protected byte[] messageData;
034
035    protected byte firstByteMod15;
036    protected byte opCode;
037    protected boolean empty = true;
038    protected final byte[] expectedCrc = {0, 0};
039
040    protected PacketArrayList(byte expectedopCode, byte expectedCargoSize, byte expectedTxId, boolean isSigned) {
041        Preconditions.checkArgument(expectedopCode != 0);
042        Preconditions.checkArgument(expectedCargoSize >= 0 || (256 + (int)expectedCargoSize) >= 0);
043        this.expectedCargoSize = expectedCargoSize;
044        if (expectedCargoSize < 0 && expectedCargoSize > -128) {
045            this.actualExpectedCargoSize = 256 + expectedCargoSize;
046        } else {
047            this.actualExpectedCargoSize = expectedCargoSize;
048        }
049        this.expectedOpCode = expectedopCode;
050        this.expectedTxId = expectedTxId;
051        this.isSigned = isSigned;
052        this.fullCargo = new byte[(this.actualExpectedCargoSize + 2)];
053        this.messageData = new byte[3];
054    }
055
056    // Returns either PacketArrayList or StreamPacketArrayList depending on the opcode
057    public static PacketArrayList build(byte expectedopCode, Characteristic characteristic, byte expectedCargoSize, byte expectedTxId, boolean isSigned) {
058        Class<? extends Message> messageClass = Messages.fromOpcode(expectedopCode, characteristic);
059        L.d(TAG, String.format("queried messageClass for expectedOpcode %d, %s: expectedCargoSize=%d", expectedopCode, messageClass, expectedCargoSize));
060        MessageProps messageProps = messageClass.getAnnotation(MessageProps.class);
061        if (messageProps.stream()) {
062            return new StreamPacketArrayList(expectedopCode, expectedCargoSize, expectedTxId, isSigned);
063        }
064
065        return new PacketArrayList(expectedopCode, expectedCargoSize, expectedTxId, isSigned);
066    }

068    public byte[] messageData() {
069        return messageData;
070    }

073        return opCode;

076    public final boolean needsMorePacket() {
077        byte b = this.firstByteMod15;
078        return b >= 0;
079    }
080
081    public boolean validate(byte[] authKey) {
082        Intrinsics.checkParameterIsNotNull(authKey, "ak");
083        if (needsMorePacket()) {
084            return false;
085        }
086        createMessageData();
087        if (this.fullCargo.length >= 2) {
088            System.arraycopy(this.fullCargo, this.fullCargo.length - 2, this.expectedCrc, 0, 2);
089        }
090        byte[] a = Bytes.calculateCRC16(this.messageData);
091        byte[] lastTwoB = this.expectedCrc;
092        boolean ok = true;
093
094        if (!(a[0] == lastTwoB[0] && a[1] == lastTwoB[1])) {
095            ok = false;
096        }
097        // the fullCargo size is 2 + the message length.
098        L.d(TAG, "validate fullCargo size="+fullCargo.length+" ok="+ok);
099        if (!ok) {
100            if (shouldIgnoreInvalidHmac(authKey)) {
101                L.e(TAG, "CRC validation failed for: " + ((int) this.expectedOpCode) + ". a: " + Hex.encodeHexString(a) + " lastTwoB: " + Hex.encodeHexString(lastTwoB) + ". fullCargo len=" + fullCargo.length + " opCode="+opCode);
102            } else {
103                throw new InvalidCRCException("CRC validation failed for: " + ((int) this.expectedOpCode) + ". a: " + Hex.encodeHexString(a) + " lastTwoB: " + Hex.encodeHexString(lastTwoB) + ". fullCargo len=" + fullCargo.length + " opCode="+opCode+" authKey="+new String(authKey));
104            }
105        } else if (this.isSigned) {
106            L.d(TAG, "validate(" + Hex.encodeHexString(authKey) + ") messageData: " + Hex.encodeHexString(messageData) + " len: " + messageData.length + " fullCargo: " + Hex.encodeHexString(fullCargo) + " len: " + fullCargo.length);
107            byte[] byteArray = Bytes.dropLastN(this.messageData, 20);
108            byte[] bArr2 = this.messageData;
109            byte[] expectedHmac = Bytes.dropFirstN(bArr2, bArr2.length - 20);
110            byte[] hmacSha = Packetize.doHmacSha1(byteArray, authKey);
111            if (!Arrays.equals(expectedHmac, hmacSha)) {
112                L.e(TAG, "Pump response invalid signature: expectedHmac=" + Hex.encodeHexString(expectedHmac)+" hmacSha="+Hex.encodeHexString(hmacSha));
113                if (shouldIgnoreInvalidHmac(authKey)) {
114                    return true;
115                }
116
117
118                throw new InvalidSignedMessageHMACSignatureException("Pump response invalid: SIGNATURE");
119            }
120        }
121        return ok;
122    }
123
124    protected void createMessageData() {
125        byte[] bArr = this.messageData;
126        bArr[0] = this.expectedOpCode;
127        bArr[1] = this.expectedTxId;
128        bArr[2] = this.expectedCargoSize;
129        this.messageData = Bytes.combine(bArr, Bytes.dropLastN(this.fullCargo, 2));
130    }
131
132    protected void parse(byte[] bArr) {
133        byte opCode = bArr[2];
134        int cargoSize = bArr[4];
135        if (cargoSize < 0 && cargoSize > -128) {
136            cargoSize += 256;
137        }
138        // ErrorResponse handling. Can have a dynamic cargo size in different situations.
139        if (77 == opCode && 2 == cargoSize) {
140            this.expectedOpCode = 77;
141            this.expectedCargoSize = 2;
142            this.actualExpectedCargoSize = 2;
143            this.fullCargo = new byte[4];
144        } else if (77 == opCode && 26 == cargoSize) {
145            this.expectedOpCode = 77;
146            this.expectedCargoSize = 26;
147            this.actualExpectedCargoSize = 26;
148            this.fullCargo = new byte[28];
149        }
150        // ApiVersionRequest can either have 0 cargo size or 2
151        // its cargo size is set as 2 in the MessageProps but we allow the 0-cargo here as well
152        if (32 == opCode && 0 == cargoSize) {
153            this.expectedCargoSize = 0;
154            this.actualExpectedCargoSize = 0;
155        }
156        if (opCode == this.expectedOpCode) {
157            this.firstByteMod15 = (byte) (bArr[0] & 15);
158            this.opCode = bArr[2];
159            byte txId = bArr[3];
160            L.d(TAG, "PacketArrayList firstByteMod15="+firstByteMod15+" opCode="+opCode+" txId="+txId+" expectedTxId="+expectedTxId+" bArr="+JavaHelpers.display(bArr));
161            if (txId != this.expectedTxId) {
162                throwUnexpectedTransactionIdException(txId);
163            } else if (cargoSize != this.actualExpectedCargoSize) {
164                if (cargoSize == this.actualExpectedCargoSize + 24 && isSigned) {
165                    L.i(TAG, "adding +24 expectedCargoSize for already signed request which contains an existing trailer");
166                    expectedCargoSize += 24;
167                    actualExpectedCargoSize += 24;
168                } else {
169                    throw new IllegalArgumentException("Unexpected cargo size: " + ((int) cargoSize) + ", expecting " + ((int) this.actualExpectedCargoSize) + " for opCode="+opCode);
170                }
171            }
172            this.fullCargo = Bytes.dropFirstN(bArr, 5);
173        } else {
174            throw new UnexpectedOpCodeException(opCode, this.expectedOpCode);
175        }
176        L.d(TAG, "PacketArrayParse ok: opCode="+opCode+" firstByteMod15="+firstByteMod15+" cargoSize="+cargoSize);
177    }

180    public void validatePacket(byte[] packetData) {
181        Intrinsics.checkParameterIsNotNull(packetData, "packetData");
182        if (packetData.length == 0) {
183            throw new IllegalArgumentException("Empty data");
184        } else if (packetData.length >= 3) {
185            byte firstByteMod15 = (byte) (packetData[0] & 15);
186            byte secondByte = packetData[1];
187            L.d(TAG, "Found tx id: "+secondByte+" expected "+this.expectedTxId);
188            if (secondByte == this.expectedTxId) {
189                if (this.empty) {
190                    this.firstByteMod15 = firstByteMod15;
191                    L.d(TAG, "txid matches, firstByteMod15="+firstByteMod15+", parsing packetData");
192                    parse(packetData);
193                } else if (((byte) 0) == firstByteMod15) {
194                    L.d(TAG, "firstByteMod15=0");
195                    if (this.firstByteMod15 == firstByteMod15) {
196                        this.fullCargo = Bytes.combine(this.fullCargo, Bytes.dropFirstN(packetData, 2));
197                        L.d(TAG, "firstByteMod15 matches expected, fullCargo="+ Hex.encodeHexString(fullCargo));
198                    } else {
199                        throw new IllegalArgumentException("Unexpected packets remaining 3: " + ((int) firstByteMod15) + ", expected " + ((int) this.firstByteMod15) + ", opCode: " + ((int) this.expectedOpCode));
200                    }
201                } else if (this.firstByteMod15 == firstByteMod15) {
202                    this.fullCargo = Bytes.combine(this.fullCargo, Bytes.dropFirstN(packetData, 2));
203                    L.d(TAG, "firstByteMod15 matches expected, nonzero, fullCargo=" + Hex.encodeHexString(fullCargo));
204                } else {
205                    throw new IllegalArgumentException("Unexpected packets remaining 2: " + ((int) firstByteMod15) + ", expected " + ((int) this.firstByteMod15) + ", opCode: " + ((int) this.expectedOpCode));
206                }
207                this.empty = false;
208                this.firstByteMod15 = (byte) (this.firstByteMod15 - 1);
209                L.d(TAG, "validatePacket: decrementing firstByteMod15 to " + firstByteMod15);
210                return;
211            }
212            throwUnexpectedTransactionIdException(secondByte);
213        } else {
214            throw new IllegalArgumentException("Invalid data size: " + packetData.length);
215        }
216    }
217
218    private void throwUnexpectedTransactionIdException(byte foundTxId) {
219        RuntimeException ex = new UnexpectedTransactionIdException(foundTxId, this.expectedTxId, this.expectedOpCode);
220        if (ignoreInvalidTxId) {
221            L.e(TAG, ex);
222            return;
223        }
224        throw ex;
225    }
226
227    private boolean shouldIgnoreInvalidHmac(byte[] authKey) {
228        return Arrays.equals(
229                Arrays.copyOfRange(authKey, 0, IGNORE_INVALID_HMAC.length()),
230                Arrays.copyOfRange(IGNORE_INVALID_HMAC.getBytes(StandardCharsets.UTF_8), 0, IGNORE_INVALID_HMAC.length())
231        );
232    }
