001package com.jwoglom.pumpx2.pump.messages.builders;
002
003import com.jwoglom.pumpx2.pump.messages.Message;
004import com.jwoglom.pumpx2.pump.messages.models.ApiVersion;
005import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
006import com.jwoglom.pumpx2.pump.messages.request.currentStatus.PumpFeaturesV1Request;
007import com.jwoglom.pumpx2.pump.messages.request.currentStatus.PumpFeaturesV2Request;
008
009/**
010 * On an earlier than V2_API pump, PumpFeaturesV2Request returns a bad opcode ErrorResponse
011 */
012public class PumpFeaturesRequestBuilder {
013    public static Message create(ApiVersion apiVersion) {
014        if (apiVersion.greaterThan(KnownApiVersion.API_V2_1)) {
015            return new PumpFeaturesV2Request();
016        } else {
017            return new PumpFeaturesV1Request();
018        }
019    }
020}
