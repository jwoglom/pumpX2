001package com.jwoglom.pumpx2.pump.messages.models;
002
003import java.util.HashSet;
004import java.util.Set;
005
006public enum MultiDay {
007    MONDAY(1),
008    TUESDAY(2),
009    WEDNESDAY(4),
010    THURSDAY(8),
011    FRIDAY(16),
012    SATURDAY(32),
013    SUNDAY(64),
014    ;
015
016    public static final Set<MultiDay> ALL_DAYS = fromBitmask(127);
017
018    private final int id;
019
020    MultiDay(int id) {
021        this.id = id;
022    }
023
024    public int id() {
025        return id;
026    }
027
028    public static Set<MultiDay> fromBitmask(int bitmask) {
029        Set<MultiDay> set = new HashSet<>();
030        for (MultiDay day : values()) {
031            if (bitmask % day.id() == 0) {
032                set.add(day);
033            }
034        }
035
036        return set;
037    }
038
039    public static int toBitmask(MultiDay... days) {
040        int mask = 0;
041        for (MultiDay day : days) {
042            mask += day.id();
043        }
044
045        return mask;
046    }
047}
