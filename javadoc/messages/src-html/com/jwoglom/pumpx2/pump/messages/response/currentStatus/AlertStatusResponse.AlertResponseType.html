014@MessageProps(

020public class AlertStatusResponse extends Message {
021    private BigInteger intMap;
022
023    public AlertStatusResponse() {}
024
025    public AlertStatusResponse(BigInteger intMap) {
026        this.cargo = buildCargo(intMap);
027        this.intMap = intMap;
028    }
029
030    private static byte[] buildCargo(BigInteger byte0uint64) {
031        return Bytes.toUint64(byte0uint64.longValue());
032    }
033
034    public void parse(byte[] raw) {
035        Preconditions.checkArgument(raw.length == props().size());
036        intMap = Bytes.readUint64(raw, 0);
037        cargo = raw;
038    }
039
040    public BigInteger getIntMap() {
041        return intMap;
042    }
043    public long getBitMap() {
044        return intMap.longValue();
045    }

113        private final int bitmask;
114        AlertResponseType(int bitmask) {
115            this.bitmask = bitmask;
116        }
117
118        public int bitmask() {
119            return bitmask;
120        }
121
122        public String toString() {
123            return name();
124        }
125
126        public BigInteger withBit() {
127            return BigInteger.ZERO.setBit(bitmask);
128        }
129
130        public static AlertResponseType fromSingularId(long id) {
131            for (AlertResponseType type : values()) {
132                if (type.bitmask == id) {
133                    return type;
134                }
135            }
136            return null;
137        }
138    }
139
140    public Set<AlertResponseType> getAlerts() {
141        Set<AlertResponseType> current = new HashSet<>();
142        for (AlertResponseType type : AlertResponseType.values()) {
143            if (intMap.testBit(type.bitmask())) {
144                current.add(type);
145            }
146        }
147
148        return current;
149    }
