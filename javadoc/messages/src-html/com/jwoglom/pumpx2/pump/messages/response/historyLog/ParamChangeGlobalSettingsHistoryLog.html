001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007@HistoryLogProps(
008    opCode = 74,
009    displayName = "Global Settings Change",
010    usedByTidepool = true
011)
012public class ParamChangeGlobalSettingsHistoryLog extends HistoryLog {
013    
014    private int modifiedData;
015    private int qbDataStatus;
016    private int qbActive;
017    private int qbDataEntryType;
018    private int qbIncrementUnits;
019    private int qbIncrementCarbs;
020    private int buttonVolume;
021    private int qbVolume;
022    private int bolusVolume;
023    private int reminderVolume;
024    private int alertVolume;
025    
026    public ParamChangeGlobalSettingsHistoryLog() {}
027    public ParamChangeGlobalSettingsHistoryLog(long pumpTimeSec, long sequenceNum, int modifiedData, int qbDataStatus, int qbActive, int qbDataEntryType, int qbIncrementUnits, int qbIncrementCarbs, int buttonVolume, int qbVolume, int bolusVolume, int reminderVolume, int alertVolume) {
028        super(pumpTimeSec, sequenceNum);
029        this.cargo = buildCargo(pumpTimeSec, sequenceNum, modifiedData, qbDataStatus, qbActive, qbDataEntryType, qbIncrementUnits, qbIncrementCarbs, buttonVolume, qbVolume, bolusVolume, reminderVolume, alertVolume);
030        this.modifiedData = modifiedData;
031        this.qbDataStatus = qbDataStatus;
032        this.qbActive = qbActive;
033        this.qbDataEntryType = qbDataEntryType;
034        this.qbIncrementUnits = qbIncrementUnits;
035        this.qbIncrementCarbs = qbIncrementCarbs;
036        this.buttonVolume = buttonVolume;
037        this.qbVolume = qbVolume;
038        this.bolusVolume = bolusVolume;
039        this.reminderVolume = reminderVolume;
040        this.alertVolume = alertVolume;
041        
042    }
043
044    public ParamChangeGlobalSettingsHistoryLog(int modifiedData, int qbDataStatus, int qbActive, int qbDataEntryType, int qbIncrementUnits, int qbIncrementCarbs, int buttonVolume, int qbVolume, int bolusVolume, int reminderVolume, int alertVolume) {
045        this(0, 0, modifiedData, qbDataStatus, qbActive, qbDataEntryType, qbIncrementUnits, qbIncrementCarbs, buttonVolume, qbVolume, bolusVolume, reminderVolume, alertVolume);
046    }
047
048    public int typeId() {
049        return 74;
050    }
051
052    public void parse(byte[] raw) {
053        Preconditions.checkArgument(raw.length == 26);
054        this.cargo = raw;
055        parseBase(raw);
056        this.modifiedData = raw[10];
057        this.qbDataStatus = raw[11];
058        this.qbActive = raw[12];
059        this.qbDataEntryType = raw[13];
060        this.qbIncrementUnits = Bytes.readShort(raw, 14);
061        this.qbIncrementCarbs = Bytes.readShort(raw, 16);
062        this.buttonVolume = raw[18];
063        this.qbVolume = raw[19];
064        this.bolusVolume = raw[20];
065        this.reminderVolume = raw[21];
066        this.alertVolume = raw[22];
067        
068    }
069
070    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int modifiedData, int qbDataStatus, int qbActive, int qbDataEntryType, int qbIncrementUnits, int qbIncrementCarbs, int buttonVolume, int qbVolume, int bolusVolume, int reminderVolume, int alertVolume) {
071        return HistoryLog.fillCargo(Bytes.combine(
072            new byte[]{74, 0},
073            Bytes.toUint32(pumpTimeSec),
074            Bytes.toUint32(sequenceNum),
075            new byte[]{ (byte) modifiedData }, 
076            new byte[]{ (byte) qbDataStatus }, 
077            new byte[]{ (byte) qbActive }, 
078            new byte[]{ (byte) qbDataEntryType }, 
079            Bytes.firstTwoBytesLittleEndian(qbIncrementUnits), 
080            Bytes.firstTwoBytesLittleEndian(qbIncrementCarbs), 
081            new byte[]{ (byte) buttonVolume }, 
082            new byte[]{ (byte) qbVolume }, 
083            new byte[]{ (byte) bolusVolume }, 
084            new byte[]{ (byte) reminderVolume }, 
085            new byte[]{ (byte) alertVolume }));
086    }
087    public int getModifiedData() {
088        return modifiedData;
089    }
090    public int getQbDataStatus() {
091        return qbDataStatus;
092    }
093    public int getQbActive() {
094        return qbActive;
095    }
096    public int getQbDataEntryType() {
097        return qbDataEntryType;
098    }
099    public int getQbIncrementUnits() {
100        return qbIncrementUnits;
101    }
102    public int getQbIncrementCarbs() {
103        return qbIncrementCarbs;
104    }
105    public int getButtonVolume() {
106        return buttonVolume;
107    }
108    public int getQbVolume() {
109        return qbVolume;
110    }
111    public int getBolusVolume() {
112        return bolusVolume;
113    }
114    public int getReminderVolume() {
115        return reminderVolume;
116    }
117    public int getAlertVolume() {
118        return alertVolume;
119    }
120    
121}
