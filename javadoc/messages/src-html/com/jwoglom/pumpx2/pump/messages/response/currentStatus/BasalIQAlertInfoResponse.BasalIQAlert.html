001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.BasalIQAlertInfoRequest;
009
010@MessageProps(
011    opCode=103,
012    size=4,
013    type=MessageType.RESPONSE,
014    request=BasalIQAlertInfoRequest.class
015)
016public class BasalIQAlertInfoResponse extends Message {
017    
018    private long alertId;
019    
020    public BasalIQAlertInfoResponse() {}
021    
022    public BasalIQAlertInfoResponse(long alertId) {
023        this.cargo = buildCargo(alertId);
024        this.alertId = alertId;
025        
026    }
027
028    public void parse(byte[] raw) {
029        Preconditions.checkArgument(raw.length == props().size());
030        this.cargo = raw;
031        this.alertId = Bytes.readUint32(raw, 0);
032        
033    }
034
035    
036    public static byte[] buildCargo(long alertId) {
037        return Bytes.combine(
038            Bytes.toUint32(alertId));
039    }
040    
041    public long getAlertId() {
042        return alertId;
043    }
044    public BasalIQAlert getAlert() {
045        return BasalIQAlert.fromId(alertId);
046    }
047
048    public enum BasalIQAlert {
049        NO_ALERT(0),
050        INSULIN_SUSPENDED_ALERT(24576),
051        INSULIN_RESUMED_ALERT(24577),
052        INSULIN_RESUMED_TIMEOUT(24578),
053        ;
054
055        private final long id;
056        BasalIQAlert(long id) {
057            this.id = id;
058        }
059
060        public long id() {
061            return id;
062        }
063
064        public static BasalIQAlert fromId(long id) {
065            for (BasalIQAlert a : values()) {
066                if (a.id() == id) {
067                    return a;
068                }
069            }
070            return null;
071        }
072    }
073    
074}
