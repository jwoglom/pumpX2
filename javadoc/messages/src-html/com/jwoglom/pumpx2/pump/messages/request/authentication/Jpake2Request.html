001package com.jwoglom.pumpx2.pump.messages.request.authentication;
002
003import com.jwoglom.pumpx2.pump.messages.Message;
004import com.jwoglom.pumpx2.pump.messages.MessageType;
005import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
006import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
009import com.jwoglom.pumpx2.pump.messages.response.authentication.Jpake2Response;
010
011import java.util.Arrays;
012
013/**
014 * The third authorization message sent to the pump with V2 authorization style which contains the
015 * second round of EcJpake. The message structure is identical to {@link Jpake1aRequest} and
016 * {@link Jpake1bRequest}, the second EcJpake round has half the length though so does not need to
017 * be split into two sub-messages.
018 *
019 * When the Android libary is used, this message is invoked automatically by PumpX2 on Bluetooth
020 * connection initialization as part of performing the initial pump pairing process.
021 */
022@MessageProps(
023    opCode=36,
024    size=167,
025    type=MessageType.REQUEST,
026    minApi=KnownApiVersion.API_V3_2,
027    characteristic=Characteristic.AUTHORIZATION,
028    response= Jpake2Response.class
029)
030public class Jpake2Request extends Message {
031    private int appInstanceId;
032    private byte[] centralChallenge;
033
034    public Jpake2Request() {}
035
036    public Jpake2Request(int appInstanceId, byte[] centralChallenge) {
037        this.cargo = buildCargo(appInstanceId, centralChallenge);
038        this.appInstanceId = appInstanceId;
039        this.centralChallenge = centralChallenge; // 165
040    }
041
042    public Jpake2Request(byte[] rawCargo) {
043        parse(rawCargo);
044    }
045
046    public int getAppInstanceId() {
047        return appInstanceId;
048    }
049
050    public byte[] getCentralChallenge() {
051        return centralChallenge;
052    }
053
054    private static byte[] buildCargo(int appInstanceId, byte[] centralChallenge) {
055        byte[] cargo = new byte[167];
056        System.arraycopy(Bytes.combine(Bytes.firstTwoBytesLittleEndian(appInstanceId), centralChallenge), 0, cargo, 0, 167);
057
058        return cargo;
059    }
060
061    public void parse(byte[] raw) {
062        this.cargo = raw;
063        this.appInstanceId = Bytes.readShort(Arrays.copyOfRange(raw, 0, 2), 0);
064        this.centralChallenge = Arrays.copyOfRange(raw, 2, raw.length);
065    }
066}
