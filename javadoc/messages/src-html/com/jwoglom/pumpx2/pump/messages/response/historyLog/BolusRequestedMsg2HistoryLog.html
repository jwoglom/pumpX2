001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007@HistoryLogProps(
008    opCode = 65,
009    displayName = "Bolus Requested 2/3",
010    usedByAndroid = true,
011    usedByTidepool = true
012)
013public class BolusRequestedMsg2HistoryLog extends HistoryLog {
014    
015    private int bolusId;
016    private int options;
017    private int standardPercent;
018    private int duration;
019    private int spare1;
020    private int isf;
021    private int targetBG;
022    private boolean userOverride;
023    private boolean declinedCorrection;
024    private int selectedIOB;
025    private int spare2;
026    
027    public BolusRequestedMsg2HistoryLog() {}
028    
029    public BolusRequestedMsg2HistoryLog(long pumpTimeSec, long sequenceNum, int bolusId, int options, int standardPercent, int duration, int spare1, int isf, int targetBG, boolean userOverride, boolean declinedCorrection, int selectedIOB, int spare2) {
030        this.cargo = buildCargo(pumpTimeSec, sequenceNum, bolusId, options, standardPercent, duration, spare1, isf, targetBG, userOverride, declinedCorrection, selectedIOB, spare2);
031        this.pumpTimeSec = pumpTimeSec;
032        this.sequenceNum = sequenceNum;
033        this.bolusId = bolusId;
034        this.options = options;
035        this.standardPercent = standardPercent;
036        this.duration = duration;
037        this.spare1 = spare1;
038        this.isf = isf;
039        this.targetBG = targetBG;
040        this.userOverride = userOverride;
041        this.declinedCorrection = declinedCorrection;
042        this.selectedIOB = selectedIOB;
043        this.spare2 = spare2;
044    }
045
046    public BolusRequestedMsg2HistoryLog(int bolusId, int options, int standardPercent, int duration, int spare1, int isf, int targetBG, boolean userOverride, boolean declinedCorrection, int selectedIOB, int spare2) {
047        this(0, 0, bolusId, options, standardPercent, duration, spare1, isf, targetBG, userOverride, declinedCorrection, selectedIOB, spare2);
048    }
049
050    public int typeId() {
051        return 65;
052    }
053
054    public void parse(byte[] raw) {
055        Preconditions.checkArgument(raw.length == 26);
056        this.cargo = raw;
057        parseBase(raw);
058        this.bolusId = Bytes.readShort(raw, 10);
059        this.options = raw[12];
060        this.standardPercent = raw[13];
061        this.duration = Bytes.readShort(raw, 14);
062        this.spare1 = Bytes.readShort(raw, 16);
063        this.isf = Bytes.readShort(raw, 18);
064        this.targetBG = Bytes.readShort(raw, 20);
065        this.userOverride = raw[22] != 0;
066        this.declinedCorrection = raw[23] != 0;
067        this.selectedIOB = raw[24];
068        this.spare2 = raw[1];
069        
070    }
071
072    
073    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int bolusId, int options, int standardPercent, int duration, int spare1, int isf, int targetBG, boolean userOverride, boolean declinedCorrection, int selectedIOB, int spare2) {
074        return Bytes.combine(
075            new byte[] { (byte) 65, 0},
076            Bytes.toUint32(pumpTimeSec),
077            Bytes.toUint32(sequenceNum),
078            Bytes.firstTwoBytesLittleEndian(bolusId), 
079            new byte[]{ (byte) options }, 
080            new byte[]{ (byte) standardPercent }, 
081            Bytes.firstTwoBytesLittleEndian(duration), 
082            Bytes.firstTwoBytesLittleEndian(spare1), 
083            Bytes.firstTwoBytesLittleEndian(isf), 
084            Bytes.firstTwoBytesLittleEndian(targetBG), 
085            new byte[]{ (byte) (userOverride ? 1 : 0) }, 
086            new byte[]{ (byte) (declinedCorrection ? 1 : 0) }, 
087            new byte[]{ (byte) selectedIOB }, 
088            new byte[]{ (byte) spare2 });
089    }
090
091    /**
092     * @return bolus ID
093     */
094    public int getBolusId() {
095        return bolusId;
096    }
097    public int getOptions() {
098        return options;
099    }
100
101    /**
102     * @return 100 if a standard bolus, otherwise the percent of the bolus which is performed up
103     * front in an extended bolus
104     */
105    public int getStandardPercent() {
106        return standardPercent;
107    }
108
109    /**
110     * @return 0 if a standard bolus, otherwise the duration between the standard and extended bolus
111     */
112    public int getDuration() {
113        return duration;
114    }
115
116    public int getSpare1() {
117        return spare1;
118    }
119
120    /**
121     * @return insulin sensitivity factor in the insulin delivery profile
122     */
123    public int getIsf() {
124        return isf;
125    }
126
127    /**
128     * @return target BG in mg/dL in the insulin delivery profile
129     */
130    public int getTargetBG() {
131        return targetBG;
132    }
133
134    /**
135     * @return true if the user overrode the insulin delivery amount (i.e., not calculated just
136     * from the combination of carbs + BG correction)
137     */
138    public boolean getUserOverride() {
139        return userOverride;
140    }
141
142    /**
143     * @return true if the correction amount due to the current BG+IOB was declined
144     */
145    public boolean getDeclinedCorrection() {
146        return declinedCorrection;
147    }
148
149    /**
150     * @return TODO(unknown)
151     */
152    public int getSelectedIOB() {
153        return selectedIOB;
154    }
155    public int getSpare2() {
156        return spare2;
157    }
158    
159}
