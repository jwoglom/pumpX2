003import com.google.common.base.Preconditions;

009@HistoryLogProps(
010    opCode = 198,
011    displayName = "Hypo Minimizer Suspend"
012)
013public class HypoMinimizerSuspendHistoryLog extends HistoryLog {
014    
015    
016    public HypoMinimizerSuspendHistoryLog() {}
017    public HypoMinimizerSuspendHistoryLog(long pumpTimeSec, long sequenceNum) {
018        super(pumpTimeSec, sequenceNum);
019        this.cargo = buildCargo(pumpTimeSec, sequenceNum);
020        
021    }
022
023    public int typeId() {
024        return 198;
025    }
026
027    public void parse(byte[] raw) {
028        Preconditions.checkArgument(raw.length == 26);
029        this.cargo = raw;
030        parseBase(raw);
031        
032    }
033
034    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum) {
035        return HistoryLog.fillCargo(Bytes.combine(
036            new byte[]{-58, 0}, // (byte) 198
037            Bytes.toUint32(pumpTimeSec),
038            Bytes.toUint32(sequenceNum)));
039    }
040
