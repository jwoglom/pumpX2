017    // Streams don't have transaction IDs
018    public StreamPacketArrayList(byte expectedopCode, byte expectedCargoSize, byte expectedTxId, boolean isSigned) {
019        super(expectedopCode, expectedCargoSize, (byte) 0, isSigned);
020    }
021
022    public void validatePacket(byte[] packetData) {
023        Intrinsics.checkParameterIsNotNull(packetData, "packetData");
024        if (packetData.length == 0) {
025            throw new IllegalArgumentException("Empty data");
026        } else if (packetData.length >= 3) {
027            byte firstByteMod15 = (byte) (packetData[0] & 15);
028            byte secondByte = packetData[1];
029            L.d(TAG, "Found tx id: "+secondByte+" expected "+this.expectedTxId);
030            if (secondByte == this.expectedTxId) {
031                if (this.empty) {
032                    this.firstByteMod15 = firstByteMod15;
033                    L.d(TAG, "txid matches, firstByteMod15="+firstByteMod15+", parsing packetData");
034                    parse(packetData);
035                } else if (((byte) 0) == firstByteMod15) {
036                    L.d(TAG, "firstByteMod15=0");
037                    if (this.firstByteMod15 == firstByteMod15) {
038                        this.fullCargo = Bytes.combine(this.fullCargo, Bytes.dropFirstN(packetData, 2));
039                        L.d(TAG, "firstByteMod15 matches expected, fullCargo="+ Hex.encodeHexString(fullCargo));
040                    } else {
041                        throw new IllegalArgumentException("Unexpected packets remaining 3: " + ((int) firstByteMod15) + ", expected " + ((int) this.firstByteMod15) + ", opCode: " + ((int) this.expectedOpCode));
042                    }
043                } else if (this.firstByteMod15 == firstByteMod15) {
044                    this.fullCargo = Bytes.combine(this.fullCargo, Bytes.dropFirstN(packetData, 2));
045                    L.d(TAG, "firstByteMod15 matches expected, nonzero, fullCargo=" + Hex.encodeHexString(fullCargo));
046                } else {
047                    throw new IllegalArgumentException("Unexpected packets remaining 2: " + ((int) firstByteMod15) + ", expected " + ((int) this.firstByteMod15) + ", opCode: " + ((int) this.expectedOpCode));
048                }
049                //if (!moreHistoryLogsRemaining()) {
050                if (!needsMorePacket()) {
051                    L.d(TAG, String.format("Completed parsing historyLog: %s", Hex.encodeHexString(this.messageData)));
052                    this.empty = true;
053                }
054                this.firstByteMod15 = (byte) (this.firstByteMod15 - 1);
055                return;
056            }
057            throw new IllegalArgumentException("Unexpected transaction ID 1: " + ((int) secondByte) + ", expecting " + ((int) this.expectedTxId) + ", opCode: " + ((int) this.expectedOpCode));
058        } else {
059            throw new IllegalArgumentException("Invalid data size: " + packetData.length);
060        }
061    }
062
063    protected void parse(byte[] bArr) {
064        L.d(TAG, String.format("Parsing historyLog byte array: %s", Hex.encodeHexString(bArr)));
065        byte opCode = bArr[2];
066        byte cargoSize = bArr[4];
067        if (opCode == this.expectedOpCode) {
068            this.firstByteMod15 = (byte) (bArr[0] & 15);
069            this.opCode = bArr[2];
070            byte txId = bArr[3];
071
072            // Replace cargo size
073            expectedCargoSize = cargoSize;
074            if (txId != this.expectedTxId) {
075                throw new UnexpectedTransactionIdException(txId, this.expectedTxId, this.expectedOpCode);
076            } else if (cargoSize <= 255) {
077                byte numHistoryLogs = bArr[5];
078                if (cargoSize == (numHistoryLogs*26) + 2) {
079                    this.fullCargo = Bytes.dropFirstN(bArr, 5);
080                    this.empty = false;
081                    return;
082                }
083                throw new IllegalArgumentException("Cargo size doesn't match number of history logs requested");
084            } else {
085                throw new IllegalArgumentException("Cargo size beyond maximum: " + ((int) cargoSize));
086            }
087        } else {
088            throw new UnexpectedOpCodeException(opCode, this.expectedOpCode);
089        }
090    }
091
092
093    private boolean moreHistoryLogsRemaining() {
094        if (firstByteMod15 == 0) {
095            return !validate("");
096        }
097        return firstByteMod15 > 0;
098    }

110    private void setExpectedCRC() {
111        if (this.fullCargo.length >= 2) {
112            System.arraycopy(this.fullCargo, this.fullCargo.length - 2, this.expectedCrc, 0, 2);
113        }
114    }
115
116    public boolean validate(String authKey) {
117        createMessageData();
118        setExpectedCRC();
119
120        byte[] crcOut = Bytes.calculateCRC16(messageData);
121        if (crcOut[0] == expectedCrc[0] && crcOut[1] == expectedCrc[1]) {
122            return true;
123        }
124
125        throw new IllegalArgumentException("CRC error with history log: originalMessageData: "+ Hex.encodeHexString(originalMessageData)+" messageData: "+Hex.encodeHexString(messageData)+" expectedCrc: "+Hex.encodeHexString(expectedCrc)+" crcOut: "+Hex.encodeHexString(crcOut));

127    }
