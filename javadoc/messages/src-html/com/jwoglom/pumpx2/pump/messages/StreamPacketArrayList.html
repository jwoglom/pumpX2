001package com.jwoglom.pumpx2.pump.messages;
002
003import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
004import com.jwoglom.pumpx2.pump.messages.models.UnexpectedOpCodeException;
005import com.jwoglom.pumpx2.pump.messages.models.UnexpectedTransactionIdException;
006import com.jwoglom.pumpx2.shared.L;
007
008import com.jwoglom.pumpx2.shared.Hex;
009
010import kotlin.jvm.internal.Intrinsics;
011
012public class StreamPacketArrayList extends PacketArrayList {
013    protected static final String TAG = "StreamPacketArrayList";
014    private static final int HISTORY_LOG_STREAM_OPCODE = -127;
015
016    private byte[] originalMessageData = new byte[0];
017
018    // Streams sometimes(?) have transaction IDs
019    public StreamPacketArrayList(byte expectedopCode, byte expectedCargoSize, byte expectedTxId, boolean isSigned) {
020        super(expectedopCode, expectedCargoSize, expectedTxId, isSigned);
021    }
022
023    public void validatePacket(byte[] packetData) {
024        Intrinsics.checkParameterIsNotNull(packetData, "packetData");
025        if (packetData.length == 0) {
026            throw new IllegalArgumentException("Empty data");
027        } else if (packetData.length >= 3) {
028            byte firstByteMod15 = (byte) (packetData[0] & 15);
029            byte secondByte = packetData[1];
030            L.d(TAG, "Found tx id: "+secondByte+" expected "+this.expectedTxId);
031            if (secondByte == this.expectedTxId) {
032                if (this.empty) {
033                    this.firstByteMod15 = firstByteMod15;
034                    L.d(TAG, "txid matches, firstByteMod15="+firstByteMod15+", parsing packetData");
035                    parse(packetData);
036                } else if (((byte) 0) == firstByteMod15) {
037                    L.d(TAG, "firstByteMod15=0");
038                    if (this.firstByteMod15 == firstByteMod15) {
039                        this.fullCargo = Bytes.combine(this.fullCargo, Bytes.dropFirstN(packetData, 2));
040                        L.d(TAG, "firstByteMod15 matches expected, fullCargo="+ Hex.encodeHexString(fullCargo));
041                    } else {
042                        throw new IllegalArgumentException("Unexpected packets remaining 3: " + ((int) firstByteMod15) + ", expected " + ((int) this.firstByteMod15) + ", opCode: " + ((int) this.expectedOpCode));
043                    }
044                } else if (this.firstByteMod15 == firstByteMod15) {
045                    this.fullCargo = Bytes.combine(this.fullCargo, Bytes.dropFirstN(packetData, 2));
046                    L.d(TAG, "firstByteMod15 matches expected, nonzero, fullCargo=" + Hex.encodeHexString(fullCargo));
047                } else {
048                    throw new IllegalArgumentException("Unexpected packets remaining 2: " + ((int) firstByteMod15) + ", expected " + ((int) this.firstByteMod15) + ", opCode: " + ((int) this.expectedOpCode));
049                }
050                //if (!moreHistoryLogsRemaining()) {
051                if (!needsMorePacket()) {
052                    L.d(TAG, String.format("Completed parsing historyLog: %s", Hex.encodeHexString(this.messageData)));
053                    this.empty = true;
054                }
055                this.firstByteMod15 = (byte) (this.firstByteMod15 - 1);
056                return;
057            }
058            throw new IllegalArgumentException("Unexpected transaction ID 1: " + ((int) secondByte) + ", expecting " + ((int) this.expectedTxId) + ", opCode: " + ((int) this.expectedOpCode));
059        } else {
060            throw new IllegalArgumentException("Invalid data size: " + packetData.length);
061        }
062    }
063
064    protected void parse(byte[] bArr) {
065        L.d(TAG, String.format("Parsing stream byte array opCode=%d cargoSize=%d: %s", bArr[2], bArr[4], Hex.encodeHexString(bArr)));
066        byte opCode = bArr[2];
067        byte cargoSize = bArr[4];
068        if (opCode == this.expectedOpCode) {
069            this.firstByteMod15 = (byte) (bArr[0] & 15);
070            this.opCode = bArr[2];
071            byte txId = bArr[3];
072
073            if (txId != this.expectedTxId) {
074                throw new UnexpectedTransactionIdException(txId, this.expectedTxId, this.expectedOpCode);
075            } else if (cargoSize != this.actualExpectedCargoSize) {
076                if (cargoSize == this.actualExpectedCargoSize + 24 && isSigned) {
077                    L.i(TAG, "adding +24 expectedCargoSize for already signed request which contains an existing trailer");
078                    this.expectedCargoSize += 24;
079                    this.actualExpectedCargoSize += 24;
080                } else if (opCode != HISTORY_LOG_STREAM_OPCODE) {
081                    throw new IllegalArgumentException("Unexpected cargo size: " + ((int) cargoSize) + ", expecting " + ((int) this.actualExpectedCargoSize) + " for opCode="+opCode);
082                }
083            }
084            this.fullCargo = Bytes.dropFirstN(bArr, 5);
085
086            // specific checks for HistoryLog stream
087            if ((byte) opCode == HISTORY_LOG_STREAM_OPCODE) {
088                if (cargoSize <= 255) {
089                    byte numHistoryLogs = bArr[5];
090                    L.d(TAG, "StreamPacketArrayParse check: opCode="+opCode+" numHistoryLogs="+numHistoryLogs+" expHistoryLogs="+((numHistoryLogs * 26) + 2)+" cargoSize="+cargoSize);
091                    L.d(TAG, "StreamPacketArrayParse bArr="+Hex.encodeHexString(bArr)+" oldFullCargo="+Hex.encodeHexString(fullCargo)+" newFullCargo="+Hex.encodeHexString(Bytes.dropFirstN(bArr, 5)));
092                    if (cargoSize != (numHistoryLogs * 26) + 2) {
093                        throw new IllegalArgumentException("Cargo size doesn't match number of history logs requested");
094                    } else {
095                        expectedCargoSize = cargoSize;
096                    }
097                } else {
098                    throw new IllegalArgumentException("Cargo size beyond maximum: " + ((int) cargoSize));
099                }
100
101            }
102        } else {
103            throw new UnexpectedOpCodeException(opCode, this.expectedOpCode);
104        }
105
106        L.d(TAG, "StreamPacketArrayParse ok: opCode="+opCode+" firstByteMod15="+firstByteMod15+" cargoSize="+cargoSize);
107
108    }
109
110}
