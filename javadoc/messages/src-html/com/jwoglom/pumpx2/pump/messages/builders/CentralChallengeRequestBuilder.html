001package com.jwoglom.pumpx2.pump.messages.builders;
002
003import com.jwoglom.pumpx2.pump.messages.bluetooth.PumpStateSupplier;
004import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
005import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
006import com.jwoglom.pumpx2.pump.messages.request.authentication.CentralChallengeRequest;
007import com.jwoglom.pumpx2.pump.messages.request.authentication.AbstractCentralChallengeRequest;
008
009public class CentralChallengeRequestBuilder {
010    public static AbstractCentralChallengeRequest create(int appInstanceId) {
011        if (PumpStateSupplier.pumpApiVersion.get().greaterThan(KnownApiVersion.API_V3)) {
012            return null; // Handled inside pair()
013        } else {
014            return createV1(appInstanceId);
015        }
016    }
017
018    private static AbstractCentralChallengeRequest createV1(int appInstanceId) {
019        // TODO: this should be 8 instead of 10, we cut off 2 bytes at the end for no reason
020        byte[] centralChallenge = Bytes.getSecureRandom10Bytes();
021        return new CentralChallengeRequest(appInstanceId, centralChallenge);
022    }
023}
