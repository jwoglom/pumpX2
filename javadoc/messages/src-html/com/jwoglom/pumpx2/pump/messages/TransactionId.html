001package com.jwoglom.pumpx2.pump.messages;
002
003import com.jwoglom.pumpx2.shared.L;
004
005/**
006 * Represents the NEXT ID which can be used for a transaction in communicating with the pump.
007 * We read the current txId, send a message with it, and then increment it.
008 */
009public class TransactionId {
010    private static final String TAG = "TransactionId";
011    private int txId = 0;
012
013    public void set(int i) {
014        if (i < 0 || i > 255) {
015            i = 0;
016        }
017        if (i != txId + 1) {
018            L.d(TAG, "set: old " + txId + " new " + i);
019        }
020        txId = i;
021    }
022
023    public void increment() {
024        set(txId + 1);
025    }
026
027    public void reset() {
028        set(0);
029    }
030
031    public byte get() {
032        return (byte) txId;
033    }
034
035    public String toString() {
036        return "TransactionId(txId="+txId+")";
037    }
038}
