001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007import java.math.BigInteger;
008
009@HistoryLogProps(
010    opCode = 230, // -26
011    displayName = "ControlIQ Pump Control Mode (PCM) Change",
012    usedByTidepool = true // LID_AA_PCM_CHANGE
013)
014public class ControlIQPcmChangeHistoryLog extends HistoryLog {
015    
016    private int currentPcm;
017    private int previousPcm;
018    
019    public ControlIQPcmChangeHistoryLog() {}
020    public ControlIQPcmChangeHistoryLog(long pumpTimeSec, long sequenceNum, int currentPcm, int previousPcm) {
021        super(pumpTimeSec, sequenceNum);
022        this.cargo = buildCargo(pumpTimeSec, sequenceNum, currentPcm, previousPcm);
023        this.currentPcm = currentPcm;
024        this.previousPcm = previousPcm;
025        
026    }
027
028    public ControlIQPcmChangeHistoryLog(int currentPcm, int previousPcm) {
029        this(0, 0, currentPcm, previousPcm);
030    }
031
032    public int typeId() {
033        return 230;
034    }
035
036    public void parse(byte[] raw) {
037        Preconditions.checkArgument(raw.length == 26);
038        this.cargo = raw;
039        parseBase(raw);
040        this.currentPcm = raw[10];
041        this.previousPcm = raw[11];
042        
043    }
044
045    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int currentPcm, int previousPcm) {
046        return HistoryLog.fillCargo(Bytes.combine(
047            new byte[]{-26, 0}, // (byte) 230
048            Bytes.toUint32(pumpTimeSec),
049            Bytes.toUint32(sequenceNum),
050            new byte[]{ (byte) currentPcm }, 
051            new byte[]{ (byte) previousPcm }));
052    }
053    public int getCurrentPcm() {
054        return currentPcm;
055    }
056    public int getPreviousPcm() {
057        return previousPcm;
058    }
059    
060}
