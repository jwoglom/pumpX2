044    public static byte[] combine(byte[] ...items) {
045        int sum = 0;
046        for (byte[] item : items) {
047            sum += item.length;
048        }
049
050        byte[] ret = new byte[sum];
051        int i = 0;
052        for (byte[] item : items) {
053            for (byte b : item) {
054                ret[i] = b;
055                i++;
056            }
057        }
058
059        return ret;
060    }

140    public static String readString(byte[] raw, int i, int length) {
141        if (i >= 0 && i < raw.length) {
142            byte[] strBytes = new byte[0];
143            byte b;
144            while (i < raw.length && (b = raw[i]) > 0) {
145                strBytes = combine(strBytes, new byte[]{b});
146                if (strBytes.length >= length) {
147                    break;
148                }
149                i++;
150            }
151            return new String(strBytes, Charsets.UTF_8);
152        }
153        throw new ArrayIndexOutOfBoundsException(i);
154    }
