001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.MessageType;
005import com.jwoglom.pumpx2.pump.messages.annotations.ApiVersionDependent;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
009import com.jwoglom.pumpx2.pump.messages.request.currentStatus.CurrentBatteryV2Request;
010
011/**
012 * Battery information from the pump. The additional data from the V2 API is not known, and it
013 * always seems to have value 0.
014 * For API_V2_5
015 *
016 * To get the standard battery percent use getCurrentBatteryIbc() or getBatteryPercent() as a convenience
017 *
018 * Use {@link com.jwoglom.pumpx2.pump.messages.response.currentStatus.CurrentBatteryAbstractResponse}
019 */
020@MessageProps(
021    opCode=-111,
022    size=11,
023    type=MessageType.RESPONSE,
024    request=CurrentBatteryV2Request.class,
025    minApi=KnownApiVersion.API_V2_5
026)
027@ApiVersionDependent
028public class CurrentBatteryV2Response extends CurrentBatteryAbstractResponse {
029    
030    private int currentBatteryAbc;
031    private int currentBatteryIbc;
032    private int chargingStatus;
033    private int unknown1;
034    private int unknown2;
035    private int unknown3;
036    private int unknown4;
037    
038    public CurrentBatteryV2Response() {}
039    
040    public CurrentBatteryV2Response(int currentBatteryAbc, int currentBatteryIbc, int chargingStatus, int unknown1, int unknown2, int unknown3, int unknown4) {
041        this.cargo = buildCargo(currentBatteryAbc, currentBatteryIbc, chargingStatus, unknown1, unknown2, unknown3, unknown4);
042        this.currentBatteryAbc = currentBatteryAbc;
043        this.currentBatteryIbc = currentBatteryIbc;
044        this.chargingStatus = chargingStatus;
045        this.unknown1 = unknown1;
046        this.unknown2 = unknown2;
047        this.unknown3 = unknown3;
048        this.unknown4 = unknown4;
049        
050    }
051
052    public void parse(byte[] raw) {
053        Preconditions.checkArgument(raw.length == props().size());
054        this.cargo = raw;
055        this.currentBatteryAbc = raw[0];
056        this.currentBatteryIbc = raw[1];
057        this.chargingStatus = raw[2];
058        this.unknown1 = Bytes.readShort(raw, 3);
059        this.unknown2 = Bytes.readShort(raw, 5);
060        this.unknown3 = Bytes.readShort(raw, 7);
061        this.unknown4 = Bytes.readShort(raw, 9);
062        
063    }
064
065    
066    public static byte[] buildCargo(int currentBatteryAbc, int currentBatteryIbc, int chargingStatus, int unknown1, int unknown2, int unknown3, int unknown4) {
067        return Bytes.combine(
068            new byte[]{ (byte) currentBatteryAbc }, 
069            new byte[]{ (byte) currentBatteryIbc }, 
070            new byte[]{ (byte) chargingStatus }, 
071            Bytes.firstTwoBytesLittleEndian(unknown1), 
072            Bytes.firstTwoBytesLittleEndian(unknown2), 
073            Bytes.firstTwoBytesLittleEndian(unknown3), 
074            Bytes.firstTwoBytesLittleEndian(unknown4));
075    }
076    
077    public int getCurrentBatteryAbc() {
078        return currentBatteryAbc;
079    }
080    public int getCurrentBatteryIbc() {
081        return currentBatteryIbc;
082    }
083    public int getChargingStatus() {
084        return chargingStatus;
085    }
086
087    public boolean isCharging() {
088        return chargingStatus == 1;
089    }
090
091    public int getUnknown1() {
092        return unknown1;
093    }
094    public int getUnknown2() {
095        return unknown2;
096    }
097    public int getUnknown3() {
098        return unknown3;
099    }
100    public int getUnknown4() {
101        return unknown4;
102    }
103}
