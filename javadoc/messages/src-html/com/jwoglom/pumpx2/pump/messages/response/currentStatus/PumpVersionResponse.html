001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.PumpVersionRequest;
009
010@MessageProps(
011    opCode=85,
012    size=48,
013    type=MessageType.RESPONSE,
014    request=PumpVersionRequest.class
015)
016public class PumpVersionResponse extends Message {
017    
018    private long armSwVer;
019    private long mspSwVer;
020    private long configABits;
021    private long configBBits;
022    private long serialNum;
023    private long partNum;
024    private String pumpRev;
025    private long pcbaSN;
026    private String pcbaRev;
027    private long modelNum;
028    
029    public PumpVersionResponse() {}
030    
031    public PumpVersionResponse(long armSwVer, long mspSwVer, long configABits, long configBBits, long serialNum, long partNum, String pumpRev, long pcbaSN, String pcbaRev, long modelNum) {
032        this.cargo = buildCargo(armSwVer, mspSwVer, configABits, configBBits, serialNum, partNum, pumpRev, pcbaSN, pcbaRev, modelNum);
033        this.armSwVer = armSwVer;
034        this.mspSwVer = mspSwVer;
035        this.configABits = configABits;
036        this.configBBits = configBBits;
037        this.serialNum = serialNum;
038        this.partNum = partNum;
039        this.pumpRev = pumpRev;
040        this.pcbaSN = pcbaSN;
041        this.pcbaRev = pcbaRev;
042        this.modelNum = modelNum;
043        
044    }
045
046    public void parse(byte[] raw) {
047        Preconditions.checkArgument(raw.length == props().size());
048        this.cargo = raw;
049        this.armSwVer = Bytes.readUint32(raw, 0);
050        this.mspSwVer = Bytes.readUint32(raw, 4);
051        this.configABits = Bytes.readUint32(raw, 8);
052        this.configBBits = Bytes.readUint32(raw, 12);
053        this.serialNum = Bytes.readUint32(raw, 16);
054        this.partNum = Bytes.readUint32(raw, 20);
055        this.pumpRev = Bytes.readString(raw, 24, 8);
056        this.pcbaSN = Bytes.readUint32(raw, 32);
057        this.pcbaRev = Bytes.readString(raw, 36, 8);
058        this.modelNum = Bytes.readUint32(raw, 44);
059        
060    }
061
062    
063    public static byte[] buildCargo(long armSwVer, long mspSwVer, long configABits, long configBBits, long serialNum, long partNum, String pumpRev, long pcbaSN, String pcbaRev, long modelNum) {
064        return Bytes.combine(
065            Bytes.toUint32(armSwVer), 
066            Bytes.toUint32(mspSwVer), 
067            Bytes.toUint32(configABits), 
068            Bytes.toUint32(configBBits), 
069            Bytes.toUint32(serialNum), 
070            Bytes.toUint32(partNum), 
071            Bytes.writeString(pumpRev, 8), 
072            Bytes.toUint32(pcbaSN), 
073            Bytes.writeString(pcbaRev, 8), 
074            Bytes.toUint32(modelNum));
075    }
076    
077    public long getArmSwVer() {
078        return armSwVer;
079    }
080    public long getMspSwVer() {
081        return mspSwVer;
082    }
083    public long getConfigABits() {
084        return configABits;
085    }
086    public long getConfigBBits() {
087        return configBBits;
088    }
089    public long getSerialNum() {
090        return serialNum;
091    }
092    public long getPartNum() {
093        return partNum;
094    }
095    public String getPumpRev() {
096        return pumpRev;
097    }
098    public long getPcbaSN() {
099        return pcbaSN;
100    }
101    public String getPcbaRev() {
102        return pcbaRev;
103    }
104    public long getModelNum() {
105        return modelNum;
106    }
107    
108}
