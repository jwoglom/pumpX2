001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.helpers.Dates;
007
008import java.time.Instant;
009
010@HistoryLogProps(
011    opCode = 14,
012    displayName = "Date Change",
013    usedByAndroid = true,
014    usedByTidepool = true
015)
016public class DateChangeHistoryLog extends HistoryLog {
017    
018    private long datePrior;
019    private long dateAfter;
020    private long rawRTCTime;
021    
022    public DateChangeHistoryLog() {}
023    
024    public DateChangeHistoryLog(long pumpTimeSec, long sequenceNum, long datePrior, long dateAfter, long rawRTCTime) {
025        this.cargo = buildCargo(pumpTimeSec, sequenceNum, datePrior, dateAfter, rawRTCTime);
026        this.pumpTimeSec = pumpTimeSec;
027        this.sequenceNum = sequenceNum;
028        this.datePrior = datePrior;
029        this.dateAfter = dateAfter;
030        this.rawRTCTime = rawRTCTime;
031        
032    }
033
034    public int typeId() {
035        return 14;
036    }
037
038    public void parse(byte[] raw) {
039        Preconditions.checkArgument(raw.length == 26);
040        this.cargo = raw;
041        parseBase(raw);
042        this.datePrior = Bytes.readUint32(raw, 10);
043        this.dateAfter = Bytes.readUint32(raw, 14);
044        this.rawRTCTime = Bytes.readUint32(raw, 18);
045        
046    }
047
048    
049    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, long datePrior, long dateAfter, long rawRTCTime) {
050        return Bytes.combine(
051            new byte[]{14, 0},
052            Bytes.toUint32(pumpTimeSec),
053            Bytes.toUint32(sequenceNum),
054            Bytes.toUint32(datePrior), 
055            Bytes.toUint32(dateAfter), 
056            Bytes.toUint32(rawRTCTime),
057            new byte[4]);
058    }
059    
060    public long getDatePrior() {
061        return datePrior;
062    }
063    public long getDateAfter() {
064        return dateAfter;
065    }
066    public long getRawRTCTime() {
067        return rawRTCTime;
068    }
069
070    public Instant getDatePriorInstant() {
071        return Dates.fromJan12008EpochDaysToDate(datePrior);
072    }
073
074    public Instant getDateAfterInstant() {
075        return Dates.fromJan12008EpochDaysToDate(dateAfter);
076    }
077}
