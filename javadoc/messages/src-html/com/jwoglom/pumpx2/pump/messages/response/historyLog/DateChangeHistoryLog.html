001package com.jwoglom.pumpx2.pump.messages.response.historyLog;

003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;

010@HistoryLogProps(
011    opCode = 14,
012    displayName = "Date Change",
013    usedByAndroid = true,
014    usedByTidepool = true
015)
016public class DateChangeHistoryLog extends HistoryLog {
017    
018    private long datePrior;
019    private long dateAfter;
020    private long rawRTCTime;
021    
022    public DateChangeHistoryLog() {}
023    
024    public DateChangeHistoryLog(long pumpTimeSec, long sequenceNum, long datePrior, long dateAfter, long rawRTCTime) {
025        this.cargo = buildCargo(pumpTimeSec, sequenceNum, datePrior, dateAfter, rawRTCTime);
026        this.pumpTimeSec = pumpTimeSec;
027        this.sequenceNum = sequenceNum;
028        this.datePrior = datePrior;
029        this.dateAfter = dateAfter;
030        this.rawRTCTime = rawRTCTime;
031        
032    }
033
034    public int typeId() {
035        return 14;
036    }
037
038    /**
039     * <table border="1">
040     * <caption>Parsed fields for DateChangeHistoryLog:</caption>
041     * <tr><td><code>datePrior</code></td><td><code>Bytes.readUint32(raw, 10)</code></td></tr>
042     * <tr><td><code>dateAfter</code></td><td><code>Bytes.readUint32(raw, 14)</code></td></tr>
043     * <tr><td><code>rawRTCTime</code></td><td><code>Bytes.readUint32(raw, 18)</code></td></tr>
044     * </table>
045     */
046    public void parse(byte[] raw) {
047        Preconditions.checkArgument(raw.length == 26);
048        this.cargo = raw;
049        parseBase(raw);
050        this.datePrior = Bytes.readUint32(raw, 10);
051        this.dateAfter = Bytes.readUint32(raw, 14);
052        this.rawRTCTime = Bytes.readUint32(raw, 18);
053        
054    }
055
056

083     * </table>
084     */
085    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, long datePrior, long dateAfter, long rawRTCTime) {
086        return Bytes.combine(
087            new byte[]{14, 0},
088            Bytes.toUint32(pumpTimeSec),
089            Bytes.toUint32(sequenceNum),
090            Bytes.toUint32(datePrior), 
091            Bytes.toUint32(dateAfter), 
092            Bytes.toUint32(rawRTCTime),
093            new byte[4]);
094    }
095    
096    public long getDatePrior() {
097        return datePrior;
098    }
099    public long getDateAfter() {
100        return dateAfter;
101    }
102    public long getRawRTCTime() {
103        return rawRTCTime;
104    }
105
106    public Instant getDatePriorInstant() {
107        return Dates.fromJan12008EpochDaysToDate(datePrior);
108    }
109
110    public Instant getDateAfterInstant() {
111        return Dates.fromJan12008EpochDaysToDate(dateAfter);
112    }
