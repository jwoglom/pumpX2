001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007import java.util.Set;
008
009@HistoryLogProps(
010    opCode = 64,
011    displayName = "Bolus Requested 1/3",
012    usedByAndroid = true,
013    usedByTidepool = true
014)
015public class BolusRequestedMsg1HistoryLog extends HistoryLog {
016    
017    private int bolusId;
018    private int bolusTypeId;
019    private boolean correctionBolusIncluded;
020    private int carbAmount;
021    private int bg;
022    private float iob;
023    private long carbRatio;
024    
025    public BolusRequestedMsg1HistoryLog() {}
026    
027    public BolusRequestedMsg1HistoryLog(long pumpTimeSec, long sequenceNum, int bolusId, int bolusTypeId, boolean correctionBolusIncluded, int carbAmount, int bg, float iob, long carbRatio) {
028        super(pumpTimeSec, sequenceNum);
029        this.cargo = buildCargo(pumpTimeSec, sequenceNum, bolusId, bolusTypeId, correctionBolusIncluded, carbAmount, bg, iob, carbRatio);
030        this.bolusId = bolusId;
031        this.bolusTypeId = bolusTypeId; // probably the same bolusType enum as BolusDeliveryHistoryLog
032        this.correctionBolusIncluded = correctionBolusIncluded;
033        this.carbAmount = carbAmount;
034        this.bg = bg;
035        this.iob = iob;
036        this.carbRatio = carbRatio;
037        
038    }
039
040    public int typeId() {
041        return 64;
042    }
043
044    public void parse(byte[] raw) {
045        Preconditions.checkArgument(raw.length == 26);
046        this.cargo = raw;
047        parseBase(raw);
048        this.bolusId = Bytes.readShort(raw, 10);
049        this.bolusTypeId = raw[12];
050        this.correctionBolusIncluded = raw[13] != 0;
051        this.carbAmount = Bytes.readShort(raw, 14);
052        this.bg = Bytes.readShort(raw, 16);
053        this.iob = Bytes.readFloat(raw, 18);
054        this.carbRatio = Bytes.readUint32(raw, 22);
055        
056    }
057
058    
059    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int bolusId, int bolusType, boolean correctionBolusIncluded, int carbAmount, int bg, float iob, long carbRatio) {
060        return Bytes.combine(
061            new byte[]{64, 0},
062            Bytes.toUint32(pumpTimeSec),
063            Bytes.toUint32(sequenceNum),
064            Bytes.firstTwoBytesLittleEndian(bolusId), 
065            new byte[]{ (byte) bolusType }, 
066            new byte[]{ (byte) (correctionBolusIncluded ? 1 : 0) }, 
067            Bytes.firstTwoBytesLittleEndian(carbAmount), 
068            Bytes.firstTwoBytesLittleEndian(bg), 
069            Bytes.toFloat(iob), 
070            Bytes.toUint32(carbRatio));
071    }
072
073    /**
074     * @return the ID of the bolus operation
075     */
076    public int getBolusId() {
077        return bolusId;
078    }
079
080    public int getBolusTypeId() {
081        return bolusTypeId;
082    }
083
084    /**
085     * @return Bolus types
086     */
087    public Set<BolusDeliveryHistoryLog.BolusType> getBolusType() {
088        return BolusDeliveryHistoryLog.BolusType.fromBitmask(bolusTypeId);
089    }
090
091    /**
092     * @return if the correction bolus is included
093     */
094    public boolean getCorrectionBolusIncluded() {
095        return correctionBolusIncluded;
096    }
097
098    /**
099     * @return carbs in grams
100     */
101    public int getCarbAmount() {
102        return carbAmount;
103    }
104
105    /**
106     * @return BG in mg/dL
107     */
108    public int getBg() {
109        return bg;
110    }
111
112    /**
113     * @return current insulin on board
114     */
115    public float getIob() {
116        return iob;
117    }
118
119    /**
120     * @return carb ratio from the insulin delivery profile
121     */
122    public long getCarbRatio() {
123        return carbRatio;
124    }
125}
