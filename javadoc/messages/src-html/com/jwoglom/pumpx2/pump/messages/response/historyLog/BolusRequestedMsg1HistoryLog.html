104        return Bytes.combine(
105            new byte[]{64, 0},
106            Bytes.toUint32(pumpTimeSec),
107            Bytes.toUint32(sequenceNum),
108            Bytes.firstTwoBytesLittleEndian(bolusId),

110            new byte[]{ (byte) (correctionBolusIncluded ? 1 : 0) }, 
111            Bytes.firstTwoBytesLittleEndian(carbAmount),
