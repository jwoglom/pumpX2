001package com.jwoglom.pumpx2.pump.messages.calculator;
002
003import java.util.Objects;
004
005public class BolusParameters {
006    Double units;
007    Integer carbsGrams;
008    Integer glucoseMgdl;
009
010    BolusParameters() {
011        // all null
012    }
013
014    public BolusParameters(Double units, Integer carbsGrams, Integer glucoseMgdl) {
015        this.units = units;
016        this.carbsGrams = carbsGrams;
017        this.glucoseMgdl = glucoseMgdl;
018    }
019
020    public BolusParameters withUnits(Double units) {
021        return new BolusParameters(units, carbsGrams, glucoseMgdl);
022    }
023
024    public BolusParameters withCarbsGrams(Integer carbsGrams) {
025        return new BolusParameters(units, carbsGrams, glucoseMgdl);
026    }
027
028    public BolusParameters withGlucoseMgdl(Integer glucoseMgdl) {
029        return new BolusParameters(units, carbsGrams, glucoseMgdl);
030    }
031
032    public Double getUnits() {
033        return units;
034    }
035
036    public Integer getCarbsGrams() {
037        return carbsGrams;
038    }
039
040    public Integer getGlucoseMgdl() {
041        return glucoseMgdl;
042    }
043
044    public String toString() {
045        return "BolusParameters(units="+units+", carbsGrams="+carbsGrams+", glucoseMgdl="+glucoseMgdl+")";
046    }
047
048    @Override
049    public boolean equals(Object o) {
050        if (this == o) return true;
051        if (o == null || getClass() != o.getClass()) return false;
052        BolusParameters that = (BolusParameters) o;
053        if ((that.units == null && units != null) || (that.units != null && units == null)) {
054            return false;
055        }
056        if ((that.carbsGrams == null && carbsGrams != null) || (that.carbsGrams != null && carbsGrams == null)) {
057            return false;
058        }
059        if ((that.glucoseMgdl == null && glucoseMgdl != null) || (that.glucoseMgdl != null && glucoseMgdl == null)) {
060            return false;
061        }
062        return Double.compare(that.units, units) == 0 && carbsGrams == that.carbsGrams && glucoseMgdl == that.glucoseMgdl;
063    }
064
065    @Override
066    public int hashCode() {
067        return Objects.hash(units, carbsGrams, glucoseMgdl);
068    }
069}
