001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.request.currentStatus.LocalizationRequest;
010
011import java.math.BigInteger;
012
013@MessageProps(
014    opCode=-89,
015    size=7,
016    type=MessageType.RESPONSE,
017    characteristic=Characteristic.CURRENT_STATUS,
018    request=LocalizationRequest.class
019)
020public class LocalizationResponse extends Message {
021    
022    private int glucoseOUM;
023    private int regionSetting;
024    private int languageSelected;
025    private long languagesAvailableBitmask;
026    
027    public LocalizationResponse() {}
028    
029    public LocalizationResponse(int glucoseOUM, int regionSetting, int languageSelected, long languagesAvailableBitmask) {
030        this.cargo = buildCargo(glucoseOUM, regionSetting, languageSelected, languagesAvailableBitmask);
031        // note: these may be out of order
032        this.glucoseOUM = glucoseOUM;
033        this.regionSetting = regionSetting;
034        this.languageSelected = languageSelected;
035        this.languagesAvailableBitmask = languagesAvailableBitmask;
036        
037    }
038
039    public void parse(byte[] raw) {
040        Preconditions.checkArgument(raw.length == props().size());
041        this.cargo = raw;
042        // note: these may be out of order
043        this.glucoseOUM = raw[0];
044        this.regionSetting = raw[1];
045        this.languageSelected = raw[2];
046        this.languagesAvailableBitmask = Bytes.readUint32(raw, 3);
047        
048    }
049
050    
051    public static byte[] buildCargo(int glucoseOUM, int regionSetting, int languageSelected, long languagesAvailableBitmask) {
052        return Bytes.combine(
053            // note: these may be out of order
054            new byte[]{ (byte) glucoseOUM }, 
055            new byte[]{ (byte) regionSetting }, 
056            new byte[]{ (byte) languageSelected }, 
057            Bytes.toUint32(languagesAvailableBitmask));
058    }
059    
060    public int getGlucoseOUM() {
061        return glucoseOUM;
062    }
063    public int getRegionSetting() {
064        return regionSetting;
065    }
066    public int getLanguageSelected() {
067        return languageSelected;
068    }
069    public long getLanguagesAvailableBitmask() {
070        return languagesAvailableBitmask;
071    }
072    
073}
