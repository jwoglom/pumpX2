001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007import java.math.BigInteger;
008
009@HistoryLogProps(
010    opCode = 160,
011    displayName = "CGM Calibration",
012    usedByTidepool = true
013)
014public class CgmCalibrationHistoryLog extends HistoryLog {
015    
016    private long currentTime;
017    private long timestamp;
018    private long calTimestamp;
019    private int value;
020    private int currentDisplayValue;
021    
022    public CgmCalibrationHistoryLog() {}
023    public CgmCalibrationHistoryLog(long pumpTimeSec, long sequenceNum, long currentTime, long timestamp, long calTimestamp, int value, int currentDisplayValue) {
024        super(pumpTimeSec, sequenceNum);
025        this.cargo = buildCargo(pumpTimeSec, sequenceNum, currentTime, timestamp, calTimestamp, value, currentDisplayValue);
026        this.currentTime = currentTime;
027        this.timestamp = timestamp;
028        this.calTimestamp = calTimestamp;
029        this.value = value;
030        this.currentDisplayValue = currentDisplayValue;
031        
032    }
033
034    public CgmCalibrationHistoryLog(long currentTime, long timestamp, long calTimestamp, int value, int currentDisplayValue) {
035        this(0, 0, currentTime, timestamp, calTimestamp, value, currentDisplayValue);
036    }
037
038    public int typeId() {
039        return 160;
040    }
041
042    public void parse(byte[] raw) {
043        Preconditions.checkArgument(raw.length == 26);
044        this.cargo = raw;
045        parseBase(raw);
046        this.currentTime = Bytes.readUint32(raw, 10);
047        this.timestamp = Bytes.readUint32(raw, 14);
048        this.calTimestamp = Bytes.readUint32(raw, 18);
049        this.value = Bytes.readShort(raw, 22);
050        this.currentDisplayValue = Bytes.readShort(raw, 24);
051        
052    }
053
054    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, long currentTime, long timestamp, long calTimestamp, int value, int currentDisplayValue) {
055        return HistoryLog.fillCargo(Bytes.combine(
056            new byte[]{-96, 0}, // (byte) 160
057            Bytes.toUint32(pumpTimeSec),
058            Bytes.toUint32(sequenceNum),
059            Bytes.toUint32(currentTime), 
060            Bytes.toUint32(timestamp), 
061            Bytes.toUint32(calTimestamp), 
062            Bytes.firstTwoBytesLittleEndian(value), 
063            Bytes.firstTwoBytesLittleEndian(currentDisplayValue)));
064    }
065    public long getCurrentTime() {
066        return currentTime;
067    }
068    public long getTimestamp() {
069        return timestamp;
070    }
071    public long getCalTimestamp() {
072        return calTimestamp;
073    }
074    public int getValue() {
075        return value;
076    }
077    public int getCurrentDisplayValue() {
078        return currentDisplayValue;
079    }
080    
081}
