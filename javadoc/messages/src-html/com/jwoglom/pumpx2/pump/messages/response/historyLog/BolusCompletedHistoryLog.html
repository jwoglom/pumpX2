001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.response.currentStatus.LastBolusStatusAbstractResponse;
007
008@HistoryLogProps(
009    opCode = 20,
010    displayName = "Bolus Completed",
011    usedByAndroid = true,
012    usedByTidepool = true
013)
014public class BolusCompletedHistoryLog extends HistoryLog {
015    
016    private int completionStatusId;
017    private int bolusId;
018    private float iob;
019    private float insulinDelivered;
020    private float insulinRequested;
021    
022    public BolusCompletedHistoryLog() {}
023    
024    public BolusCompletedHistoryLog(long pumpTimeSec, long sequenceNum, int completionStatusId, int bolusId, float iob, float insulinDelivered, float insulinRequested) {
025        this.cargo = buildCargo(pumpTimeSec, sequenceNum, completionStatusId, bolusId, iob, insulinDelivered, insulinRequested);
026        this.pumpTimeSec = pumpTimeSec;
027        this.sequenceNum = sequenceNum;
028        this.completionStatusId = completionStatusId;
029        this.bolusId = bolusId;
030        this.iob = iob;
031        this.insulinDelivered = insulinDelivered;
032        this.insulinRequested = insulinRequested;
033        
034    }
035
036    public int typeId() {
037        return 20;
038    }
039
040    /**
041     * <table border="1">
042     * <caption>Parsed fields for BolusCompletedHistoryLog:</caption>
043     * <tr><td><code>completionStatusId</code></td><td><code>Bytes.readShort(raw, 10)</code></td></tr>
044     * <tr><td><code>bolusId</code></td><td><code>Bytes.readShort(raw, 12)</code></td></tr>
045     * <tr><td><code>iob</code></td><td><code>Bytes.readFloat(raw, 14)</code></td></tr>
046     * <tr><td><code>insulinDelivered</code></td><td><code>Bytes.readFloat(raw, 18)</code></td></tr>
047     * <tr><td><code>insulinRequested</code></td><td><code>Bytes.readFloat(raw, 22)</code></td></tr>
048     * </table>
049     */
050    public void parse(byte[] raw) {
051        Preconditions.checkArgument(raw.length == 26);
052        this.cargo = raw;
053        parseBase(raw);
054        this.completionStatusId = Bytes.readShort(raw, 10);
055        this.bolusId = Bytes.readShort(raw, 12);
056        this.iob = Bytes.readFloat(raw, 14);
057        this.insulinDelivered = Bytes.readFloat(raw, 18);
058        this.insulinRequested = Bytes.readFloat(raw, 22);
059        
060    }
061
062    
063    /**
064     * <table border="1">
065     * <caption>Byte layout for BolusCompletedHistoryLog.buildCargo(...):</caption>
066     * <tr><th>Offset</th><th>Description</th></tr>
067     * <tr><td>0</td><td><code>new byte[]{20} [element 0]</code></td></tr>
068     * <tr><td>1</td><td><code>new byte[]{0} [element 1]</code></td></tr>
069     * <tr><td>2</td><td><code>toUint32(pumpTimeSec) [byte 0]</code></td></tr>
070     * <tr><td>3</td><td><code>toUint32(pumpTimeSec) [byte 1]</code></td></tr>
071     * <tr><td>4</td><td><code>toUint32(pumpTimeSec) [byte 2]</code></td></tr>
072     * <tr><td>5</td><td><code>toUint32(pumpTimeSec) [byte 3]</code></td></tr>

074     * <tr><td>7</td><td><code>toUint32(sequenceNum) [byte 1]</code></td></tr>
075     * <tr><td>8</td><td><code>toUint32(sequenceNum) [byte 2]</code></td></tr>

079     * <tr><td>12</td><td><code>firstTwoBytesLittleEndian(bolusId) [byte 0]</code></td></tr>
080     * <tr><td>13</td><td><code>firstTwoBytesLittleEndian(bolusId) [byte 1]</code></td></tr>

085     * <tr><td>18</td><td><code>toFloat(insulinDelivered) [byte 0]</code></td></tr>

092     * <tr><td>25</td><td><code>toFloat(insulinRequested) [byte 3]</code></td></tr>
093     * </table>
094     */
095    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int completionStatus, int bolusId, float iob, float insulinDelivered, float insulinRequested) {
096        return Bytes.combine(
097            new byte[]{20, 0},
098            Bytes.toUint32(pumpTimeSec),
099            Bytes.toUint32(sequenceNum),
100            Bytes.firstTwoBytesLittleEndian(completionStatus), 
101            Bytes.firstTwoBytesLittleEndian(bolusId), 
102            Bytes.toFloat(iob), 
103            Bytes.toFloat(insulinDelivered), 
104            Bytes.toFloat(insulinRequested));
105    }
106    
107    public int getCompletionStatusId() {
108        return completionStatusId;
109    }
110
111    /**
112     * @return whether the bolus was completed fully or stopped
113     */
114    public LastBolusStatusAbstractResponse.BolusStatus getCompletionStatus() {
115        return LastBolusStatusAbstractResponse.BolusStatus.fromId(completionStatusId);
116    }
117
118    /**
119     * @return the ID of the bolus
120     */
121    public int getBolusId() {
122        return bolusId;
123    }
124
125    /**
126     * @return the current insulin on board
127     */
128    public float getIob() {
129        return iob;
130    }
131
132    /**
133     * @return the amount of insulin delivered, in real units. Note that due to ieee float
134     * precision, even if the bolus was completed fully this may differ from the units requested
135     */
136    public float getInsulinDelivered() {
137        return insulinDelivered;
138    }
139
140    /**
141     * @return the amount of insulin requested, in real units
142     */
143    public float getInsulinRequested() {
144        return insulinRequested;
145    }
