022/**
023 * With an ErrorCode of INVALID_PARAMETER, requestCodeId contains the request opcode which failed
024 *
025 * There are really two different ErrorResponse's -- one with a size of 2, (for currentStatus)
026 * and one with a size of 26 (for control / signed messages, called ErrorResponseControl by tandem)
027 *
028
029 *
030 * <br>
031 *
032 * Response message for {@link com.jwoglom.pumpx2.pump.messages.request.NonexistentErrorRequest}, opCode 77, size 2
033 *

045    opCode=77,
046    size=2, // or 26
047    type=MessageType.RESPONSE,
048    request= NonexistentErrorRequest.class

050public class ErrorResponse extends Message {
051    private int requestCodeId;
052    private int errorCodeId;
053    private byte[] remainingBytes;
054    private ErrorCode errorCode;
055
056    public ErrorResponse() {}
057
058    public ErrorResponse(int requestCodeId, int errorCodeId) {
059        parse(buildCargo(requestCodeId, errorCodeId));
060    }
061
062    public void parse(byte[] raw) {
063        // disabled for ErrorResponse since it can be of size 2 or 26
064        //Preconditions.checkArgument(raw.length == props().size());
065        requestCodeId = raw[0];
066        errorCodeId = raw[1];
067        errorCode = ErrorCode.fromByte(errorCodeId);
068        if (raw.length > 2) {
069            remainingBytes = Bytes.dropFirstN(raw, 2);
070        }
071        cargo = raw;
072    }
073
074    /**
075     * <table border="1">
076     * <caption>Byte layout for ErrorResponse.buildCargo(...):</caption>
077     * <tr><th>Offset</th><th>Description</th></tr>
078     * <tr><td>0</td><td><code>new byte[]{...} [element 0]</code></td></tr>
079     * <tr><td>1</td><td><code>new byte[]{...} [element 0]</code></td></tr>
080     * </table>
081     */
082    public static byte[] buildCargo(int unknownByte0, int errorCodeId) {
083        return Bytes.combine(
084                new byte[]{ (byte) unknownByte0 },
085                new byte[]{ (byte) errorCodeId });
086    }
087
088    public int getRequestCodeId() {
089        return requestCodeId;
090    }
091
092    public ErrorCode getErrorCode() {
093        return errorCode;
094    }
095
096    public byte[] getRemainingBytes() {
097        return remainingBytes;
098    }
099
100    public enum ErrorCode {
101        UNDEFINED_ERROR(0),
102        CRC_MISMATCH(1),
103        TRANSACTION_ID_MISMATCH(3),
104        BAD_CARGO_LENGTH(4),
105        BAD_OPCODE(6),
106        INVALID_REQUIRED_PARAMETER(7),
107        MESSAGE_BUFFER_FULL(8),
108        INVALID_AUTHENTICATION_ERROR(9)
109        ;
110
111        private static final Map<Integer, ErrorCode> idMap = new HashMap<>();
112        static {
113            for (ErrorCode item : values()) {
114                idMap.put(item.id(), item);
115            }
116        }
117
118        ErrorCode(int i) {
119            this.id = i;
120        }
121        private final int id;
122
123        public int id() {
124            return id;
125        }
126
127        public static ErrorCode fromByte(int id) {
128            return idMap.get(id);
129        }
130    }
