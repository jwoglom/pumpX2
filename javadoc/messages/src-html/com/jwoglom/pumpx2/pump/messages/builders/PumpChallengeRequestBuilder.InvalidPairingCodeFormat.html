010import java.nio.charset.Charset;
011
012public class PumpChallengeRequestBuilder {
013    public static PumpChallengeRequest create(AbstractCentralChallengeResponse challengeResponse, String pairingCode) throws InvalidPairingCodeFormat {
014        if (challengeResponse instanceof CentralChallengeResponse) {
015            return createV1((CentralChallengeResponse) challengeResponse, pairingCode);
016        } else if (challengeResponse instanceof Jpake1aResponse) {
017            return createV2((Jpake1aResponse) challengeResponse, pairingCode);
018        } else {
019            throw new RuntimeException("invalid CentralChallengeResponse");
020        }
021    }
022
023    public static String processPairingCode(String pairingCode, PairingCodeType type) throws InvalidPairingCodeFormat {
024        if (type == PairingCodeType.LONG_16CHAR) {
025            String processed = type.filterCharacters(pairingCode);
026            if (processed.length() != 16) {
027                throw new InvalidLongPairingCodeFormat();
028            }
029            return processed;
030        } else if (type == PairingCodeType.SHORT_6CHAR) {
031            String processed = type.filterCharacters(pairingCode);
032            if (processed.length() != 6) {
033                throw new InvalidShortPairingCodeFormat();
034            }
035            return processed;
036        } else {
037            throw new InvalidPairingCodeFormat("invalid PairingCodeType");
038        }
039    }
040
041    public static String processPairingCode(String pairingCode) throws InvalidPairingCodeFormat {
042        if (pairingCode.length() == 6 || PairingCodeType.SHORT_6CHAR.filterCharacters(pairingCode).length() == 6) {
043            return processPairingCode(pairingCode, PairingCodeType.SHORT_6CHAR);
044        }
045        return processPairingCode(pairingCode, PairingCodeType.LONG_16CHAR);
046    }
047
048    // HMAC sha1
049    public static PumpChallengeRequest createV1(CentralChallengeResponse challengeResponse, String pairingCode) throws InvalidPairingCodeFormat {
050        int appInstanceId = challengeResponse.getAppInstanceId();
051        byte[] hmacKey = challengeResponse.getHmacKey();
052
053        String pairingChars = processPairingCode(pairingCode, PairingCodeType.LONG_16CHAR);
054        byte[] challengeHash = Packetize.doHmacSha1(hmacKey, pairingChars.getBytes(Charset.forName("UTF-8")));
055        return new PumpChallengeRequest(
056                appInstanceId,
057                challengeHash);
058    }
059

064        return null;
065    }
066
067    public static class InvalidPairingCodeFormat extends Exception {
068        InvalidPairingCodeFormat(String reason) {
069            super("The pairing code entered does not match the expected format. " + reason);
070        }
071    }
072
073    public static class InvalidLongPairingCodeFormat extends InvalidPairingCodeFormat {
074        InvalidLongPairingCodeFormat() {
075            super("It should be 16 alphanumeric characters total across 5 groups of 4 characters each.");
076        }
077    }
078    public static class InvalidShortPairingCodeFormat extends InvalidPairingCodeFormat {
079        InvalidShortPairingCodeFormat() {
080            super("It should be 6 numbers.");
081        }
082    }
083}
