001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.CurrentEGVGuiDataRequest;
009
010/**
011 * Gets the current CGM reading
012 */
013@MessageProps(
014    opCode=35,
015    size=8,
016    type=MessageType.RESPONSE,
017    request=CurrentEGVGuiDataRequest.class
018)
019public class CurrentEGVGuiDataResponse extends Message {
020    
021    private long bgReadingTimestampSeconds;
022    private int cgmReading;
023    private int egvStatus;
024    private int trendRate;
025    
026    public CurrentEGVGuiDataResponse() {}
027    
028    public CurrentEGVGuiDataResponse(long bgReadingTimestampSeconds, int cgmReading, int egvStatus, int trendRate) {
029        this.cargo = buildCargo(bgReadingTimestampSeconds, cgmReading, egvStatus, trendRate);
030        this.bgReadingTimestampSeconds = bgReadingTimestampSeconds;
031        this.cgmReading = cgmReading;
032        this.egvStatus = egvStatus;
033        this.trendRate = trendRate;
034        
035    }
036
037    public void parse(byte[] raw) {
038        Preconditions.checkArgument(raw.length == props().size());
039        this.cargo = raw;
040        this.bgReadingTimestampSeconds = Bytes.readUint32(raw, 0);
041        this.cgmReading = Bytes.readShort(raw, 4);
042        this.egvStatus = raw[6];
043        this.trendRate = raw[7];
044        
045    }
046
047    
048    public static byte[] buildCargo(long bgReadingTimestampSeconds, int cgmReading, int egvStatus, int trendRate) {
049        return Bytes.combine(
050            Bytes.toUint32(bgReadingTimestampSeconds), 
051            Bytes.firstTwoBytesLittleEndian(cgmReading), 
052            new byte[]{ (byte) egvStatus }, 
053            new byte[]{ (byte) trendRate });
054    }
055    
056    public long getBgReadingTimestampSeconds() {
057        return bgReadingTimestampSeconds;
058    }
059    public int getCgmReading() {
060        return cgmReading;
061    }
062    public EGVStatus getEgvStatus() {
063        return EGVStatus.fromId(egvStatus);
064    }
065    public int getTrendRate() {
066        return trendRate;
067    }
068
069    public enum EGVStatus {
070        INVALID(0),
071        VALID(1),
072        LOW(2),
073        HIGH(3),
074        UNAVAILABLE(4),
075        ;
076
077        private final int id;
078        EGVStatus(int id) {
079            this.id = id;
080        }
081
082        public int id() {
083            return id;
084        }
085
086        public static EGVStatus fromId(int id) {
087            for (EGVStatus e : values()) {
088                if (e.id() == id) {
089                    return e;
090                }
091            }
092            return null;
093        }
094    }
095    
096}
