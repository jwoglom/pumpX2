001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007/**
008 * A message which appears in the history log every hour at minimum, more frequently when the basal
009 * rate changes, and contains a summary of the total basal deliveries up to that point in the current
010 * day. Resets at 00:00 midnight when there is a {@link NewDayHistoryLog}.
011 */
012@HistoryLogProps(
013    opCode = 81,
014    displayName = "Daily Basal",
015    usedByTidepool = true
016)
017public class DailyBasalHistoryLog extends HistoryLog {
018    
019    private float dailyTotalBasal;
020    private float lastBasalRate;
021    private float iob;
022    private boolean finalEventForDay;
023    private int batteryChargeRaw;
024    private int lipoMv;
025    
026    public DailyBasalHistoryLog() {}
027    public DailyBasalHistoryLog(long pumpTimeSec, long sequenceNum, float dailyTotalBasal, float lastBasalRate, float iob, boolean finalEventForDay, int actualBatteryCharge, int lipoMv) {
028        super(pumpTimeSec, sequenceNum);
029        this.cargo = buildCargo(pumpTimeSec, sequenceNum, dailyTotalBasal, lastBasalRate, iob, finalEventForDay, actualBatteryCharge, lipoMv);
030        this.dailyTotalBasal = dailyTotalBasal;
031        this.lastBasalRate = lastBasalRate;
032        this.iob = iob;
033        this.finalEventForDay = finalEventForDay;
034        this.batteryChargeRaw = actualBatteryCharge;
035        this.lipoMv = lipoMv;
036        
037    }
038
039    public DailyBasalHistoryLog(float dailyTotalBasal, float lastBasalRate, float iob, boolean finalEventForDay, int actualBatteryCharge, int lipoMv) {
040        this(0, 0, dailyTotalBasal, lastBasalRate, iob, finalEventForDay, actualBatteryCharge, lipoMv);
041    }
042
043    public int typeId() {
044        return 81;
045    }
046
047    /**
048     * <table border="1">
049     * <caption>Parsed fields for DailyBasalHistoryLog:</caption>
050     * <tr><td><code>dailyTotalBasal</code></td><td><code>Bytes.readFloat(raw, 10)</code></td></tr>
051     * <tr><td><code>lastBasalRate</code></td><td><code>Bytes.readFloat(raw, 14)</code></td></tr>
052     * <tr><td><code>iob</code></td><td><code>Bytes.readFloat(raw, 18)</code></td></tr>
053     * <tr><td><code>finalEventForDay</code></td><td><code>raw[22] == 1</code></td></tr>
054     * <tr><td><code>batteryChargeRaw</code></td><td><code>raw[23]</code></td></tr>
055     * <tr><td><code>lipoMv</code></td><td><code>Bytes.readShort(raw, 24)</code></td></tr>
056     * </table>
057     */
058    public void parse(byte[] raw) {
059        Preconditions.checkArgument(raw.length == 26);
060        this.cargo = raw;
061        parseBase(raw);
062        this.dailyTotalBasal = Bytes.readFloat(raw, 10);
063        this.lastBasalRate = Bytes.readFloat(raw, 14);
064        this.iob = Bytes.readFloat(raw, 18);
065        this.finalEventForDay = raw[22] == 1;
066        this.batteryChargeRaw = raw[23];
067        this.lipoMv = Bytes.readShort(raw, 24);
068        
069    }
070
071    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, float dailyTotalBasal, float lastBasalRate, float iob, boolean finalEventForDay, int actualBatteryCharge, int lipoMv) {
072        return HistoryLog.fillCargo(Bytes.combine(
073            new byte[]{81, 0},
074            Bytes.toUint32(pumpTimeSec),
075            Bytes.toUint32(sequenceNum),
076            Bytes.toFloat(dailyTotalBasal), 
077            Bytes.toFloat(lastBasalRate), 
078            Bytes.toFloat(iob),
079            new byte[]{ (byte) (finalEventForDay ? 1 : 0) },
080            new byte[]{ (byte) actualBatteryCharge },
081            Bytes.firstTwoBytesLittleEndian(lipoMv)));
082    }
083
084    /**
085     * @return the total basal amount delivered in the gap between this message and the previous
086     */
087    public float getDailyTotalBasal() {
088        return dailyTotalBasal;
089    }
090
091    /**
092     * @return the last basal rate used as of this message
093     */
094    public float getLastBasalRate() {
095        return lastBasalRate;
096    }
097
098    /**
099     * @return the current IOB
100     */
101    public float getIob() {
102        return iob;
103    }
104
105    /**
106     * @return TODO(confirm): whether this is the final event of the day
107     */
108    public boolean getFinalEventForDay() {
109        return finalEventForDay;
110    }
111
112
113    public int getBatteryChargeRaw() {
114        return batteryChargeRaw;
115    }
116
117    /**
118     * @return the reported battery charge in percent
119     */
120    public double getBatteryChargePercent() {
121       return (100 * ((batteryChargeRaw+256.0) / 512.0));
122    }
123
124    /**
125     * @return the reported LIPO battery amount in millivolts
126     */
127    public int getLipoMv() {
128        return lipoMv;
129    }
130}
