001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007/**
008 * A message which appears in the history log every hour at minimum, more frequently when the basal
009 * rate changes, and contains a summary of the total basal deliveries up to that point in the current
010 * day. Resets at 00:00 midnight when there is a {@link NewDayHistoryLog}.
011 */
012@HistoryLogProps(
013    opCode = 81,
014    displayName = "Daily Basal",
015    usedByTidepool = true
016)
017public class DailyBasalHistoryLog extends HistoryLog {
018    
019    private float dailyTotalBasal;
020    private float lastBasalRate;
021    private float iob;
022    private boolean finalEventForDay;
023    private int batteryChargeRaw;
024    private int lipoMv;
025    
026    public DailyBasalHistoryLog() {}
027    public DailyBasalHistoryLog(long pumpTimeSec, long sequenceNum, float dailyTotalBasal, float lastBasalRate, float iob, boolean finalEventForDay, int actualBatteryCharge, int lipoMv) {
028        super(pumpTimeSec, sequenceNum);
029        this.cargo = buildCargo(pumpTimeSec, sequenceNum, dailyTotalBasal, lastBasalRate, iob, finalEventForDay, actualBatteryCharge, lipoMv);
030        this.dailyTotalBasal = dailyTotalBasal;
031        this.lastBasalRate = lastBasalRate;
032        this.iob = iob;
033        this.finalEventForDay = finalEventForDay;
034        this.batteryChargeRaw = actualBatteryCharge;
035        this.lipoMv = lipoMv;
036        
037    }
038
039    public DailyBasalHistoryLog(float dailyTotalBasal, float lastBasalRate, float iob, boolean finalEventForDay, int actualBatteryCharge, int lipoMv) {
040        this(0, 0, dailyTotalBasal, lastBasalRate, iob, finalEventForDay, actualBatteryCharge, lipoMv);
041    }
042
043    public int typeId() {
044        return 81;
045    }
046
047    public void parse(byte[] raw) {
048        Preconditions.checkArgument(raw.length == 26);
049        this.cargo = raw;
050        parseBase(raw);
051        this.dailyTotalBasal = Bytes.readFloat(raw, 10);
052        this.lastBasalRate = Bytes.readFloat(raw, 14);
053        this.iob = Bytes.readFloat(raw, 18);
054        this.finalEventForDay = raw[22] == 1;
055        this.batteryChargeRaw = raw[23];
056        this.lipoMv = Bytes.readShort(raw, 24);
057        
058    }
059
060    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, float dailyTotalBasal, float lastBasalRate, float iob, boolean finalEventForDay, int actualBatteryCharge, int lipoMv) {
061        return HistoryLog.fillCargo(Bytes.combine(
062            new byte[]{81, 0},
063            Bytes.toUint32(pumpTimeSec),
064            Bytes.toUint32(sequenceNum),
065            Bytes.toFloat(dailyTotalBasal), 
066            Bytes.toFloat(lastBasalRate), 
067            Bytes.toFloat(iob),
068            new byte[]{ (byte) (finalEventForDay ? 1 : 0) },
069            new byte[]{ (byte) actualBatteryCharge },
070            Bytes.firstTwoBytesLittleEndian(lipoMv)));
071    }
072
073    /**
074     * @return the total basal amount delivered in the gap between this message and the previous
075     */
076    public float getDailyTotalBasal() {
077        return dailyTotalBasal;
078    }
079
080    /**
081     * @return the last basal rate used as of this message
082     */
083    public float getLastBasalRate() {
084        return lastBasalRate;
085    }
086
087    /**
088     * @return the current IOB
089     */
090    public float getIob() {
091        return iob;
092    }
093
094    /**
095     * @return TODO(confirm): whether this is the final event of the day
096     */
097    public boolean getFinalEventForDay() {
098        return finalEventForDay;
099    }
100
101
102    public int getBatteryChargeRaw() {
103        return batteryChargeRaw;
104    }
105
106    /**
107     * @return the reported battery charge in percent
108     */
109    public double getBatteryChargePercent() {
110       return (100 * ((batteryChargeRaw+256.0) / 512.0));
111    }
112
113    /**
114     * @return the reported LIPO battery amount in millivolts
115     */
116    public int getLipoMv() {
117        return lipoMv;
118    }
119}
