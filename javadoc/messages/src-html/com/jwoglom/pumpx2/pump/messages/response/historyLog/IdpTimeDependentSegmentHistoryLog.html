064    public void parse(byte[] raw) {
065        Preconditions.checkArgument(raw.length == 26);
066        this.cargo = raw;
067        parseBase(raw);
068        this.idp = raw[10];
069        this.status = raw[11];
070        this.segmentIndex = raw[12];
071        this.modificationType = raw[13];
072        this.startTime = Bytes.readShort(raw, 14);
073        this.basalRate = Bytes.readShort(raw, 16);

080    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int idp, int status, int segmentIndex, int modificationType, int startTime, int basalRate, int isf, long targetBg, int carbRatio) {
081        return HistoryLog.fillCargo(Bytes.combine(
082            new byte[]{68, 0},
083            Bytes.toUint32(pumpTimeSec),
084            Bytes.toUint32(sequenceNum),
085            new byte[]{ (byte) idp }, 
086            new byte[]{ (byte) status }, 
087            new byte[]{ (byte) segmentIndex }, 
088            new byte[]{ (byte) modificationType }, 
089            Bytes.firstTwoBytesLittleEndian(startTime), 
090            Bytes.firstTwoBytesLittleEndian(basalRate), 
091            Bytes.firstTwoBytesLittleEndian(isf), 
092            Bytes.toUint32(targetBg), 
093            new byte[]{ (byte) carbRatio }));
094    }
