001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.response.currentStatus.AlarmStatusResponse;
007
008import java.math.BigInteger;
009
010@HistoryLogProps(
011    opCode = 5,
012    displayName = "Alarm Activated",
013    usedByTidepool = true
014)
015public class AlarmActivatedHistoryLog extends HistoryLog {
016    
017    private long alarmId;
018    
019    public AlarmActivatedHistoryLog() {}
020    public AlarmActivatedHistoryLog(long pumpTimeSec, long sequenceNum, long alarmId) {
021        super(pumpTimeSec, sequenceNum);
022        this.cargo = buildCargo(pumpTimeSec, sequenceNum, alarmId);
023        this.alarmId = alarmId;
024        
025    }
026
027    public AlarmActivatedHistoryLog(long alarmId) {
028        this(0, 0, alarmId);
029    }
030
031    public int typeId() {
032        return 5;
033    }
034
035    public void parse(byte[] raw) {
036        Preconditions.checkArgument(raw.length == 26);
037        this.cargo = raw;
038        parseBase(raw);
039        this.alarmId = Bytes.readUint32(raw, 10);
040        
041    }
042
043    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, long alarmId) {
044        return HistoryLog.fillCargo(Bytes.combine(
045            new byte[]{5, 0},
046            Bytes.toUint32(pumpTimeSec),
047            Bytes.toUint32(sequenceNum),
048            Bytes.toUint32(alarmId)));
049    }
050    public long getAlarmId() {
051        return alarmId;
052    }
053
054    /**
055     * @return the type of alarm
056     */
057    public AlarmStatusResponse.AlarmResponseType getAlarmResponseType() {
058        return AlarmStatusResponse.AlarmResponseType.fromSingularId(alarmId);
059    }
060}
