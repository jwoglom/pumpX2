001package com.jwoglom.pumpx2.pump.messages.util;
002
003import com.jwoglom.pumpx2.pump.messages.Messages;
004import com.jwoglom.pumpx2.shared.JavaHelpers;
005
006import java.util.Arrays;
007import java.util.List;
008import java.util.stream.Collectors;
009
010public class MessageHelpers {
011    public static final String REQUEST_PACKAGE = "com.jwoglom.pumpx2.pump.messages.request";
012    public static List<String> getAllPumpRequestMessages() {
013        return Arrays.stream(Messages.values())
014                .map(messages -> messages.requestClass().getName())
015                .filter(name -> name.startsWith(REQUEST_PACKAGE))
016                .filter(name -> !name.endsWith("Test"))
017                .map(MessageHelpers::lastTwoParts)
018                .sorted()
019                .collect(Collectors.toList());
020    }
021
022    public static String lastTwoParts(String name) {
023        String[] parts = name.split("\\.");
024        return parts[parts.length-2] + "." + parts[parts.length-1];
025    }
026}
