021/**
022 * Helper class which parses arbitrary messages (with or without the known opcode and transaction ID)
023 */
024public class ArbitraryMessageParser {
025
026    public static void setPumpTimeSinceReset(String pumpTimeSinceReset) {
027        if (!Strings.isNullOrEmpty(pumpTimeSinceReset)){
028            PumpStateSupplier.pumpTimeSinceReset = () -> Long.valueOf(pumpTimeSinceReset);
029        }
030    }
031
032    @SuppressWarnings("deprecation")
033    public static Optional<Message> parse(String rawHex) throws DecoderException {
034        byte[] initialRead = Hex.decodeHex(rawHex);
035        int txId = initialRead[1];
036        int opCode = initialRead[2];
037        Message message = null;
038        try {
039            Set<Characteristic> possibilities = Messages.findPossibleCharacteristicsForOpcode(opCode);
040            if (possibilities.size() > 1) {
041                System.err.print("Multiple characteristics possible for opCode: "+opCode+": "+possibilities);
042                return Optional.empty();
043            }
044            message = Messages.fromOpcode(opCode, possibilities.stream().iterator().next()).newInstance();
045            message.fillWithEmptyCargo();
046        } catch (NullPointerException e) {
047            System.err.print("Unknown opcode "+opCode);
048            return Optional.empty();
049        } catch (InstantiationException|IllegalAccessException e) {
050            e.printStackTrace();
051        }
052        try {
053            return testRequest(rawHex, txId, message);
054        } catch (IllegalStateException e) {
055            System.err.print("Needs more packets"); // for "+opCode+": "+e.getMessage());
056            return Optional.empty();
057        }
058    }
059
060    public static Optional<Message> testRequest(String rawHex, int txId, Message expected) throws DecoderException {
061        byte[] initialRead = Hex.decodeHex(rawHex);
062
063        UUID uuid = CharacteristicUUID.determine(expected);
064        MessageType messageType = expected.type();
065
066        TronMessageWrapper tron = new TronMessageWrapper(expected, (byte) txId);
067        PumpResponseMessage resp = BTResponseParser.parse(tron, initialRead, messageType, uuid);
068        return resp.message();
069    }
070}
