003public class InsulinUnit {
004    // 1000 = 1 unit
005    public static double from1000To1(Long units) {
006        return (units.doubleValue() / 1000);
007    }
008
009    public static long from1To1000(Float units) {
010        return Float.valueOf(units * 1000).longValue();
011    }
012
013    public static long from1To1000(Double units) {
014        return Double.valueOf(units * 1000).longValue();
015    }
016}
