001package com.jwoglom.pumpx2.pump.messages.response.control;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
010import com.jwoglom.pumpx2.pump.messages.request.control.BolusPermissionReleaseRequest;
011
012/**
013 * Returns whether the bolus for which permission was requested by the pump could have its permission
014 * be successfully released. Doesn't appear to do anything user-facing, but probably closes out the
015 * potential bolus in the history logs.
016 */
017@MessageProps(
018    opCode=-15,
019    size=1,
020    type=MessageType.RESPONSE,
021    characteristic=Characteristic.CONTROL,
022    request= BolusPermissionReleaseRequest.class,
023    signed=true,
024    minApi=KnownApiVersion.API_V2_5
025)
026public class BolusPermissionReleaseResponse extends Message {
027    private int status;
028    
029    public BolusPermissionReleaseResponse() {}
030
031    public BolusPermissionReleaseResponse(int status) {
032        this.cargo = buildCargo(status);
033        this.status = status;
034    }
035
036    public static byte[] buildCargo(int status) {
037        return Bytes.combine(
038                Bytes.firstByteLittleEndian(status)
039        );
040    }
041
042    public void parse(byte[] raw) {
043        raw = removeSignedRequestHmacBytes(raw);
044        Preconditions.checkArgument(raw.length == props().size(), "length: " + raw.length);
045        this.status = raw[0];
046        this.cargo = raw;
047    }
048
049
050    public int getStatus() {
051        return status;
052    }
053
054    public ReleaseStatus getReleaseStatus() {
055        return ReleaseStatus.fromId(status);
056    }
057
058    public enum ReleaseStatus {
059        SUCCESS(0),
060        FAILURE(1),
061
062        ;
063
064        private final int id;
065        ReleaseStatus(int id) {
066            this.id = id;
067        }
068
069        public static ReleaseStatus fromId(int id) {
070            for (ReleaseStatus r : values()) {
071                if (r.id == id) {
072                    return r;
073                }
074            }
075            return null;
076        }
077    }
078}
