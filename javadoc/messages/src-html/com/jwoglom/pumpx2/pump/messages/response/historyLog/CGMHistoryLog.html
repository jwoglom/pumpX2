001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007@HistoryLogProps(
008    opCode = 256,
009    usedByAndroid = true,
010    usedByTidepool = true // LID_CGM_DATA_GXB
011)
012public class CGMHistoryLog extends HistoryLog {
013    
014    private int glucoseValueStatus;
015    private int cgmDataType;
016    private int rate;
017    private int algorithmState;
018    private int rssi;
019    private int currentGlucoseDisplayValue;
020    private long timeStampSeconds;
021    private int egvInfoBitmask;
022    private int interval;
023    
024    public CGMHistoryLog() {}
025    
026    public CGMHistoryLog(long pumpTimeSec, long sequenceNum, int glucoseValueStatus, int cgmDataType, int rate, int algorithmState, int rssi, int currentGlucoseDisplayValue, long timeStampSeconds, int egvInfoBitmask, int interval) {
027        this.cargo = buildCargo(pumpTimeSec, sequenceNum, glucoseValueStatus, cgmDataType, rate, algorithmState, rssi, currentGlucoseDisplayValue, timeStampSeconds, egvInfoBitmask, interval);
028        this.pumpTimeSec = pumpTimeSec;
029        this.sequenceNum = sequenceNum;
030        this.glucoseValueStatus = glucoseValueStatus;
031        this.cgmDataType = cgmDataType;
032        this.rate = rate;
033        this.algorithmState = algorithmState;
034        this.rssi = rssi;
035        this.currentGlucoseDisplayValue = currentGlucoseDisplayValue;
036        this.timeStampSeconds = timeStampSeconds;
037        this.egvInfoBitmask = egvInfoBitmask;
038        this.interval = interval;
039        
040    }
041
042    public int typeId() {
043        return 256;
044    }
045
046    public void parse(byte[] raw) {
047        Preconditions.checkArgument(raw.length == 26);
048        this.cargo = raw;
049        parseBase(raw);
050        this.glucoseValueStatus = Bytes.readShort(raw, 10);
051        this.cgmDataType = raw[12];
052        this.rate = raw[13];
053        this.algorithmState = raw[14];
054        this.rssi = raw[15];
055        this.currentGlucoseDisplayValue = Bytes.readShort(raw, 16);
056        this.timeStampSeconds = Bytes.readUint32(raw, 18);
057        this.egvInfoBitmask = Bytes.readShort(raw, 22);
058        this.interval = raw[24];
059        
060    }
061
062    
063    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int glucoseValueStatus, int cgmDataType, int rate, int algorithmState, int rssi, int currentGlucoseDisplayValue, long timeStampSeconds, int egvInfoBitmask, int interval) {
064        return Bytes.combine(
065            new byte[]{0, 1},
066            Bytes.toUint32(pumpTimeSec),
067            Bytes.toUint32(sequenceNum),
068            Bytes.firstTwoBytesLittleEndian(glucoseValueStatus), 
069            new byte[]{ (byte) cgmDataType }, 
070            new byte[]{ (byte) rate }, 
071            new byte[]{ (byte) algorithmState }, 
072            new byte[]{ (byte) rssi }, 
073            Bytes.firstTwoBytesLittleEndian(currentGlucoseDisplayValue), 
074            Bytes.toUint32(timeStampSeconds), 
075            Bytes.firstTwoBytesLittleEndian(egvInfoBitmask), 
076            new byte[]{ (byte) interval },
077            new byte[]{ 1 }); // missing param?
078    }
079    
080    public int getGlucoseValueStatus() {
081        return glucoseValueStatus;
082    }
083    public int getCgmDataType() {
084        return cgmDataType;
085    }
086    public int getRate() {
087        return rate;
088    }
089    public int getAlgorithmState() {
090        return algorithmState;
091    }
092    public int getRssi() {
093        return rssi;
094    }
095    public int getCurrentGlucoseDisplayValue() {
096        return currentGlucoseDisplayValue;
097    }
098    public long getTimeStampSeconds() {
099        return timeStampSeconds;
100    }
101    public int getEgvInfoBitmask() {
102        return egvInfoBitmask;
103    }
104    public int getInterval() {
105        return interval;
106    }
107    
108}
