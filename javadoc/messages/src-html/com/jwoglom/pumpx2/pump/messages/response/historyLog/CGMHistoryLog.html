061        Preconditions.checkArgument(raw.length == 26);
062        this.cargo = raw;
063        parseBase(raw);
064        this.glucoseValueStatus = Bytes.readShort(raw, 10);
065        this.cgmDataType = raw[12];
066        this.rate = raw[13];
067        this.algorithmState = raw[14];
068        this.rssi = raw[15];
069        this.currentGlucoseDisplayValue = Bytes.readShort(raw, 16);
070        this.timeStampSeconds = Bytes.readUint32(raw, 18);
071        this.egvInfoBitmask = Bytes.readShort(raw, 22);
072        this.interval = raw[24];

110        return Bytes.combine(
111            new byte[]{0, 1},
112            Bytes.toUint32(pumpTimeSec),
113            Bytes.toUint32(sequenceNum),
114            Bytes.firstTwoBytesLittleEndian(glucoseValueStatus), 
115            new byte[]{ (byte) cgmDataType }, 
116            new byte[]{ (byte) rate }, 
117            new byte[]{ (byte) algorithmState },

123            new byte[]{ 1 }); // missing param?
124    }
