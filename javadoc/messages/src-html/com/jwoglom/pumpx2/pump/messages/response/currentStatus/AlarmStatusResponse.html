112        private final int bitmask;
113        AlarmResponseType(int bitmask) {
114            this.bitmask = bitmask;
115        }
116
117        public int bitmask() {
118            return bitmask;
119        }
120
121        public static BigInteger toBitmask(AlarmResponseType ...types) {
122            BigInteger ret = BigInteger.ZERO;
123            for (AlarmResponseType type : types) {
124                ret = ret.setBit(type.bitmask());
125            }
126
127            return ret;
128        }
129
130        public static Set<AlarmResponseType> fromBitmask(BigInteger intMap) {
131            Set<AlarmResponseType> current = new HashSet<>();
132            for (AlarmResponseType type : AlarmResponseType.values()) {
133                if (intMap.testBit(type.bitmask())) {
134                    current.add(type);
135                }
136            }
137
138            return current;
139        }
140
141        public static AlarmResponseType fromSingularId(long id) {
142            for (AlarmResponseType type : AlarmResponseType.values()) {
143                if (id == type.bitmask()) {
144                    return type;
145                }
146            }
147            return null;
148        }
