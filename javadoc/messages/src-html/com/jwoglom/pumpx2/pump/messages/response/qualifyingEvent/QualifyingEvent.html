035/**
036 * A list of all QualifyingEvents which can be sent from the pump. For some events, a set of request
037 * messages which can be sent (via `sendPumpCommand`) to receive more information about the event
038 * which just occurred, in logic matching the Tandem mobile app, is included. To get these
039 * instantiated request messages, you can invoke:
040 *     QualifyingEvent.groupSuggestedHandlers(qualifyingEvents)
041 *
042 * Note that if t:connect app connection sharing is enabled and the t:connect app is running,
043 * you can use enableSendSharedConnectionResponseMessages and get the response from these callbacks
044 * for free, without duplicating requests.
045 */
046public enum QualifyingEvent {
047    ALERT(1, ImmutableSet.of(
048            AlertStatusRequest::new)),
049    ALARM(2, ImmutableSet.of(
050            AlarmStatusRequest::new)),
051    REMINDER(4, ImmutableSet.of(
052            ReminderStatusRequest::new)),
053    RESERVED(8),
054    CGM_ALERT(16, ImmutableSet.of(
055            CGMAlertStatusRequest::new)),
056    HOME_SCREEN_CHANGE(32, ImmutableSet.of(
057            CurrentBasalStatusRequest::new,
058            CurrentEGVGuiDataRequest::new,
059            HomeScreenMirrorRequest::new,
060            () -> ControlIQInfoRequestBuilder.create(PumpStateSupplier.pumpApiVersion.get()))),
061    PUMP_SUSPEND(64, ImmutableSet.of(
062            InsulinStatusRequest::new,
063            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()))),
064    PUMP_RESUME(128, ImmutableSet.of(
065            InsulinStatusRequest::new,
066            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()),
067            CurrentEGVGuiDataRequest::new,
068            ProfileStatusRequest::new)),
069    TIME_CHANGE(256, ImmutableSet.of(
070            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()),
071            CGMStatusRequest::new,
072            TimeSinceResetRequest::new)),
073    BASAL_CHANGE(512, ImmutableSet.of(
074            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()),
075            HomeScreenMirrorRequest::new,
076            CurrentBasalStatusRequest::new,
077            TempRateRequest::new)),
078    BOLUS_CHANGE(1024, ImmutableSet.of(
079            CurrentBolusStatusRequest::new,
080            ExtendedBolusStatusRequest::new,
081            () -> LastBolusStatusRequestBuilder.create(PumpStateSupplier.pumpApiVersion.get()))),
082    IOB_CHANGE(2048, ImmutableSet.of(
083            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()))),
084    EXTENDED_BOLUS_CHANGE(4096, ImmutableSet.of(
085            ExtendedBolusStatusRequest::new,
086            () -> LastBolusStatusRequestBuilder.create(PumpStateSupplier.pumpApiVersion.get()))),
087    PROFILE_CHANGE(8192, ImmutableSet.of(
088            ProfileStatusRequest::new)),
089    BG(16384, ImmutableSet.of(
090            LastBGRequest::new)),
091    CGM_CHANGE(32768, ImmutableSet.of(
092            CGMStatusRequest::new,
093            CurrentEGVGuiDataRequest::new,
094            HomeScreenMirrorRequest::new)),
095    BATTERY(65536, ImmutableSet.of(
096            () -> CurrentBatteryRequestBuilder.create(PumpStateSupplier.pumpApiVersion.get()))),
097    BASAL_IQ(131072, ImmutableSet.of(
098            BasalIQSettingsRequest::new)),
099    REMAINING_INSULIN(262144, ImmutableSet.of(
100            InsulinStatusRequest::new)),
101    SUSPEND_COMM(524288),
102    ACTIVE_SEGMENT_CHANGE(1048576, ImmutableSet.of(
103            ProfileStatusRequest::new)),
104    BASAL_IQ_STATUS(2097152, ImmutableSet.of(
105            BasalIQSettingsRequest::new,
106            BasalIQStatusRequest::new)),
107    CONTROL_IQ_INFO(4194304, ImmutableSet.of(
108            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()),
109            () -> ControlIQInfoRequestBuilder.create(PumpStateSupplier.pumpApiVersion.get()),
110            ControlIQSleepScheduleRequest::new)),
111    CONTROL_IQ_SLEEP(8388608, ImmutableSet.of(
112            ControlIQSleepScheduleRequest::new)),
113    BOLUS_PERMISSION_REVOKED(2147483648L),
114
115    ;
116    private final long id;
117    private final Set<Supplier<? extends Message>> suggestedHandlers;
118    QualifyingEvent(long id) {
119        this.id = id;
120        this.suggestedHandlers = new HashSet<>();
121    }
122
123    QualifyingEvent(long id, Set<Supplier<? extends Message>> suggestedHandlers) {
124        this.id = id;
125        this.suggestedHandlers = suggestedHandlers;
126    }
127
128    public final long getId() {
129        return id;
130    }
131
132    public final Set<Supplier<? extends Message>> getSuggestedHandlers() {
133        return suggestedHandlers;
134    }
135
136    public static Set<QualifyingEvent> fromBitmask(long bitmask) {
137        Set<QualifyingEvent> ret = new HashSet<>();
138        for (QualifyingEvent e : values()) {
139            if ((bitmask & e.getId()) != 0) {
140                ret.add(e);
141            }
142        }
143        return ret;
144    }
145
146    public static Set<QualifyingEvent> fromRawBtBytes(byte[] raw) {
147        return fromBitmask(Bytes.readUint32(raw, 0));
148    }
149
150    public static Set<? extends Message> groupSuggestedHandlers(Set<QualifyingEvent> events) {
151        Set<Supplier<? extends Message>> messages = new HashSet<>();
152        for (QualifyingEvent e : events) {
153            messages.addAll(e.suggestedHandlers);
154        }
155        return messages.stream().map(Supplier::get).collect(Collectors.toSet());
156    }
157
158    public static int toBitmask(QualifyingEvent...states) {
159        int bitmask = 0;
160        for (QualifyingEvent e : states) {
161            bitmask |= e.getId();
162        }
163        return bitmask;
164    }
