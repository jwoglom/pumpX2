037/**
038 * A list of all QualifyingEvents which can be sent from the pump. For some events, a set of request
039 * messages which can be sent (via `sendPumpCommand`) to receive more information about the event
040 * which just occurred, in logic matching the Tandem mobile app, is included. To get these
041 * instantiated request messages, you can invoke:
042 *     QualifyingEvent.groupSuggestedHandlers(qualifyingEvents)
043 *
044 * Note that if t:connect app connection sharing is enabled and the t:connect app is running,
045 * you can use enableSendSharedConnectionResponseMessages and get the response from these callbacks
046 * for free, without duplicating requests.
047 */
048public enum QualifyingEvent {
049    ALERT(1, ImmutableSet.of(
050            AlertStatusRequest::new)),
051    ALARM(2, ImmutableSet.of(
052            AlarmStatusRequest::new)),
053    REMINDER(4, ImmutableSet.of(
054            ReminderStatusRequest::new)),
055    MALFUNCTION(8),
056    CGM_ALERT(16, ImmutableSet.of(
057            CGMAlertStatusRequest::new)),
058    HOME_SCREEN_CHANGE(32, ImmutableSet.of(
059            CurrentBasalStatusRequest::new,
060            CurrentEGVGuiDataRequest::new,
061            HomeScreenMirrorRequest::new,
062            () -> ControlIQInfoRequestBuilder.create(PumpStateSupplier.pumpApiVersion.get()))),
063    PUMP_SUSPEND(64, ImmutableSet.of(
064            InsulinStatusRequest::new,
065            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()))),
066    PUMP_RESUME(128, ImmutableSet.of(
067            InsulinStatusRequest::new,
068            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()),
069            CurrentEGVGuiDataRequest::new,
070            ProfileStatusRequest::new)),
071    TIME_CHANGE(256, ImmutableSet.of(
072            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()),
073            CGMStatusRequest::new,
074            TimeSinceResetRequest::new)),
075    BASAL_CHANGE(512, ImmutableSet.of(
076            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()),
077            HomeScreenMirrorRequest::new,
078            CurrentBasalStatusRequest::new,
079            TempRateRequest::new)),
080    BOLUS_CHANGE(1024, ImmutableSet.of(
081            CurrentBolusStatusRequest::new,
082            ExtendedBolusStatusRequest::new,
083            () -> LastBolusStatusRequestBuilder.create(PumpStateSupplier.pumpApiVersion.get()))),
084    IOB_CHANGE(2048, ImmutableSet.of(
085            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()))),
086    EXTENDED_BOLUS_CHANGE(4096, ImmutableSet.of(
087            ExtendedBolusStatusRequest::new,
088            () -> LastBolusStatusRequestBuilder.create(PumpStateSupplier.pumpApiVersion.get()))),
089    PROFILE_CHANGE(8192, ImmutableSet.of(
090            ProfileStatusRequest::new)),
091    BG(16384, ImmutableSet.of(
092            LastBGRequest::new)),
093    CGM_CHANGE(32768, ImmutableSet.of(
094            CGMStatusRequest::new,
095            CurrentEGVGuiDataRequest::new,
096            HomeScreenMirrorRequest::new)),
097    BATTERY(65536, ImmutableSet.of(
098            () -> CurrentBatteryRequestBuilder.create(PumpStateSupplier.pumpApiVersion.get()))),
099    BASAL_IQ(131072, ImmutableSet.of(
100            BasalIQSettingsRequest::new)),
101    REMAINING_INSULIN(262144, ImmutableSet.of(
102            InsulinStatusRequest::new)),
103    SUSPEND_COMM(524288),
104    ACTIVE_SEGMENT_CHANGE(1048576, ImmutableSet.of(
105            ProfileStatusRequest::new)),
106    BASAL_IQ_STATUS(2097152, ImmutableSet.of(
107            BasalIQSettingsRequest::new,
108            BasalIQStatusRequest::new)),
109    CONTROL_IQ_INFO(4194304, ImmutableSet.of(
110            () -> IOBRequestBuilder.create(PumpStateSupplier.controlIQSupported.get()),
111            () -> ControlIQInfoRequestBuilder.create(PumpStateSupplier.pumpApiVersion.get()),
112            ControlIQSleepScheduleRequest::new)),
113    CONTROL_IQ_SLEEP(8388608, ImmutableSet.of(
114            ControlIQSleepScheduleRequest::new)),
115    BOLUS_PERMISSION_REVOKED(2147483648L),
116
117    ;
118    private final long id;
119    private final Set<Supplier<? extends Message>> suggestedHandlers;
120    QualifyingEvent(long id) {
121        this.id = id;
122        this.suggestedHandlers = new HashSet<>();
123    }
124
125    QualifyingEvent(long id, Set<Supplier<? extends Message>> suggestedHandlers) {
126        this.id = id;
127        this.suggestedHandlers = suggestedHandlers;
128    }
129
130    public final long getId() {
131        return id;
132    }
133
134    public final Set<Supplier<? extends Message>> getSuggestedHandlers() {
135        return suggestedHandlers;
136    }
137
138    public static Set<QualifyingEvent> fromBitmask(long bitmask) {
139        Set<QualifyingEvent> ret = new HashSet<>();
140        for (QualifyingEvent e : values()) {
141            if ((bitmask & e.getId()) != 0) {
142                ret.add(e);
143            }
144        }
145        return ret;
146    }
147
148    public static Set<QualifyingEvent> fromRawBtBytes(byte[] raw) {
149        return fromBitmask(Bytes.readUint32(raw, 0));
150    }
151
152    public static Set<? extends Message> groupSuggestedHandlers(Set<QualifyingEvent> events) {
153        Set<Supplier<? extends Message>> messages = new HashSet<>();
154        for (QualifyingEvent e : events) {
155            messages.addAll(e.suggestedHandlers);
156        }
157        return messages.stream().map(Supplier::get).collect(Collectors.toSet());
158    }
159
160    public static int toBitmask(QualifyingEvent...states) {
161        int bitmask = 0;
162        for (QualifyingEvent e : states) {
163            bitmask |= e.getId();
164        }
165        return bitmask;
166    }
167
168    public String jsonToString() {
169        JSONObject obj = new JSONObject();
170        obj.put("event", name());
171        return obj.toString(0);
172    }
173
174    public JSONObject asJsonObject() {
175        JSONObject obj = new JSONObject();
176        obj.put("event", name());
177        return obj;
178    }
