001package com.jwoglom.pumpx2.pump.messages.builders;
002
003import com.jwoglom.pumpx2.pump.messages.Message;
004import com.jwoglom.pumpx2.pump.messages.models.ApiVersion;
005import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
006import com.jwoglom.pumpx2.pump.messages.request.currentStatus.CurrentBatteryV1Request;
007import com.jwoglom.pumpx2.pump.messages.request.currentStatus.CurrentBatteryV2Request;
008
009/**
010 * On an earlier than V2_API pump, if CurrentBatteryV2Request is sent and a CurrentBatteryV1Request
011 * has never been sent to the pump, a CurrentBatteryV1Response is returned -- presumably the
012 * pump assumes that the client is a "new" client not aware of the legacy behavior. If a
013 * CurrentBatteryV1Request is sent after a CurrentBatteryV2Request, then an ErrorResponse
014 * is sent with BAD_OPCODE.
015 */
016public class CurrentBatteryRequestBuilder {
017    public static Message create(ApiVersion apiVersion) {
018        if (apiVersion.greaterThan(KnownApiVersion.API_V2_1)) {
019            return new CurrentBatteryV2Request();
020        } else {
021            return new CurrentBatteryV1Request();
022        }
023    }
024}
