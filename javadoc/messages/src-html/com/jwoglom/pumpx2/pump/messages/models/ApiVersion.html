001package com.jwoglom.pumpx2.pump.messages.models;
002
003public class ApiVersion {
004
005    private final int major;
006    private final int minor;
007    public ApiVersion(int major, int minor) {
008        this.major = major;
009        this.minor = minor;
010    }
011
012    public int getMajor() {
013        return major;
014    }
015
016    public int getMinor() {
017        return minor;
018    }
019
020    public boolean greaterThan(ApiVersion other) {
021        return getMajor() > other.getMajor() || (getMajor() == other.getMajor() && getMinor() > other.getMinor());
022    }
023
024    public boolean greaterThan(KnownApiVersion other) {
025        return greaterThan(other.get());
026    }
027
028    public String toString() {
029        return "ApiVersion(major=" + major + ", minor=" + minor + ")";
030    }
031}
