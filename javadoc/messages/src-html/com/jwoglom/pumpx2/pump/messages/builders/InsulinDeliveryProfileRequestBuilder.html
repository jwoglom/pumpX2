001package com.jwoglom.pumpx2.pump.messages.builders;
002
003import com.jwoglom.pumpx2.pump.messages.Message;
004import com.jwoglom.pumpx2.pump.messages.request.currentStatus.IDPSegmentRequest;
005import com.jwoglom.pumpx2.pump.messages.request.currentStatus.IDPSettingsRequest;
006import com.jwoglom.pumpx2.pump.messages.request.currentStatus.ProfileStatusRequest;
007import com.jwoglom.pumpx2.pump.messages.response.currentStatus.IDPSettingsResponse;
008import com.jwoglom.pumpx2.pump.messages.response.currentStatus.ProfileStatusResponse;
009
010import java.util.LinkedList;
011import java.util.Optional;
012import java.util.Queue;
013
014/**
015 * A builder which simplifies the logic for getting all insulin delivery profiles and segments.
016 * An IDP is a profile selected in the pump. The IDP contained within slot 0 is the current profile
017 * being used for active insulin delivery. The ID contained within the slot is then used to request
018 * details on that insulin profile, which contains segments for a period of time in which given
019 * insulin delivery options are used.
020 *
021 * Example usage:
022 * <pre>
023 *   InsulinDeliveryProfileRequestBuilder builder = new InsulinDeliveryProfileRequestBuilder();
024 *   request = builder.nextRequest();
025 *   pump.sendCommand(peripheral, request.get());
026 *
027 *   // Inside onRecieveMessage callback...
028 *   builder.processResponse(responseMessage);
029 *   request = builder.nextRequest();
030 *   if (request.isPresent()) {
031 *       pump.sendCommand(peripheral, request.get());
032 *   }
033 * </pre>
034 *
035 * An initial {@link ProfileStatusRequest}, followed by {@link IDPSettingsRequest} and {@link IDPSegmentRequest} with
036 * requisite parameters in order to capture the entire profile state will be returned by nextRequest()
037 * so long as you call processResponse() on the returned responses from the pump. nextRequest() will
038 * eventually return an empty optional, at which point all profile and segment data has been fetched.
039 */
040public class InsulinDeliveryProfileRequestBuilder {
041    private final Queue<Message> queue;
042
043    public InsulinDeliveryProfileRequestBuilder() {
044        queue = new LinkedList<>();
045        queue.add(new ProfileStatusRequest());
046    }
047
048    public Optional<Message> nextRequest() {
049        if (queue.isEmpty()) {
050            return Optional.empty();
051        }
052        return Optional.of(queue.poll());
053    }
054
055    public void processResponse(Message responseMessage) {
056        if (responseMessage instanceof ProfileStatusResponse) {
057            ProfileStatusResponse profiles = (ProfileStatusResponse) responseMessage;
058            profiles.getIdpSlotIds().forEach(idpId -> queue(new IDPSettingsRequest(idpId)));
059        } else if (responseMessage instanceof IDPSettingsResponse) {
060            IDPSettingsResponse settings = (IDPSettingsResponse) responseMessage;
061            for (int i=1; i<= settings.getNumberOfProfileSegments(); i++) {
062                queue(new IDPSegmentRequest(settings.getIdpId(), i));
063            }
064        }
065    }
066
067    private void queue(Message requestMessage) {
068        this.queue.add(requestMessage);
069    }
070}
