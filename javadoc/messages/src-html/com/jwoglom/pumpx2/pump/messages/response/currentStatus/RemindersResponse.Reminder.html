025    private Reminder highBGReminder;
026    private Reminder siteChangeReminder;

038    public RemindersResponse() {}
039    
040    public RemindersResponse(Reminder lowBGReminder, Reminder highBGReminder, Reminder siteChangeReminder, Reminder missedBolusReminder0, Reminder missedBolusReminder1, Reminder missedBolusReminder2, Reminder missedBolusReminder3, Reminder afterBolusReminder, Reminder additionalBolusReminder, int lowBGThreshold, int highBGThreshold, int siteChangeDays, int status) {
041        this.cargo = buildCargo(lowBGReminder, highBGReminder, siteChangeReminder, missedBolusReminder0, missedBolusReminder1, missedBolusReminder2, missedBolusReminder3, afterBolusReminder, additionalBolusReminder, lowBGThreshold, highBGThreshold, siteChangeDays, status);
042        this.lowBGReminder = lowBGReminder;
043        this.highBGReminder = highBGReminder;
044        this.siteChangeReminder = siteChangeReminder;
045        this.missedBolusReminder0 = missedBolusReminder0;
046        this.missedBolusReminder1 = missedBolusReminder1;
047        this.missedBolusReminder2 = missedBolusReminder2;
048        this.missedBolusReminder3 = missedBolusReminder3;
049        this.afterBolusReminder = afterBolusReminder;
050        this.additionalBolusReminder = additionalBolusReminder;
051        this.lowBGThreshold = lowBGThreshold;
052        this.highBGThreshold = highBGThreshold;
053        this.siteChangeDays = siteChangeDays;
054        this.status = status;
055        
056    }
057
058    public void parse(byte[] raw) {
059        Preconditions.checkArgument(raw.length == props().size());
060        this.cargo = raw;
061        this.lowBGReminder = new Reminder(raw, 0);
062        this.highBGReminder = new Reminder(raw, 11);
063        this.siteChangeReminder = new Reminder(raw, 22);
064        this.missedBolusReminder0 = new Reminder(raw, 33);
065        this.missedBolusReminder1 = new Reminder(raw, 44);
066        this.missedBolusReminder2 = new Reminder(raw, 55);
067        this.missedBolusReminder3 = new Reminder(raw, 66);
068        this.afterBolusReminder = new Reminder(raw, 77);
069        this.additionalBolusReminder = new Reminder(raw, 88);
070        this.lowBGThreshold = Bytes.readShort(raw, 99);
071        this.highBGThreshold = Bytes.readShort(raw, 101);
072        this.siteChangeDays = raw[103];
073        this.status = raw[104];
074        
075    }
076

078    public static byte[] buildCargo(Reminder lowBGReminder, Reminder highBGReminder, Reminder siteChangeReminder, Reminder missedBolusReminder0, Reminder missedBolusReminder1, Reminder missedBolusReminder2, Reminder missedBolusReminder3, Reminder afterBolusReminder, Reminder additionalBolusReminder, int lowBGThreshold, int highBGThreshold, int siteChangeDays, int status)  {
079        return Bytes.combine(
080            lowBGReminder.buildCargo(),
081            highBGReminder.buildCargo(),
082            siteChangeReminder.buildCargo(),
083            missedBolusReminder0.buildCargo(),
084            missedBolusReminder1.buildCargo(),
085            missedBolusReminder2.buildCargo(),
086            missedBolusReminder3.buildCargo(),
087            afterBolusReminder.buildCargo(),
088            additionalBolusReminder.buildCargo(),
089            Bytes.firstTwoBytesLittleEndian(lowBGThreshold), 
090            Bytes.firstTwoBytesLittleEndian(highBGThreshold), 
091            new byte[]{ (byte) siteChangeDays }, 
092            new byte[]{ (byte) status });
093    }

135    public static class Reminder {
136        private final long frequency;
137        private final int startTime;
138        private final int endTime;
139        private final int activeDays;
140        private final int enabled;
141        private final int validityStatus;
142
143        public Reminder(long frequency, int startTime, int endTime, int activeDays, int enabled, int validityStatus) {
144            this.frequency = frequency;
145            this.startTime = startTime;
146            this.endTime = endTime;
147            this.activeDays = activeDays;
148            this.enabled = enabled;
149            this.validityStatus = validityStatus; // TODO: figure out what this represents
150        }
151
152        public Reminder(byte[] cargo, int i) {
153            this.frequency = Bytes.readUint32(cargo, i);
154            this.startTime = Bytes.readShort(cargo, i+4);
155            this.endTime = Bytes.readShort(cargo, i+6);
156            this.activeDays = cargo[i+8];
157            this.enabled = cargo[i+9];
158            this.validityStatus = cargo[i+10];
159        }
160
161        public byte[] buildCargo() {
162            return Bytes.combine(
163                Bytes.toUint32(frequency),
164                    Bytes.firstTwoBytesLittleEndian(startTime),
165                    Bytes.firstTwoBytesLittleEndian(endTime),
166                    new byte[]{ (byte) activeDays},
167                    new byte[]{ (byte) enabled},
168                    new byte[]{ (byte) validityStatus});
169        }
170
171        public long getFrequencyMins() {
172            return frequency;
173        }
174
175        public MinsTime getFrequency() {
176            return new MinsTime((int) frequency);
177        }
178
179        public int getStartTimeMins() {
180            return startTime;
181        }
182
183        public MinsTime getStartTime() {
184            return new MinsTime(startTime);
185        }
186
187        public int getEndTimeMins() {
188            return endTime;
189        }
190
191        public MinsTime getEndTime() {
192            return new MinsTime(endTime);
193        }
194
195        public int getActiveDaysBitmask() {
196            return activeDays;
197        }
198
199        public Set<MultiDay> getActiveDays() {
200            return MultiDay.fromBitmask(activeDays);
201        }
202
203        public int getEnabled() {
204            return enabled;
205        }
206
207        public int getValidityStatus() {
208            return validityStatus;
209        }
210
211        public String toString() {
212            return JavaHelpers.autoToString(this, new HashSet<String>());
213        }
214    }
