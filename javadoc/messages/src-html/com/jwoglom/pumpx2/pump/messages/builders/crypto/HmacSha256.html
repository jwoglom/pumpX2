001package com.jwoglom.pumpx2.pump.messages.builders.crypto;
002
003import com.jwoglom.pumpx2.shared.L;
004
005import javax.crypto.Mac;
006import javax.crypto.spec.SecretKeySpec;
007
008public class HmacSha256 {
009    private static final String TAG = "HmacSha256";
010
011
012    private static byte mod255(int i) {
013        if (i < 0) {
014            return (byte) ((i + 255 + 1) & 255);
015        }
016        return (byte) i;
017    }
018
019    private static byte[] mod255(byte[] data) {
020        for (int i = 0; i < data.length; i++) {
021            byte b = data[i];
022            if (b < 0) {
023                data[i] = mod255(b);
024            }
025        }
026        return data;
027    }
028
029    public static byte[] hmacSha256(byte[] data, byte[] key) {
030        try {
031            SecretKeySpec secretKeySpec = new SecretKeySpec(mod255(key), "HmacSHA256");
032            Mac mac = Mac.getInstance("HmacSHA256");
033            mac.init(secretKeySpec);
034            return mac.doFinal(mod255(data));
035        } catch (Exception e) {
036            L.e(TAG, e);
037            return new byte[]{};
038        }
039    }
040}
