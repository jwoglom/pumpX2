001package com.jwoglom.pumpx2.pump.messages.bluetooth;
002
003import com.google.common.base.Strings;
004import com.jwoglom.pumpx2.pump.messages.builders.crypto.Hkdf;
005import com.jwoglom.pumpx2.pump.messages.models.ApiVersion;
006import com.jwoglom.pumpx2.shared.Hex;
007import com.jwoglom.pumpx2.shared.L;
008
009import org.apache.commons.codec.DecoderException;
010
011import java.nio.charset.StandardCharsets;
012import java.util.function.Supplier;
013
014public class PumpStateSupplier {
015    private static final String TAG = "PumpStateSupplier";
016
017    public static Supplier<byte[]> authenticationKey = PumpStateSupplier::determinePumpAuthKey;
018    public static Supplier<String> pumpPairingCode = null;
019    public static Supplier<String> jpakeDerivedSecretHex = null;
020    public static Supplier<String> jpakeServerNonceHex = null;
021    public static Supplier<Long> pumpTimeSinceReset = null;
022    public static Supplier<ApiVersion> pumpApiVersion = null;
023    public static Supplier<Boolean> controlIQSupported = () -> false;
024    public static Supplier<Boolean> actionsAffectingInsulinDeliveryEnabled = () -> false;
025
026
027    private static byte[] determinePumpAuthKey() {
028        String derivedSecret = jpakeDerivedSecretHex == null ? null : jpakeDerivedSecretHex.get();
029        String serverNonce = jpakeServerNonceHex == null ? null : jpakeServerNonceHex.get();
030        String code = pumpPairingCode == null ? null : pumpPairingCode.get();
031
032        if (derivedSecret == null && code == null) {
033            throw new IllegalStateException("no pump authenticationKey");
034        }
035
036        // stored jpake raw derived secret is decoded from hex for use in hmac
037        if (!Strings.isNullOrEmpty(derivedSecret) && !Strings.isNullOrEmpty(serverNonce)) {
038            try {
039                L.i(TAG, "PUMP_JPAKE_DERIVED_SECRET=" + derivedSecret + " PUMP_JPAKE_SERVER_NONCE=" + serverNonce);
040                byte[] jpakeSecret = Hex.decodeHex(derivedSecret);
041                byte[] jpakeNonce = Hex.decodeHex(serverNonce);
042                byte[] authKey = Hkdf.build(jpakeNonce, jpakeSecret);
043                L.i(TAG, "PUMP_AUTHENTICATION_KEY=" + Hex.encodeHexString(authKey) + " PUMP_JPAKE_DERIVED_SECRET=" + derivedSecret + " PUMP_JPAKE_SERVER_NONCE=" + serverNonce);
044
045                return authKey;
046            } catch (DecoderException e) {
047                L.e(TAG, e);
048            }
049        }
050
051        L.i(TAG, "PUMP_AUTHENTICATION_KEY=" + code);
052
053        // pairing code is passed as raw ascii to hmac
054        return code.getBytes(StandardCharsets.UTF_8);
055    }
056}
