001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.helpers.Dates;
010import com.jwoglom.pumpx2.pump.messages.request.currentStatus.LastBolusStatusRequest;
011import com.jwoglom.pumpx2.pump.messages.response.historyLog.BolusDeliveryHistoryLog;
012
013import java.math.BigInteger;
014import java.time.Instant;
015import java.util.Set;
016
017/**
018 * Does not contain requestedVolume which {@link LastBolusStatusV2Response} has
019 */
020@MessageProps(
021    opCode=49,
022    size=20,
023    type=MessageType.RESPONSE,
024    characteristic=Characteristic.CURRENT_STATUS,
025    request=LastBolusStatusRequest.class
026)
027public class LastBolusStatusResponse extends LastBolusStatusAbstractResponse {
028    
029    private int status;
030    private int bolusId;
031    private long timestamp;
032    private long deliveredVolume;
033    private int bolusStatusId;
034    private int bolusSourceId;
035    private int bolusTypeBitmask;
036    private long extendedBolusDuration;
037    private byte[] unknown;
038    
039    public LastBolusStatusResponse() {}
040    
041    public LastBolusStatusResponse(int status, int bolusId, long timestamp, long deliveredVolume, int bolusStatusId, int bolusSourceId, int bolusTypeBitmask, long extendedBolusDuration, byte[] unknown) {
042        this.cargo = buildCargo(status, bolusId, timestamp, deliveredVolume, bolusStatusId, bolusSourceId, bolusTypeBitmask, extendedBolusDuration, unknown);
043        this.status = status;
044        this.bolusId = bolusId;
045        this.timestamp = timestamp;
046        this.deliveredVolume = deliveredVolume;
047        this.bolusStatusId = bolusStatusId;
048        this.bolusSourceId = bolusSourceId;
049        this.bolusTypeBitmask = bolusTypeBitmask;
050        this.extendedBolusDuration = extendedBolusDuration;
051        this.unknown = unknown;
052    }
053
054    public void parse(byte[] raw) {
055        Preconditions.checkArgument(raw.length == props().size());
056        this.cargo = raw;
057        this.status = raw[0];
058        this.bolusId = Bytes.readShort(raw, 1);
059        this.unknown = new byte[]{raw[3], raw[4]};
060        this.timestamp = Bytes.readUint32(raw, 5);
061        this.deliveredVolume = Bytes.readUint32(raw, 9);
062        this.bolusStatusId = raw[13];
063        this.bolusSourceId = raw[14];
064        this.bolusTypeBitmask = raw[15];
065        this.extendedBolusDuration = Bytes.readUint32(raw, 16);
066    }
067
068    
069    public static byte[] buildCargo(int status, int bolusId, long timestamp, long deliveredVolume, int bolusStatusId, int bolusSourceId, int bolusTypeBitmask, long extendedBolusDuration, byte[] unknown) {
070        return Bytes.combine(
071            new byte[]{ (byte) status }, 
072            Bytes.firstTwoBytesLittleEndian(bolusId),
073            unknown,
074            Bytes.toUint32(timestamp), 
075            Bytes.toUint32(deliveredVolume), 
076            new byte[]{ (byte) bolusStatusId }, 
077            new byte[]{ (byte) bolusSourceId }, 
078            new byte[]{ (byte) bolusTypeBitmask }, 
079            Bytes.toUint32(extendedBolusDuration)
080        );
081    }
082
083    public int getStatus() {
084        return status;
085    }
086    public int getBolusId() {
087        return bolusId;
088    }
089    public long getTimestamp() {
090        return timestamp;
091    }
092
093    @Override
094    public Instant getTimestampInstant() {
095        return Dates.fromJan12008EpochSecondsToDate(timestamp);
096    }
097
098    public long getDeliveredVolume() {
099        return deliveredVolume;
100    }
101    public int getBolusStatusId() {
102        return bolusStatusId;
103    }
104    public int getBolusSourceId() {
105        return bolusSourceId;
106    }
107    public int getBolusTypeBitmask() {
108        return bolusTypeBitmask;
109    }
110    public Set<BolusDeliveryHistoryLog.BolusType> getBolusTypes() {
111        return BolusDeliveryHistoryLog.BolusType.fromBitmask(bolusTypeBitmask);
112    }
113    public long getExtendedBolusDuration() {
114        return extendedBolusDuration;
115    }
116    
117}
