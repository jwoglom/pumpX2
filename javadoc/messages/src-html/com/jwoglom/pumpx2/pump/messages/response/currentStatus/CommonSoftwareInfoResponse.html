001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.request.currentStatus.CommonSoftwareInfoRequest;
010
011@MessageProps(
012    opCode=-113,
013    size=60,
014    type=MessageType.RESPONSE,
015    characteristic=Characteristic.CURRENT_STATUS,
016    request=CommonSoftwareInfoRequest.class
017)
018public class CommonSoftwareInfoResponse extends Message {
019    
020    private String appSoftwareVersion;
021    private long appSoftwarePartNumber;
022    private long appSoftwarePartDashNumber;
023    private long appSoftwarePartRevisionNumber;
024    private String bootloaderVersion;
025    private long bootloaderPartNumber;
026    private long bootloaderPartDashNumber;
027    private long bootloaderPartRevisionNumber;
028    
029    public CommonSoftwareInfoResponse() {}
030    
031    public CommonSoftwareInfoResponse(String appSoftwareVersion, long appSoftwarePartNumber, long appSoftwarePartDashNumber, long appSoftwarePartRevisionNumber, String bootloaderVersion, long bootloaderPartNumber, long bootloaderPartDashNumber, long bootloaderPartRevisionNumber) {
032        this.cargo = buildCargo(appSoftwareVersion, appSoftwarePartNumber, appSoftwarePartDashNumber, appSoftwarePartRevisionNumber, bootloaderVersion, bootloaderPartNumber, bootloaderPartDashNumber, bootloaderPartRevisionNumber);
033        this.appSoftwareVersion = appSoftwareVersion;
034        this.appSoftwarePartNumber = appSoftwarePartNumber;
035        this.appSoftwarePartDashNumber = appSoftwarePartDashNumber;
036        this.appSoftwarePartRevisionNumber = appSoftwarePartRevisionNumber;
037        this.bootloaderVersion = bootloaderVersion;
038        this.bootloaderPartNumber = bootloaderPartNumber;
039        this.bootloaderPartDashNumber = bootloaderPartDashNumber;
040        this.bootloaderPartRevisionNumber = bootloaderPartRevisionNumber;
041        
042    }
043
044    public void parse(byte[] raw) {
045        Preconditions.checkArgument(raw.length == props().size());
046        this.cargo = raw;
047        this.appSoftwareVersion = Bytes.readString(raw, 0, 18);
048        this.appSoftwarePartNumber = Bytes.readUint32(raw, 18);
049        this.appSoftwarePartDashNumber = Bytes.readUint32(raw, 22);
050        this.appSoftwarePartRevisionNumber = Bytes.readUint32(raw, 26);
051        this.bootloaderVersion = Bytes.readString(raw, 30, 17);
052        this.bootloaderPartNumber = Bytes.readUint32(raw, 47);
053        this.bootloaderPartDashNumber = Bytes.readUint32(raw, 51);
054        this.bootloaderPartRevisionNumber = Bytes.readUint32(raw, 55);
055        
056    }
057
058    
059    public static byte[] buildCargo(String appSoftwareVersion, long appSoftwarePartNumber, long appSoftwareDashNumber, long appSoftwarePartRevisionNumber, String bootloaderVersion, long bootloaderPartNumber, long bootloaderPartDashNumber, long bootloaderPartRevisionNumber) {
060        return Bytes.combine(
061            Bytes.writeString(appSoftwareVersion, 18),
062            Bytes.toUint32(appSoftwarePartNumber), 
063            Bytes.toUint32(appSoftwareDashNumber), 
064            Bytes.toUint32(appSoftwarePartRevisionNumber), 
065            Bytes.writeString(bootloaderVersion, 17),
066            Bytes.toUint32(bootloaderPartNumber), 
067            Bytes.toUint32(bootloaderPartDashNumber), 
068            Bytes.toUint32(bootloaderPartRevisionNumber));
069    }
070    
071    public String getAppSoftwareVersion() {
072        return appSoftwareVersion;
073    }
074    public long getAppSoftwarePartNumber() {
075        return appSoftwarePartNumber;
076    }
077    public long getAppSoftwarePartDashNumber() {
078        return appSoftwarePartDashNumber;
079    }
080    public long getAppSoftwarePartRevisionNumber() {
081        return appSoftwarePartRevisionNumber;
082    }
083    public String getBootloaderVersion() {
084        return bootloaderVersion;
085    }
086    public long getBootloaderPartNumber() {
087        return bootloaderPartNumber;
088    }
089    public long getBootloaderPartDashNumber() {
090        return bootloaderPartDashNumber;
091    }
092    public long getBootloaderPartRevisionNumber() {
093        return bootloaderPartRevisionNumber;
094    }
095    
096}
