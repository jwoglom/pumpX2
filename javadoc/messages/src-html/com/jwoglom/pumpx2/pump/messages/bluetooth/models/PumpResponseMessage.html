001package com.jwoglom.pumpx2.pump.messages.bluetooth.models;
002
003import com.jwoglom.pumpx2.pump.messages.Message;
004
005import com.jwoglom.pumpx2.pump.messages.response.qualifyingEvent.QualifyingEvent;
006import com.jwoglom.pumpx2.shared.Hex;
007
008import java.util.Optional;
009import java.util.Set;
010
011public class PumpResponseMessage {
012    private final byte[] data;
013    private final Optional<Message> message;
014    private final Optional<Set<QualifyingEvent>> qualifyingEvents;
015
016    public PumpResponseMessage(byte[] data) {
017        this.data = data;
018        this.message = Optional.empty();
019        this.qualifyingEvents = Optional.empty();
020    }
021    public PumpResponseMessage(byte[] data, Message message) {
022        this.data = data;
023        this.message = Optional.of(message);
024        this.qualifyingEvents = Optional.empty();
025    }
026    public PumpResponseMessage(byte[] data, Set<QualifyingEvent> qualifyingEvents) {
027        this.data = data;
028        this.message = Optional.empty();
029        this.qualifyingEvents = Optional.of(qualifyingEvents);
030    }
031
032    public byte[] data() {
033        return data;
034    }
035
036    public Optional<Message> message() {
037        return message;
038    }
039
040    public Optional<Set<QualifyingEvent>> qualifyingEvents() {
041        return qualifyingEvents;
042    }
043
044    public String toString() {
045        return "PumpResponseMessage(data=" + Hex.encodeHexString(data) + ", message=" + message + ", event=" + qualifyingEvents + ")";
046    }
047}
