001package com.jwoglom.pumpx2.pump.messages.bluetooth.models;
002
003import com.jwoglom.pumpx2.pump.messages.Message;
004
005import com.jwoglom.pumpx2.shared.Hex;
006
007import java.util.Optional;
008
009public class PumpResponseMessage {
010    private final byte[] data;
011    private final Optional<Message> message;
012
013    public PumpResponseMessage(byte[] data) {
014        this.data = data;
015        this.message = Optional.empty();
016    }
017    public PumpResponseMessage(byte[] data, Message message) {
018        this.data = data;
019        this.message = Optional.of(message);
020    }
021
022    public byte[] data() {
023        return data;
024    }
025
026    public Optional<Message> message() {
027        return message;
028    }
029
030    public String toString() {
031        return "PumpResponseMessage(data=" + Hex.encodeHexString(data) + ", message=" + message + ")";
032    }
033}
