001package com.jwoglom.pumpx2.pump.messages.bluetooth.models;
002
003import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
004
005import kotlin.collections.ArraysKt;
006
007/**
008 * Packet represents the raw byte string included within a single Bluetooth response packet
009 *
010 * A combination of Packet's sent over a Bluetooth characteristic by one end of the BT communication
011 * represents one {@link com.jwoglom.pumpx2.pump.messages.Message}.
012 */
013public class Packet {
014    private final byte packetsRemaining;
015    private final byte transactionId;
016    private final byte[] internalCargo;
017
018    public Packet(byte packetsRemaining, byte transactionId, byte[] internalCargo) {
019        this.packetsRemaining = packetsRemaining;
020        this.transactionId = transactionId;
021        this.internalCargo = internalCargo;
022    }
023
024    public byte[] build() {
025        byte[] ret = new byte[internalCargo.length + 2];
026        ret[0] = (byte)packetsRemaining;
027        ret[1] = transactionId;
028        System.arraycopy(internalCargo, 0, ret, 2, internalCargo.length);
029        return ret;
030    }
031
032    public byte transactionId() {
033        return transactionId;
034    }
035
036    public byte[] internalCargo() {
037        return internalCargo;
038    }
039
040    public Packet merge(Packet newPacket) {
041        return new Packet(newPacket.packetsRemaining, transactionId, Bytes.combine(this.internalCargo, newPacket.internalCargo));
042    }
043
044}
