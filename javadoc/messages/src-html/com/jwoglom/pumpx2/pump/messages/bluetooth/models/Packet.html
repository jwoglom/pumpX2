001package com.jwoglom.pumpx2.pump.messages.bluetooth.models;
002
003import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
004
005import kotlin.collections.ArraysKt;
006
007public class Packet {
008    private final byte packetsRemaining;
009    private final byte transactionId;
010    private final byte[] internalCargo;
011
012    public Packet(byte packetsRemaining, byte transactionId, byte[] internalCargo) {
013        this.packetsRemaining = packetsRemaining;
014        this.transactionId = transactionId;
015        this.internalCargo = internalCargo;
016    }
017
018    public byte[] build() {
019        byte[] ret = new byte[internalCargo.length + 2];
020        ret[0] = (byte)packetsRemaining;
021        ret[1] = transactionId;
022        System.arraycopy(internalCargo, 0, ret, 2, internalCargo.length);
023        return ret;
024    }
025
026    public byte transactionId() {
027        return transactionId;
028    }
029
030    public byte[] internalCargo() {
031        return internalCargo;
032    }
033
034    public Packet merge(Packet newPacket) {
035        return new Packet(newPacket.packetsRemaining, transactionId, Bytes.combine(this.internalCargo, newPacket.internalCargo));
036    }
037
038}
