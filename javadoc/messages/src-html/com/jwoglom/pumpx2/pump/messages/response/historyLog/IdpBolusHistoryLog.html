001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007@HistoryLogProps(
008    opCode = 70,
009    displayName = "Personal Profile (IDP) Bolus Data Change",
010    usedByTidepool = true
011)
012public class IdpBolusHistoryLog extends HistoryLog {
013    
014    private int idp;
015    private int modification;
016    private int bolusStatus;
017    private int insulinDuration;
018    private int maxBolusSize;
019    private int bolusEntryType;
020    
021    public IdpBolusHistoryLog() {}
022    public IdpBolusHistoryLog(long pumpTimeSec, long sequenceNum, int idp, int modification, int bolusStatus, int insulinDuration, int maxBolusSize, int bolusEntryType) {
023        super(pumpTimeSec, sequenceNum);
024        this.cargo = buildCargo(pumpTimeSec, sequenceNum, idp, modification, bolusStatus, insulinDuration, maxBolusSize, bolusEntryType);
025        this.idp = idp;
026        this.modification = modification;
027        this.bolusStatus = bolusStatus;
028        this.insulinDuration = insulinDuration;
029        this.maxBolusSize = maxBolusSize;
030        this.bolusEntryType = bolusEntryType;
031        
032    }
033
034    public IdpBolusHistoryLog(int idp, int modification, int bolusStatus, int insulinDuration, int maxBolusSize, int bolusEntryType) {
035        this(0, 0, idp, modification, bolusStatus, insulinDuration, maxBolusSize, bolusEntryType);
036    }
037
038    public int typeId() {
039        return 70;
040    }
041
042    /**
043     * <table border="1">
044     * <caption>Parsed fields for IdpBolusHistoryLog:</caption>
045     * <tr><td><code>idp</code></td><td><code>raw[10]</code></td></tr>
046     * <tr><td><code>modification</code></td><td><code>raw[11]</code></td></tr>
047     * <tr><td><code>bolusStatus</code></td><td><code>raw[12]</code></td></tr>
048     * <tr><td><code>insulinDuration</code></td><td><code>Bytes.readShort(raw, 14)</code></td></tr>
049     * <tr><td><code>maxBolusSize</code></td><td><code>Bytes.readShort(raw, 16)</code></td></tr>
050     * <tr><td><code>bolusEntryType</code></td><td><code>raw[18]</code></td></tr>
051     * </table>
052     */
053    public void parse(byte[] raw) {
054        Preconditions.checkArgument(raw.length == 26);
055        this.cargo = raw;
056        parseBase(raw);
057        this.idp = raw[10];
058        this.modification = raw[11];
059        this.bolusStatus = raw[12];
060        this.insulinDuration = Bytes.readShort(raw, 14);
061        this.maxBolusSize = Bytes.readShort(raw, 16);
062        this.bolusEntryType = raw[18];
063        
064    }
065
066    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int idp, int modification, int bolusStatus, int insulinDuration, int maxBolusSize, int bolusEntryType) {
067        return HistoryLog.fillCargo(Bytes.combine(
068            new byte[]{70, 0},
069            Bytes.toUint32(pumpTimeSec),
070            Bytes.toUint32(sequenceNum),
071            new byte[]{ (byte) idp }, 
072            new byte[]{ (byte) modification }, 
073            new byte[]{ (byte) bolusStatus }, 
074            Bytes.firstTwoBytesLittleEndian(insulinDuration), 
075            Bytes.firstTwoBytesLittleEndian(maxBolusSize), 
076            new byte[]{ (byte) bolusEntryType }));
077    }
078    public int getIdp() {
079        return idp;
080    }
081    public int getModification() {

083    }
084    public int getBolusStatus() {

086    }
087    public int getInsulinDuration() {
088        return insulinDuration;
089    }
090    public int getMaxBolusSize() {
091        return maxBolusSize;
092    }
093    public int getBolusEntryType() {
094        return bolusEntryType;
095    }

097}
