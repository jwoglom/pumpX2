001package com.jwoglom.pumpx2.pump.messages.models;
002
003/**
004 * PairingCodeType represents different types of Tandem pairing codes:
005 * LONG_16CHAR is used on t:slim X2 before firmware v7.7 (KnownApiVersion.API_V2_5)
006 * SHORT_6CHAR is used on t:slim X2 on firmware v7.7 and above (KnownApiVersion.API_V3_2)
007 */
008public enum PairingCodeType {
009    LONG_16CHAR("LONG_16CHAR"),
010    SHORT_6CHAR("SHORT_6CHAR"),
011
012    ;
013
014    private final String label;
015    PairingCodeType(String label) {
016        this.label = label;
017    }
018
019    public String getLabel() {
020        return label;
021    }
022
023    public String filterCharacters(String pairingCode) {
024        String processed = "";
025        for (Character c : pairingCode.toCharArray()) {
026            if (this == LONG_16CHAR) {
027                // Remove all dashes and spaces
028                if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || (c >= '0' && c <= '9')) {
029                    processed += c;
030                }
031            } else if (this == SHORT_6CHAR) {
032                // Remove all non-numbers and spaces
033                if ((c >= '0' && c <= '9')) {
034                    processed += c;
035                }
036            }
037        }
038        return processed;
039    }
040
041    public static PairingCodeType fromLabel(String label) {
042        for (PairingCodeType t : values()) {
043            if (t.getLabel().equals(label)) {
044                return t;
045            }
046        }
047        return null;
048    }
049}
