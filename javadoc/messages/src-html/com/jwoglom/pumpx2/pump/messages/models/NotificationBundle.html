001package com.jwoglom.pumpx2.pump.messages.models;
002
003import com.google.common.base.Strings;
004import com.google.common.collect.ImmutableList;
005import com.jwoglom.pumpx2.pump.messages.Message;
006import com.jwoglom.pumpx2.pump.messages.request.currentStatus.AlarmStatusRequest;
007import com.jwoglom.pumpx2.pump.messages.request.currentStatus.AlertStatusRequest;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.CGMAlertStatusRequest;
009import com.jwoglom.pumpx2.pump.messages.request.currentStatus.MalfunctionStatusRequest;
010import com.jwoglom.pumpx2.pump.messages.request.currentStatus.OtherNotification2StatusRequest;
011import com.jwoglom.pumpx2.pump.messages.request.currentStatus.OtherNotificationStatusRequest;
012import com.jwoglom.pumpx2.pump.messages.response.currentStatus.AlarmStatusResponse;
013import com.jwoglom.pumpx2.pump.messages.response.currentStatus.AlertStatusResponse;
014import com.jwoglom.pumpx2.pump.messages.response.currentStatus.CGMAlertStatusResponse;
015import com.jwoglom.pumpx2.pump.messages.response.currentStatus.MalfunctionStatusResponse;
016import com.jwoglom.pumpx2.pump.messages.response.currentStatus.OtherNotification2StatusResponse;
017import com.jwoglom.pumpx2.pump.messages.response.currentStatus.OtherNotificationStatusResponse;
018import com.jwoglom.pumpx2.pump.messages.response.currentStatus.ReminderStatusResponse;
019
020import java.time.Instant;
021import java.util.ArrayList;
022import java.util.HashMap;
023import java.util.List;
024import java.util.Map;
025
026/**
027 * Bundle of notification related responses for easier processing
028 */
029public class NotificationBundle {
030    private AlertStatusResponse alertStatusResponse;
031    private ReminderStatusResponse reminderStatusResponse;
032    private AlarmStatusResponse alarmStatusResponse;
033    private CGMAlertStatusResponse cgmAlertStatusResponse;
034    private MalfunctionStatusResponse malfunctionStatusResponse;
035    private OtherNotificationStatusResponse otherNotificationStatusResponse;
036    private OtherNotification2StatusResponse otherNotification2StatusResponse;
037
038    private Map<Class<? extends Message>, Instant> lastUpdatedTimes = new HashMap<>();
039
040    public static List<Message> allRequests() {
041        return ImmutableList.of(
042                new AlertStatusRequest(),
043                new AlarmStatusRequest(),
044                new CGMAlertStatusRequest(),
045                new MalfunctionStatusRequest(),
046                new OtherNotification2StatusRequest(),
047                new OtherNotificationStatusRequest(new byte[]{2}),
048                new OtherNotificationStatusRequest(new byte[]{4})
049        );
050    }
051
052    public static List<Class<? extends Message>> responseClasses() {
053        return ImmutableList.of(
054                AlertStatusResponse.class,
055                ReminderStatusResponse.class,
056                AlarmStatusResponse.class,
057                CGMAlertStatusResponse.class,
058                MalfunctionStatusResponse.class,
059                OtherNotification2StatusResponse.class,
060                OtherNotificationStatusResponse.class
061        );
062    }
063
064    public static boolean isNotificationResponse(Message message) {
065        for (Class<? extends Message> m : responseClasses()) {
066            if (m.isInstance(message)) return true;
067        }
068
069        return false;
070    }
071
072    public NotificationBundle() {
073
074    }
075
076    public NotificationBundle add(Message message) {
077        if (!isNotificationResponse(message)) return this;
078
079        if (message instanceof AlertStatusResponse) {
080            alertStatusResponse = (AlertStatusResponse) message;
081            lastUpdatedTimes.put(AlertStatusResponse.class, Instant.now());
082        } else if (message instanceof ReminderStatusResponse) {
083            reminderStatusResponse = (ReminderStatusResponse) message;
084            lastUpdatedTimes.put(ReminderStatusResponse.class, Instant.now());
085        } else if (message instanceof AlarmStatusResponse) {
086            alarmStatusResponse = (AlarmStatusResponse) message;
087            lastUpdatedTimes.put(AlarmStatusResponse.class, Instant.now());
088        } else if (message instanceof CGMAlertStatusResponse) {
089            cgmAlertStatusResponse = (CGMAlertStatusResponse) message;
090            lastUpdatedTimes.put(CGMAlertStatusResponse.class, Instant.now());
091        } else if (message instanceof MalfunctionStatusResponse) {
092            malfunctionStatusResponse = (MalfunctionStatusResponse) message;
093            lastUpdatedTimes.put(MalfunctionStatusResponse.class, Instant.now());
094        } else if (message instanceof OtherNotificationStatusResponse) {
095            otherNotificationStatusResponse = (OtherNotificationStatusResponse) message;
096            lastUpdatedTimes.put(OtherNotificationStatusResponse.class, Instant.now());
097        } else if (message instanceof OtherNotification2StatusResponse) {
098            otherNotification2StatusResponse = (OtherNotification2StatusResponse) message;
099            lastUpdatedTimes.put(OtherNotification2StatusResponse.class, Instant.now());
100        }
101
102        return this;
103    }
104
105    /**
106     * @return list of raw alert/alarm enums or string errors
107     */
108    public List<Object> get() {
109        ArrayList<Object> slugs = new ArrayList<>();
110
111        if (alertStatusResponse != null) {
112            alertStatusResponse.getAlerts().stream().sorted().forEach(alert -> {
113                slugs.add(alert);
114            });
115        }
116        if (reminderStatusResponse != null) {
117            reminderStatusResponse.getReminders().stream().sorted().forEach(alert -> {
118                slugs.add(alert);
119            });
120        }
121        if (alarmStatusResponse != null) {
122            alarmStatusResponse.getAlarms().stream().sorted().forEach(alarm -> {
123                slugs.add(alarm);
124            });
125        }
126        if (cgmAlertStatusResponse != null) {
127            cgmAlertStatusResponse.getCgmAlerts().stream().sorted().forEach(cgmAlert -> {
128                slugs.add(cgmAlert);
129            });
130        }
131        if (malfunctionStatusResponse != null) {
132            if (!Strings.isNullOrEmpty(malfunctionStatusResponse.getErrorString())) {
133                slugs.add(malfunctionStatusResponse);
134            }
135        }
136        // todo remaining
137
138        return slugs;
139    }
140
141    public Map<Class<? extends Message>, Instant> getLastUpdatedTimes() {
142        return lastUpdatedTimes;
143    }
144
145}
