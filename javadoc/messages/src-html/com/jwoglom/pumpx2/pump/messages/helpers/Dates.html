001package com.jwoglom.pumpx2.pump.messages.helpers;
002
003import static java.time.ZoneOffset.UTC;
004
005import java.time.Instant;
006import java.time.temporal.TemporalAdjuster;
007
008public class Dates {
009    public static final long JANUARY_1_2008_UNIX_EPOCH = 1199145600;
010
011    public static long fromJan12008ToUnixEpochSeconds(long seconds) {
012        return seconds + JANUARY_1_2008_UNIX_EPOCH;
013    }
014
015    public static Instant fromJan12008EpochSecondsToDate(long seconds) {
016        return Instant.ofEpochSecond(fromJan12008ToUnixEpochSeconds(seconds));
017    }
018
019    private static final long SECONDS_IN_DAY = 60 * 60 * 24;
020
021    public static Instant fromJan12008EpochDaysToDate(long days) {
022        return Instant.ofEpochSecond(fromJan12008ToUnixEpochSeconds(days * SECONDS_IN_DAY));
023    }
024}
