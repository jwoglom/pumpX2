001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.ControlIQInfoV2Request;
009
010@MessageProps(
011    opCode=-77,
012    size=19,
013    type=MessageType.RESPONSE,
014    request=ControlIQInfoV2Request.class
015)
016public class ControlIQInfoV2Response extends ControlIQInfoAbstractResponse {
017    
018    private boolean closedLoopEnabled;
019    private int weight;
020    private int weightUnit;
021    private int totalDailyInsulin;
022    private int currentUserModeType;
023    private int byte6;
024    private int byte7;
025    private int byte8;
026    private int controlStateType;
027    private int exerciseChoice;
028    private int exerciseDuration;
029    
030    public ControlIQInfoV2Response() {}
031    
032    public ControlIQInfoV2Response(boolean closedLoopEnabled, int weight, int weightUnit, int totalDailyInsulin, int currentUserModeType, int byte6, int byte7, int byte8, int controlStateType, int exerciseChoice, int exerciseDuration) {
033        this.cargo = buildCargo(closedLoopEnabled, weight, weightUnit, totalDailyInsulin, currentUserModeType, byte6, byte7, byte8, controlStateType, exerciseChoice, exerciseDuration);
034        this.closedLoopEnabled = closedLoopEnabled;
035        this.weight = weight;
036        this.weightUnit = weightUnit;
037        this.totalDailyInsulin = totalDailyInsulin;
038        this.currentUserModeType = currentUserModeType;
039        this.byte6 = byte6;
040        this.byte7 = byte7;
041        this.byte8 = byte8;
042        this.controlStateType = controlStateType;
043        this.exerciseChoice = exerciseChoice;
044        this.exerciseDuration = exerciseDuration;
045        
046    }
047
048    public void parse(byte[] raw) {
049        Preconditions.checkArgument(raw.length == props().size());
050        this.cargo = raw;
051        this.closedLoopEnabled = raw[0] != 0;
052        this.weight = Bytes.readShort(raw, 1);
053        this.weightUnit = raw[3];
054        this.totalDailyInsulin = raw[4];
055        this.currentUserModeType = raw[5];
056        this.byte6 = raw[6];
057        this.byte7 = raw[7];
058        this.byte8 = raw[8];
059        this.controlStateType = raw[9];
060        this.exerciseChoice = raw[10];
061        this.exerciseDuration = raw[11];
062        
063    }
064
065    
066    public static byte[] buildCargo(boolean closedLoopEnabled, int weight, int weightUnit, int totalDailyInsulin, int currentUserModeType, int byte6, int byte7, int byte8, int controlStateType, int exerciseChoice, int exerciseDuration) {
067        return Bytes.combine(
068            new byte[]{ (byte) (closedLoopEnabled ? 1 : 0) }, 
069            Bytes.firstTwoBytesLittleEndian(weight), 
070            new byte[]{ (byte) weightUnit }, 
071            new byte[]{ (byte) totalDailyInsulin }, 
072            new byte[]{ (byte) currentUserModeType }, 
073            new byte[]{ (byte) byte6 }, 
074            new byte[]{ (byte) byte7 }, 
075            new byte[]{ (byte) byte8 }, 
076            new byte[]{ (byte) controlStateType }, 
077            new byte[]{ (byte) exerciseChoice }, 
078            new byte[]{ (byte) exerciseDuration },
079            Bytes.emptyBytes(7));
080    }
081    
082    public boolean getClosedLoopEnabled() {
083        return closedLoopEnabled;
084    }
085    public int getWeight() {
086        return weight;
087    }
088    public int getWeightUnitId() {
089        return weightUnit;
090    }
091    public int getTotalDailyInsulin() {
092        return totalDailyInsulin;
093    }
094    public int getCurrentUserModeTypeId() {
095        return currentUserModeType;
096    }
097    public int getByte6() {
098        return byte6;
099    }
100    public int getByte7() {
101        return byte7;
102    }
103    public int getByte8() {
104        return byte8;
105    }
106    public int getControlStateType() {
107        return controlStateType;
108    }
109    public int getExerciseChoice() {
110        return exerciseChoice;
111    }
112    public int getExerciseDuration() {
113        return exerciseDuration;
114    }
115    
116}
