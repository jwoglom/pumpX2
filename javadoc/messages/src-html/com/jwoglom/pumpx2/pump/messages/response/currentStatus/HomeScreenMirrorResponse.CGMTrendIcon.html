030    public HomeScreenMirrorResponse(int cgmTrendIcon, int cgmAlertIcon, int statusIcon0, int statusIcon1, int bolusStatusIcon, int basalStatusIcon, int apControlStateIcon, boolean remainingInsulinPlusIcon, boolean cgmDisplayData) {
031        this.cargo = buildCargo(cgmTrendIcon, cgmAlertIcon, statusIcon0, statusIcon1, bolusStatusIcon, basalStatusIcon, apControlStateIcon, remainingInsulinPlusIcon, cgmDisplayData);
032        this.cgmTrendIcon = cgmTrendIcon;
033        this.cgmAlertIcon = cgmAlertIcon;
034        this.statusIcon0 = statusIcon0;
035        this.statusIcon1 = statusIcon1;
036        this.bolusStatusIcon = bolusStatusIcon;
037        this.basalStatusIcon = basalStatusIcon;
038        this.apControlStateIcon = apControlStateIcon;
039        this.remainingInsulinPlusIcon = remainingInsulinPlusIcon;
040        this.cgmDisplayData = cgmDisplayData;
041        
042    }
043
044    public void parse(byte[] raw) {
045        Preconditions.checkArgument(raw.length == props().size());
046        this.cargo = raw;
047        this.cgmTrendIcon = Byte.toUnsignedInt(raw[0]);
048        this.cgmAlertIcon = Byte.toUnsignedInt(raw[1]);
049        this.statusIcon0 = Byte.toUnsignedInt(raw[2]);
050        this.statusIcon1 = Byte.toUnsignedInt(raw[3]);
051        this.bolusStatusIcon = Byte.toUnsignedInt(raw[4]);
052        this.basalStatusIcon = Byte.toUnsignedInt(raw[5]);
053        this.apControlStateIcon = Byte.toUnsignedInt(raw[6]);
054        this.remainingInsulinPlusIcon = raw[7] != 0;
055        this.cgmDisplayData = raw[8] != 0;
056        
057    }
058

060    public static byte[] buildCargo(int cgmTrendIcon, int cgmAlertIcon, int statusIcon0, int statusIcon1, int bolusStatusIcon, int basalStatusIcon, int apControlStateIcon, boolean remainingInsulinPlusIcon, boolean cgmDisplayData) {
061        return Bytes.combine(
062            new byte[]{ (byte) cgmTrendIcon }, 
063            new byte[]{ (byte) cgmAlertIcon }, 
064            new byte[]{ (byte) statusIcon0 }, 
065            new byte[]{ (byte) statusIcon1 }, 
066            new byte[]{ (byte) bolusStatusIcon }, 
067            new byte[]{ (byte) basalStatusIcon }, 
068            new byte[]{ (byte) apControlStateIcon }, 
069            new byte[]{ (byte) (remainingInsulinPlusIcon ? 1 : 0) }, 
070            new byte[]{ (byte) (cgmDisplayData ? 1 : 0) });
071    }
