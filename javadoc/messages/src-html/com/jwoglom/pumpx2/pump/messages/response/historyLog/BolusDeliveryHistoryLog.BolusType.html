001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007import java.util.HashSet;
008import java.util.Set;
009
010@HistoryLogProps(
011    opCode = 280,
012    usedByAndroid = true
013)
014public class BolusDeliveryHistoryLog extends HistoryLog {
015    
016    private int bolusID;
017    private int bolusDeliveryStatus;
018    private int bolusTypeBitmask;
019    private int bolusSource;
020    private int reserved;
021    private int requestedNow;
022    private int requestedLater;
023    private int correction;
024    private int extendedDurationRequested;
025    private int deliveredTotal;
026    
027    public BolusDeliveryHistoryLog() {}
028    
029    public BolusDeliveryHistoryLog(long pumpTimeSec, long sequenceNum, int bolusID, int bolusDeliveryStatus, Set<BolusType> bolusTypes, BolusSource bolusSource, int reserved, int requestedNow, int requestedLater, int correction, int extendedDurationRequested, int deliveredTotal) {
030        this.cargo = buildCargo(pumpTimeSec, sequenceNum, bolusID, bolusDeliveryStatus, BolusType.toBitmask(bolusTypes.toArray(new BolusType[]{})), bolusSource.id(), reserved, requestedNow, requestedLater, correction, extendedDurationRequested, deliveredTotal);
031        this.pumpTimeSec = pumpTimeSec;
032        this.sequenceNum = sequenceNum;
033        this.bolusID = bolusID;
034        this.bolusDeliveryStatus = bolusDeliveryStatus;
035        this.bolusTypeBitmask = BolusType.toBitmask(bolusTypes.toArray(new BolusType[]{}));
036        this.bolusSource = bolusSource.id();
037        this.reserved = reserved;
038        this.requestedNow = requestedNow;
039        this.requestedLater = requestedLater;
040        this.correction = correction;
041        this.extendedDurationRequested = extendedDurationRequested;
042        this.deliveredTotal = deliveredTotal;
043        
044    }
045
046    public int typeId() {
047        return 280;
048    }
049
050    public void parse(byte[] raw) {
051        Preconditions.checkArgument(raw.length == 26);
052        this.cargo = raw;
053        parseBase(raw);
054        this.bolusID = Bytes.readShort(raw, 10);
055        this.bolusDeliveryStatus = raw[12];
056        this.bolusTypeBitmask = raw[13];
057        this.bolusSource = raw[14];
058        this.reserved = raw[15];
059        this.requestedNow = Bytes.readShort(raw, 16);
060        this.requestedLater = Bytes.readShort(raw, 18);
061        this.correction = Bytes.readShort(raw, 20);
062        this.extendedDurationRequested = Bytes.readShort(raw, 22);
063        this.deliveredTotal = Bytes.readShort(raw, 24);
064    }
065
066    
067    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int bolusID, int bolusDeliveryStatus, int bolusType, int bolusSource, int reserved, int requestedNow, int requestedLater, int correction, int extendedDurationRequested, int deliveredTotal) {
068        return Bytes.combine(
069            new byte[]{24, 1}, // 280 across 2 bytes
070            Bytes.toUint32(pumpTimeSec),
071            Bytes.toUint32(sequenceNum),
072            Bytes.firstTwoBytesLittleEndian(bolusID), 
073            new byte[]{ (byte) bolusDeliveryStatus }, 
074            new byte[]{ (byte) bolusType }, 
075            new byte[]{ (byte) bolusSource }, 
076            new byte[]{ (byte) reserved },
077            Bytes.firstTwoBytesLittleEndian(requestedNow),
078            Bytes.firstTwoBytesLittleEndian(requestedLater), 
079            Bytes.firstTwoBytesLittleEndian(correction), 
080            Bytes.firstTwoBytesLittleEndian(extendedDurationRequested), 
081            Bytes.firstTwoBytesLittleEndian(deliveredTotal));
082    }
083    
084    public int getBolusID() {
085        return bolusID;
086    }
087    public int getBolusDeliveryStatus() {
088        return bolusDeliveryStatus;
089    }
090    public int getBolusTypeBitmask() {
091        return bolusTypeBitmask;
092    }
093    public int getBolusSourceId() {
094        return bolusSource;
095    }
096    public int getReserved() {
097        return reserved;
098    }
099    public int getRequestedNow() {
100        return requestedNow;
101    }
102    public int getRequestedLater() {
103        return requestedLater;
104    }
105    public int getCorrection() {
106        return correction;
107    }
108    public int getExtendedDurationRequested() {
109        return extendedDurationRequested;
110    }
111    public int getDeliveredTotal() {
112        return deliveredTotal;
113    }
114
115    public BolusSource getBolusSource() {
116        return BolusSource.fromId(bolusSource);
117    }
118
119    public enum BolusSource {
120        QUICK_BOLUS(0),
121        GUI(1),
122        CONTROL_IQ_AUTO_BOLUS(7),
123        BLUETOOTH_REMOTE_BOLUS(8),
124
125        ;
126        private int id;
127        BolusSource(int id) {
128            this.id = id;
129        }
130        public int id() {
131            return id;
132        }
133        public static BolusSource fromId(int id) {
134            for (BolusSource b : values()) {
135                if (b.id == id) return b;
136            }
137            return null;
138        }
139    }
140
141    public Set<BolusType> getBolusTypes() {
142        return BolusType.fromBitmask(getBolusTypeBitmask());
143    }
144
145    public enum BolusType {
146        FOOD1(1), // used when there is carbs
147        CORRECTION(2), // used when there is a correction amount
148        EXTENDED(4),
149        FOOD2(8), // used when there is no carbs and just a distinct bolus amount
150        ;
151        private int mask;
152        BolusType(int mask) {
153            this.mask = mask;
154        }
155        public int mask() {
156            return mask;
157        }
158        public static Set<BolusType> fromBitmask(int bitmask) {
159            Set<BolusType> ret = new HashSet<>();
160            for (BolusType b : values()) {
161                if ((bitmask & b.mask()) != 0) {
162                    ret.add(b);
163                }
164            }
165            return ret;
166        }
167        public static int toBitmask(BolusType ...types) {
168            int bitmask = 0;
169            for (BolusType b : types) {
170                bitmask |= b.mask();
171            }
172            return bitmask;
173        }
174    }
