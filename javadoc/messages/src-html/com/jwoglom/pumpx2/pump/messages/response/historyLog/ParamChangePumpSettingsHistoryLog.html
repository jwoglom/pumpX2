001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006
007@HistoryLogProps(
008    opCode = 73,
009    displayName = "Pump Settings Parameter Change",
010    usedByTidepool = true
011)
012public class ParamChangePumpSettingsHistoryLog extends HistoryLog {
013    
014    private int modification;
015    private int status;
016    private int lowInsulinThreshold;
017    private int cannulaPrimeSize;
018    private int isFeatureLocked;
019    private int autoShutdownEnabled;
020    private int oledTimeout;
021    private int autoShutdownDuration;
022    
023    public ParamChangePumpSettingsHistoryLog() {}
024    public ParamChangePumpSettingsHistoryLog(long pumpTimeSec, long sequenceNum, int modification, int status, int lowInsulinThreshold, int cannulaPrimeSize, int isFeatureLocked, int autoShutdownEnabled, int oledTimeout, int autoShutdownDuration) {
025        super(pumpTimeSec, sequenceNum);
026        this.cargo = buildCargo(pumpTimeSec, sequenceNum, modification, status, lowInsulinThreshold, cannulaPrimeSize, isFeatureLocked, autoShutdownEnabled, oledTimeout, autoShutdownDuration);
027        this.modification = modification;
028        this.status = status;
029        this.lowInsulinThreshold = lowInsulinThreshold;
030        this.cannulaPrimeSize = cannulaPrimeSize;
031        this.isFeatureLocked = isFeatureLocked;
032        this.autoShutdownEnabled = autoShutdownEnabled;
033        this.oledTimeout = oledTimeout;
034        this.autoShutdownDuration = autoShutdownDuration;
035        
036    }
037
038    public ParamChangePumpSettingsHistoryLog(int modification, int status, int lowInsulinThreshold, int cannulaPrimeSize, int isFeatureLocked, int autoShutdownEnabled, int oledTimeout, int autoShutdownDuration) {
039        this(0, 0, modification, status, lowInsulinThreshold, cannulaPrimeSize, isFeatureLocked, autoShutdownEnabled, oledTimeout, autoShutdownDuration);
040    }
041
042    public int typeId() {
043        return 73;
044    }
045
046    public void parse(byte[] raw) {
047        Preconditions.checkArgument(raw.length == 26);
048        this.cargo = raw;
049        parseBase(raw);
050        this.modification = raw[10];
051        this.status = Bytes.readShort(raw, 12);
052        this.lowInsulinThreshold = raw[14];
053        this.cannulaPrimeSize = raw[15];
054        this.isFeatureLocked = raw[16];
055        this.autoShutdownEnabled = raw[17];
056        this.oledTimeout = raw[19];
057        this.autoShutdownDuration = Bytes.readShort(raw, 20);
058        
059    }
060
061    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int modification, int status, int lowInsulinThreshold, int cannulaPrimeSize, int isFeatureLocked, int autoShutdownEnabled, int oledTimeout, int autoShutdownDuration) {
062        return HistoryLog.fillCargo(Bytes.combine(
063            new byte[]{73, 0},
064            Bytes.toUint32(pumpTimeSec),
065            Bytes.toUint32(sequenceNum),
066            new byte[]{ (byte) modification }, 
067            Bytes.firstTwoBytesLittleEndian(status), 
068            new byte[]{ (byte) lowInsulinThreshold }, 
069            new byte[]{ (byte) cannulaPrimeSize }, 
070            new byte[]{ (byte) isFeatureLocked }, 
071            new byte[]{ (byte) autoShutdownEnabled }, 
072            new byte[]{ (byte) oledTimeout }, 
073            Bytes.firstTwoBytesLittleEndian(autoShutdownDuration)));
074    }
075    public int getModification() {
076        return modification;
077    }
078    public int getStatus() {
079        return status;
080    }
081    public int getLowInsulinThreshold() {
082        return lowInsulinThreshold;
083    }
084    public int getCannulaPrimeSize() {
085        return cannulaPrimeSize;
086    }
087    public int getIsFeatureLocked() {
088        return isFeatureLocked;
089    }
090    public int getAutoShutdownEnabled() {
091        return autoShutdownEnabled;
092    }
093    public int getOledTimeout() {
094        return oledTimeout;
095    }
096    public int getAutoShutdownDuration() {
097        return autoShutdownDuration;
098    }
099    
100}
