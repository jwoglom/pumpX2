279    public static Map<Pair<Characteristic, Integer>, Class<? extends Message>> OPCODES = new HashMap<>();
280    public static Map<Integer, Messages> REQUESTS = new HashMap<>();
281    public static Map<Integer, Messages> RESPONSES = new HashMap<>();
282
283    static {
284        for (Messages m : Messages.values()) {
285            OPCODES.put(Pair.of(m.requestProps().characteristic(), m.requestOpCode), m.requestClass);
286            REQUESTS.put(m.requestOpCode, m);
287
288            OPCODES.put(Pair.of(m.responseProps().characteristic(), m.responseOpCode), m.responseClass);
289            RESPONSES.put(m.responseOpCode, m);
290
291            Preconditions.checkArgument(m.requestProps().minApi().equals(m.responseProps().minApi()), "the minApi should match for " + m);
292        }
293
294        for (Characteristic c : Characteristic.values()) {
295            int errorOpcode = new ErrorResponse().opCode();
296            if (!OPCODES.containsKey(Pair.of(c, errorOpcode))) {
297                OPCODES.put(Pair.of(c, errorOpcode), ErrorResponse.class);
298            }
299        }
300    }

302    public static Message parse(byte[] data, int opCode, Characteristic characteristic) {
303        try {
304            Class<? extends Message> clazz = OPCODES.get(Pair.of(characteristic, opCode));
305            if (clazz == null) {
306                L.w(TAG, "Unable to find message for opCode: " + opCode +" for " + characteristic + " with data: " + Hex.encodeHexString(data));
307                return null;
308            }
309            Message msg = clazz.newInstance();
310            msg.parse(data);
311            return msg;
312        } catch (Exception e) {
313            L.w(TAG, "Unable to invoke parse of data: " + Hex.encodeHexString(data) + " opCode: " + opCode + " " + characteristic, e);
314            e.printStackTrace();
315            return null;
316        }
317    }

319    public static Set<Characteristic> findPossibleCharacteristicsForOpcode(int opCode) {
320        Set<Characteristic> items = new HashSet<>();
321        for (Pair<Characteristic, Integer> entry : OPCODES.keySet()) {
322            if (entry.getRight().equals(opCode)) {
323                items.add(entry.getLeft());
324            }
325        }
326
327        return items;
328    }

330    public static Set<Class<? extends Message>> findPossibleMessagesForCharacteristic(Characteristic characteristic) {
331        Set<Class<? extends Message>> items = new HashSet<>();
332        for (Map.Entry<Pair<Characteristic, Integer>, Class<? extends Message>> entry : OPCODES.entrySet()) {
333            if (entry.getKey().getLeft().equals(characteristic)) {
334                items.add(entry.getValue());
335            }
336        }
337
338        return items;
339    }
