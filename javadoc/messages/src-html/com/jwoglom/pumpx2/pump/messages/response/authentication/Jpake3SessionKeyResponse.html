001package com.jwoglom.pumpx2.pump.messages.response.authentication;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.Message;
005import com.jwoglom.pumpx2.pump.messages.MessageType;
006import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
007import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
008import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
009import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
010import com.jwoglom.pumpx2.pump.messages.request.authentication.Jpake3SessionKeyRequest;
011import com.jwoglom.pumpx2.shared.Hex;
012
013import java.util.Arrays;
014
015@MessageProps(
016    opCode=39,
017    size=18,
018    type=MessageType.RESPONSE,
019    minApi=KnownApiVersion.API_V3_2,
020    characteristic=Characteristic.AUTHORIZATION,
021    request=Jpake3SessionKeyRequest.class
022)
023public class Jpake3SessionKeyResponse extends Message {
024    private int appInstanceId;
025    private byte[] deviceKeyNonce;
026    private byte[] deviceKeyReserved;
027
028    public static byte[] RESERVED = new byte[]{0,0,0,0,0,0,0,0};
029
030    public Jpake3SessionKeyResponse() {}
031
032    public Jpake3SessionKeyResponse(byte[] raw) {
033        parse(raw);
034    }
035
036    public Jpake3SessionKeyResponse(int appInstanceId, byte[] nonce, byte[] reserved) {
037        parse(buildCargo(appInstanceId, nonce, reserved));
038        Preconditions.checkState(this.appInstanceId == appInstanceId);
039        Preconditions.checkState(Hex.encodeHexString(this.deviceKeyNonce).equals(Hex.encodeHexString(nonce)));
040        Preconditions.checkState(Hex.encodeHexString(this.deviceKeyReserved).equals(Hex.encodeHexString(reserved)));
041    }
042
043    public void parse(byte[] raw) {
044        Preconditions.checkArgument(raw.length == props().size());
045        this.cargo = raw;
046        this.appInstanceId = Bytes.readShort(Arrays.copyOfRange(raw, 0, 2), 0);
047        this.deviceKeyNonce = Arrays.copyOfRange(raw, 2, 10); // 8
048        this.deviceKeyReserved = Arrays.copyOfRange(raw, 10, 18); // 8
049    }
050
051    public static byte[] buildCargo(int appInstanceId, byte[] nonce, byte[] reserved) {
052        return Bytes.combine(
053                Bytes.firstTwoBytesLittleEndian(appInstanceId),
054                nonce,
055                reserved
056        );
057    }
058
059    public int getAppInstanceId() {
060        return appInstanceId;
061    }
062
063    public byte[] getDeviceKeyNonce() {
064        return deviceKeyNonce;
065    }
066
067    public byte[] getDeviceKeyReserved() {
068        return deviceKeyReserved;
069    }
070}
