001package com.jwoglom.pumpx2.pump.messages.response.historyLog;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.annotations.HistoryLogProps;

007import java.math.BigInteger;
008
009@HistoryLogProps(
010    opCode = 55,
011    displayName = "Bolus Activated",
012    usedByTidepool = true
013)
014public class BolusActivatedHistoryLog extends HistoryLog {
015    
016    private int bolusId;
017    private float iob;
018    private float bolusSize;
019    
020    public BolusActivatedHistoryLog() {}
021    public BolusActivatedHistoryLog(long pumpTimeSec, long sequenceNum, int bolusId, float iob, float bolusSize) {
022        super(pumpTimeSec, sequenceNum);
023        this.cargo = buildCargo(pumpTimeSec, sequenceNum, bolusId, iob, bolusSize);
024        this.bolusId = bolusId;
025        this.iob = iob;
026        this.bolusSize = bolusSize;
027        
028    }
029
030    public BolusActivatedHistoryLog(int bolusId, float iob, float bolusSize) {
031        this(0, 0, bolusId, iob, bolusSize);
032    }
033
034    public int typeId() {
035        return 55;
036    }
037
038    /**
039     * <table border="1">
040     * <caption>Parsed fields for BolusActivatedHistoryLog:</caption>
041     * <tr><td><code>bolusId</code></td><td><code>Bytes.readShort(raw, 10)</code></td></tr>
042     * <tr><td><code>iob</code></td><td><code>Bytes.readFloat(raw, 14)</code></td></tr>
043     * <tr><td><code>bolusSize</code></td><td><code>Bytes.readFloat(raw, 18)</code></td></tr>
044     * </table>
045     */
046    public void parse(byte[] raw) {
047        Preconditions.checkArgument(raw.length == 26);
048        this.cargo = raw;
049        parseBase(raw);
050        this.bolusId = Bytes.readShort(raw, 10);
051        this.iob = Bytes.readFloat(raw, 14);
052        this.bolusSize = Bytes.readFloat(raw, 18);
053        
054    }
055
056    public static byte[] buildCargo(long pumpTimeSec, long sequenceNum, int bolusId, float iob, float bolusSize) {
057        return HistoryLog.fillCargo(Bytes.combine(
058            new byte[]{55, 0},
059            Bytes.toUint32(pumpTimeSec),
060            Bytes.toUint32(sequenceNum),
061            Bytes.firstTwoBytesLittleEndian(bolusId), 
062            Bytes.toFloat(iob), 
063            Bytes.toFloat(bolusSize)));
064    }
065    public int getBolusId() {
066        return bolusId;
067    }
068    public float getIob() {
069        return iob;
070    }
071    public float getBolusSize() {
072        return bolusSize;
073    }
074    
075}
