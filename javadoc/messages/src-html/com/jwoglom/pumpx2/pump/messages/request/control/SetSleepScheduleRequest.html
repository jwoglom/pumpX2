001package com.jwoglom.pumpx2.pump.messages.request.control;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.bluetooth.Characteristic;
005import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
006import com.jwoglom.pumpx2.pump.messages.Message;
007import com.jwoglom.pumpx2.pump.messages.MessageType;
008import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
009import com.jwoglom.pumpx2.pump.messages.models.KnownApiVersion;
010import com.jwoglom.pumpx2.pump.messages.models.SupportedDevices;
011import com.jwoglom.pumpx2.pump.messages.response.control.SetSleepScheduleResponse;
012import com.jwoglom.pumpx2.pump.messages.response.currentStatus.ControlIQSleepScheduleResponse;
013
014@MessageProps(
015    opCode=-50,
016    size=8,
017    type=MessageType.REQUEST,
018    characteristic=Characteristic.CONTROL,
019    signed=true,
020    minApi=KnownApiVersion.MOBI_API_V3_5,
021    supportedDevices=SupportedDevices.MOBI_ONLY,
022    response=SetSleepScheduleResponse.class
023)
024public class SetSleepScheduleRequest extends Message {
025
026    private int slot;
027    private byte[] rawSchedule;
028    private int flag;
029    
030    public SetSleepScheduleRequest() {}
031
032    public SetSleepScheduleRequest(int slot, byte[] rawSchedule, int flag) {
033        this.cargo = buildCargo(slot, rawSchedule, flag);
034    }
035
036    public SetSleepScheduleRequest(int slot, ControlIQSleepScheduleResponse.SleepSchedule schedule, int flag) {
037        this.cargo = buildCargo(slot, schedule.build(), flag);
038    }
039
040    public SetSleepScheduleRequest(byte[] raw) {
041        parse(raw);
042        
043    }
044
045    public void parse(byte[] raw) {
046        raw = this.removeSignedRequestHmacBytes(raw);
047        Preconditions.checkArgument(raw.length == props().size());
048        this.cargo = raw;
049        this.slot = raw[0];
050        this.rawSchedule = Bytes.dropLastN(Bytes.dropFirstN(raw, 1), 1);
051        this.flag = raw[7];
052
053    }
054
055    
056    public static byte[] buildCargo(int slot, byte[] schedule, int flag) {
057        return Bytes.combine(
058            new byte[]{ (byte) slot },
059            schedule,
060            new byte[]{ (byte) flag }
061        );
062    }
063
064    public int getSlot() {
065        return slot;
066    }
067
068    public byte[] getRawSchedule() {
069        return rawSchedule;
070    }
071
072    public ControlIQSleepScheduleResponse.SleepSchedule getSleepSchedule() {
073        return new ControlIQSleepScheduleResponse.SleepSchedule(rawSchedule, 0);
074    }
075
076    public int getFlag() {
077        return flag;
078    }
079}
