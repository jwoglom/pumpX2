001package com.jwoglom.pumpx2.pump.messages.response.currentStatus;
002
003import com.google.common.base.Preconditions;
004import com.jwoglom.pumpx2.pump.messages.helpers.Bytes;
005import com.jwoglom.pumpx2.pump.messages.Message;
006import com.jwoglom.pumpx2.pump.messages.MessageType;
007import com.jwoglom.pumpx2.pump.messages.annotations.MessageProps;
008import com.jwoglom.pumpx2.pump.messages.request.currentStatus.IDPSettingsRequest;
009
010/**
011 * Contains information about a given insulin delivery profile.
012 * Some information is specific to a time segment on the profile.
013 *
014 * @see com.jwoglom.pumpx2.pump.messages.response.currentStatus.IDPSegmentResponse
015 */
016@MessageProps(
017    opCode=65,
018    size=23,
019    type=MessageType.RESPONSE,
020    request=IDPSettingsRequest.class
021)
022public class IDPSettingsResponse extends Message {
023    
024    private int idpId;
025    private String name;
026    private int numberOfProfileSegments;
027    private int insulinDuration;
028    private int maxBolus;
029    private boolean carbEntry;
030    
031    public IDPSettingsResponse() {}
032    
033    public IDPSettingsResponse(int idpId, String name, int numberOfProfileSegments, int insulinDuration, int maxBolus, boolean carbEntry) {
034        this.cargo = buildCargo(idpId, name, numberOfProfileSegments, insulinDuration, maxBolus, carbEntry);
035        this.idpId = idpId;
036        this.name = name;
037        this.numberOfProfileSegments = numberOfProfileSegments;
038        this.insulinDuration = insulinDuration;
039        this.maxBolus = maxBolus;
040        this.carbEntry = carbEntry;
041        
042    }
043
044    public void parse(byte[] raw) {
045        Preconditions.checkArgument(raw.length == props().size());
046        this.cargo = raw;
047        this.idpId = raw[0];
048        this.name = Bytes.readString(raw, 1, 16);
049        this.numberOfProfileSegments = raw[17];
050        this.insulinDuration = Bytes.readShort(raw, 18);
051        this.maxBolus = Bytes.readShort(raw, 20);
052        this.carbEntry = raw[22] != 0;
053        
054    }
055
056    
057    public static byte[] buildCargo(int idpId, String name, int numberOfProfileSegments, int insulinDuration, int maxBolus, boolean carbEntry) {
058        return Bytes.combine(
059            new byte[]{ (byte) idpId },
060            Bytes.writeString(name, 16),
061            new byte[]{ (byte) numberOfProfileSegments },
062            Bytes.firstTwoBytesLittleEndian(insulinDuration), 
063            Bytes.firstTwoBytesLittleEndian(maxBolus), 
064            new byte[]{ (byte) (carbEntry ? 1 : 0) });
065    }
066    
067    public int getIdpId() {
068        return idpId;
069    }
070    public String getName() {
071        return name;
072    }
073    public int getNumberOfProfileSegments() {
074        return numberOfProfileSegments;
075    }
076    public int getInsulinDuration() {
077        return insulinDuration;
078    }
079    public int getMaxBolus() {
080        return maxBolus;
081    }
082    public boolean getCarbEntry() {
083        return carbEntry;
084    }
085    
086}
